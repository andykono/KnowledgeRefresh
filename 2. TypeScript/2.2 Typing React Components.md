# Typing React Components

[[2. TypeScript.md]]

---

Type React components: props, children, events, refs, forwardRef, and higher-order components.

## FC Type (FunctionComponent)

`React.FC` or `React.FunctionComponent` includes `children` implicitly. Many teams avoid it and type props explicitly for clarity.

```typescript
// With FC (children included by default)
const Greeting: React.FC<{ name: string }> = ({ name }) => {
  return <h1>Hello, {name}</h1>;
};

// Preferred: explicit props (no implicit children)
const GreetingExplicit = ({ name }: { name: string }) => {
  return <h1>Hello, {name}</h1>;
};
```

**Recommendation:** Use explicit props interfaces instead of `FC` for better control over `children`.

## PropsWithChildren

Use when a component explicitly accepts `children`.

```typescript
import { PropsWithChildren } from "react";

interface CardProps {
  title: string;
  variant?: "default" | "outlined";
}

// Option 1: PropsWithChildren
const Card = ({ title, variant = "default", children }: PropsWithChildren<CardProps>) => {
  return (
    <div className={`card card--${variant}`}>
      <h2>{title}</h2>
      {children}
    </div>
  );
};

// Option 2: Add children to your interface
interface CardPropsWithChildren extends CardProps {
  children: React.ReactNode;
}
```

## Typing Props

### Required and Optional
```typescript
interface ButtonProps {
  label: string;           // required
  onClick: () => void;     // required
  disabled?: boolean;      // optional
  variant?: "primary" | "secondary";  // optional with union
}

const Button = ({ label, onClick, disabled = false, variant = "primary" }: ButtonProps) => {
  return (
    <button onClick={onClick} disabled={disabled} className={`btn btn--${variant}`}>
      {label}
    </button>
  );
};
```

### Default Props
```typescript
interface InputProps {
  value: string;
  onChange: (value: string) => void;
  placeholder?: string;
  type?: "text" | "password" | "email";
}

const Input = ({ value, onChange, placeholder = "", type = "text" }: InputProps) => {
  return (
    <input
      type={type}
      value={value}
      onChange={(e) => onChange(e.target.value)}
      placeholder={placeholder}
    />
  );
};
```

## Typing Children

```typescript
// ReactNode: most permissive (elements, strings, numbers, fragments, etc.)
interface LayoutProps {
  children: React.ReactNode;
}

// ReactElement: only JSX elements
interface StrictLayoutProps {
  children: React.ReactElement;
}

// Specific element type
interface FormProps {
  children: React.ReactElement<InputProps> | React.ReactElement<InputProps>[];
}
```

## Typing Events

### MouseEvent, ChangeEvent, FormEvent, KeyboardEvent
```typescript
const Clickable = () => {
  const handleClick = (e: React.MouseEvent<HTMLButtonElement>) => {
    e.preventDefault();
    console.log(e.currentTarget); // HTMLButtonElement
  };

  return <button onClick={handleClick}>Click</button>;
};

const ControlledInput = () => {
  const handleChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    console.log(e.target.value); // string
  };

  return <input onChange={handleChange} />;
};

const Form = () => {
  const handleSubmit = (e: React.FormEvent<HTMLFormElement>) => {
    e.preventDefault();
    // form submission logic
  };

  return <form onSubmit={handleSubmit}>...</form>;
};

const KeyHandler = () => {
  const handleKeyDown = (e: React.KeyboardEvent<HTMLInputElement>) => {
    if (e.key === "Enter") {
      e.currentTarget.blur();
    }
  };

  return <input onKeyDown={handleKeyDown} />;
};
```

**Pattern:** `React.XxxEvent<HTMLElement>` — the generic is the element type.

## Typing Refs

```typescript
// useRef for DOM elements
const TextInput = () => {
  const inputRef = useRef<HTMLInputElement>(null);

  useEffect(() => {
    inputRef.current?.focus();
  }, []);

  return <input ref={inputRef} />;
};

// useRef for mutable values (not DOM)
const Timer = () => {
  const countRef = useRef<number>(0);
  countRef.current += 1; // OK, mutable
  return <span>{countRef.current}</span>;
};
```

## Typing forwardRef

Components that need to expose a ref to the parent.

```typescript
import { forwardRef } from "react";

interface InputProps {
  label: string;
  error?: string;
}

const Input = forwardRef<HTMLInputElement, InputProps>(
  ({ label, error, ...rest }, ref) => {
    return (
      <div>
        <label>{label}</label>
        <input ref={ref} {...rest} />
        {error && <span className="error">{error}</span>}
      </div>
    );
  }
);

// Usage
const Form = () => {
  const inputRef = useRef<HTMLInputElement>(null);
  return (
    <form>
      <Input ref={inputRef} label="Email" />
    </form>
  );
};
```

**Signature:** `forwardRef<RefType, PropsType>(component)`

## Typing HOCs (Higher-Order Components)

```typescript
// HOC that injects a prop
function withAuth<P extends object>(
  WrappedComponent: React.ComponentType<P & { isAuthenticated: boolean }>
) {
  return function WithAuthComponent(props: P) {
    const isAuthenticated = useAuth(); // assume this hook exists
    return <WrappedComponent {...props} isAuthenticated={isAuthenticated} />;
  };
}

// Usage
interface DashboardProps {
  isAuthenticated: boolean;
  userId: string;
}

const Dashboard = ({ isAuthenticated, userId }: DashboardProps) => {
  if (!isAuthenticated) return <Redirect to="/login" />;
  return <div>User: {userId}</div>;
};

const ProtectedDashboard = withAuth(Dashboard);
// ProtectedDashboard receives only { userId: string }
```

### Generic HOC with preserved props
```typescript
function withLoading<P extends object>(
  WrappedComponent: React.ComponentType<P>
): React.FC<P & { loading?: boolean }> {
  return function WithLoading({ loading, ...props }) {
    if (loading) return <Spinner />;
    return <WrappedComponent {...(props as P)} />;
  };
}
```

## Extending HTML Elements

Merge custom props with native HTML attributes.

```typescript
interface ButtonProps extends React.ButtonHTMLAttributes<HTMLButtonElement> {
  variant?: "primary" | "secondary";
  loading?: boolean;
}

const Button = ({ variant = "primary", loading, children, ...rest }: ButtonProps) => {
  return (
    <button className={`btn btn--${variant}`} disabled={loading} {...rest}>
      {loading ? "Loading..." : children}
    </button>
  );
};

// Full IntrinsicElements support
<Button onClick={() => {}} type="submit" variant="secondary">
  Submit
</Button>
```
