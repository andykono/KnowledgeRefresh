Build a type-safe API layer: typed fetch wrapper, generic API client, request/response types, error handling, Zod for runtime validation, and type-safe route params.

## Typed Fetch Wrapper

Wrap `fetch` to return typed responses and handle errors.

```typescript
interface FetchOptions extends RequestInit {
  params?: Record<string, string>;
}

async function typedFetch<T>(
  url: string,
  options: FetchOptions = {}
): Promise<T> {
  const { params, ...init } = options;
  const searchUrl = params
    ? `${url}?${new URLSearchParams(params).toString()}`
    : url;

  const response = await fetch(searchUrl, init);

  if (!response.ok) {
    throw new ApiError(response.status, await response.text());
  }

  const data = await response.json();
  return data as T;
}

// Usage
interface User {
  id: number;
  name: string;
  email: string;
}

const user = await typedFetch<User>("/api/users/1");
// user is User
```

## Generic API Client

Encapsulate base URL, headers, and error handling.

```typescript
class ApiClient {
  constructor(
    private baseUrl: string,
    private defaultHeaders: Record<string, string> = {}
  ) {}

  private async request<T>(
    method: string,
    path: string,
    body?: unknown
  ): Promise<T> {
    const url = `${this.baseUrl}${path}`;
    const res = await fetch(url, {
      method,
      headers: {
        "Content-Type": "application/json",
        ...this.defaultHeaders,
      },
      body: body ? JSON.stringify(body) : undefined,
    });

    if (!res.ok) {
      throw new ApiError(res.status, await res.text());
    }

    return res.json() as Promise<T>;
  }

  async get<T>(path: string): Promise<T> {
    return this.request<T>("GET", path);
  }

  async post<T, B>(path: string, body: B): Promise<T> {
    return this.request<T>("POST", path, body);
  }

  async put<T, B>(path: string, body: B): Promise<T> {
    return this.request<T>("PUT", path, body);
  }

  async delete<T>(path: string): Promise<T> {
    return this.request<T>("DELETE", path);
  }
}

const api = new ApiClient("https://api.example.com", {
  Authorization: `Bearer ${token}`,
});

const users = await api.get<User[]>("/users");
const newUser = await api.post<User, CreateUserDto>("/users", { name: "Alice", email: "a@b.com" });
```

## Typing Request and Response

Define explicit interfaces for each endpoint.

```typescript
// Request types
interface CreateUserRequest {
  name: string;
  email: string;
  password: string;
}

interface UpdateUserRequest {
  name?: string;
  email?: string;
}

// Response types
interface UserResponse {
  id: number;
  name: string;
  email: string;
  createdAt: string;
}

interface PaginatedResponse<T> {
  data: T[];
  total: number;
  page: number;
  pageSize: number;
}

// Typed endpoints
const api = {
  createUser: (data: CreateUserRequest) =>
    typedFetch<UserResponse>("/api/users", { method: "POST", body: JSON.stringify(data) }),
  getUsers: (page = 1) =>
    typedFetch<PaginatedResponse<UserResponse>>(`/api/users?page=${page}`),
  updateUser: (id: number, data: UpdateUserRequest) =>
    typedFetch<UserResponse>(`/api/users/${id}`, { method: "PUT", body: JSON.stringify(data) }),
};
```

## Error Types

```typescript
class ApiError extends Error {
  constructor(
    public status: number,
    message: string,
    public body?: unknown
  ) {
    super(message);
    this.name = "ApiError";
  }

  get is404(): boolean {
    return this.status === 404;
  }

  get is5xx(): boolean {
    return this.status >= 500;
  }
}

// Typed error handling
type ApiResult<T> = { success: true; data: T } | { success: false; error: ApiError };

async function safeFetch<T>(url: string): Promise<ApiResult<T>> {
  try {
    const data = await typedFetch<T>(url);
    return { success: true, data };
  } catch (e) {
    return {
      success: false,
      error: e instanceof ApiError ? e : new ApiError(0, String(e)),
    };
  }
}
```

## Zod for Runtime Validation + Static Types

Zod validates at runtime and infers TypeScript types.

```typescript
import { z } from "zod";

// Schema
const UserSchema = z.object({
  id: z.number(),
  name: z.string(),
  email: z.string().email(),
  createdAt: z.string().datetime(),
});

// Inferred type
type User = z.infer<typeof UserSchema>;

// Validate and parse
async function fetchUser(id: number): Promise<User> {
  const res = await fetch(`/api/users/${id}`);
  const json = await res.json();
  return UserSchema.parse(json);  // throws if invalid
}

// Safe parse (returns result)
const result = UserSchema.safeParse(json);
if (result.success) {
  const user = result.data;  // User
} else {
  console.error(result.error.errors);
}

// Partial for updates
const UpdateUserSchema = UserSchema.partial().pick({ name: true, email: true });
type UpdateUser = z.infer<typeof UpdateUserSchema>;
```

## Type-Safe Route Params

Define routes and params so they stay in sync.

```typescript
// Route definitions
const routes = {
  "/users": {},
  "/users/:id": { id: "string" },
  "/users/:id/posts": { id: "string" },
  "/users/:id/posts/:postId": { id: "string"; postId: "string" },
} as const;

type RoutePath = keyof typeof routes;

// Extract params from path
type ParamsForPath<P extends RoutePath> = (typeof routes)[P];

type RouteParams = {
  "/users": Record<string, never>;
  "/users/:id": { id: string };
  "/users/:id/posts": { id: string };
  "/users/:id/posts/:postId": { id: string; postId: string };
};

function buildPath<P extends RoutePath>(
  path: P,
  params: ParamsForPath<P> extends Record<string, never> ? void : ParamsForPath<P>
): string {
  if (!params || Object.keys(params).length === 0) return path;
  return path.replace(/:(\w+)/g, (_, key) => (params as Record<string, string>)[key] ?? "");
}

buildPath("/users");                    // OK
buildPath("/users/:id", { id: "1" });   // OK
buildPath("/users/:id", { id: 1 });     // Error: id should be string
buildPath("/users/:id", {});            // Error: id required
```

## React Query Integration

```typescript
import { useQuery, useMutation, useQueryClient } from "@tanstack/react-query";

const userKeys = {
  all: ["users"] as const,
  detail: (id: number) => ["users", id] as const,
};

function useUser(id: number) {
  return useQuery({
    queryKey: userKeys.detail(id),
    queryFn: () => typedFetch<User>(`/api/users/${id}`),
  });
}

function useCreateUser() {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: (data: CreateUserRequest) =>
      typedFetch<User>("/api/users", { method: "POST", body: JSON.stringify(data) }),
    onSuccess: () => queryClient.invalidateQueries({ queryKey: userKeys.all }),
  });
}
```
