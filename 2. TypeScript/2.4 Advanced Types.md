# Advanced Types

[[2. TypeScript.md]]

---

Advanced TypeScript: conditional types, mapped types, template literal types, infer, discriminated unions, branded types, generic constraints, and function overloads.

## Conditional Types

`T extends U ? X : Y` — choose a type based on a condition.

```typescript
// Basic conditional
type IsString<T> = T extends string ? true : false;
type A = IsString<"hello">;  // true
type B = IsString<number>;   // false

// Extract array element type
type ExtractElement<T> = T extends (infer E)[] ? E : never;
type Num = ExtractElement<number[]>;   // number
type Str = ExtractElement<string[]>;   // string

// Flatten array type
type Flatten<T> = T extends (infer E)[] ? E : T;
type Flat1 = Flatten<string[]>;  // string
type Flat2 = Flatten<number>;    // number

// Nullable to non-nullable
type NonNullable<T> = T extends null | undefined ? never : T;
type Safe = NonNullable<string | null>;  // string
```

## Mapped Types

Transform each property of a type.

```typescript
// Make all properties optional
type Partial<T> = {
  [P in keyof T]?: T[P];
};

// Make all properties readonly
type Readonly<T> = {
  readonly [P in keyof T]: T[P];
};

// Add prefix to keys
type Getters<T> = {
  [P in keyof T as `get${Capitalize<string & P>}`]: () => T[P];
};
interface User {
  name: string;
  age: number;
}
type UserGetters = Getters<User>;
// { getName: () => string; getAge: () => number }

// Nullable values
type Nullable<T> = {
  [P in keyof T]: T[P] | null;
};
```

## Template Literal Types

Build string types from template literals.

```typescript
type EventName = "click" | "scroll" | "focus";
type HandlerName = `on${Capitalize<EventName>}`;
// "onClick" | "onScroll" | "onFocus"

// API route pattern
type HttpMethod = "GET" | "POST" | "PUT" | "DELETE";
type ApiRoute = `/api/${string}`;
type FullRoute = `${HttpMethod} ${ApiRoute}`;
// "GET /api/..." | "POST /api/..." | etc.

// CSS property
type CssProp = "margin" | "padding";
type CssSide = "Top" | "Right" | "Bottom" | "Left";
type CssProperty = `${CssProp}${CssSide}`;
// "marginTop" | "marginRight" | "marginBottom" | "marginLeft" | "paddingTop" | ...
```

## infer Keyword

Extract a type from within another type.

```typescript
// Extract return type of a function
type ReturnType<T> = T extends (...args: never[]) => infer R ? R : never;
type FnReturn = ReturnType<() => number>;  // number

// Extract first argument
type FirstArg<T> = T extends (arg: infer A, ...args: never[]) => unknown ? A : never;
type Arg = FirstArg<(x: string) => void>;  // string

// Extract promise type
type Awaited<T> = T extends Promise<infer U> ? U : T;
type Res = Awaited<Promise<User>>;  // User

// Extract array element
type ArrayElement<T> = T extends (infer E)[] ? E : never;
```

## Discriminated Unions for State Machines

Model state machines with a union type and a discriminant field.

```typescript
type AsyncState<T> =
  | { status: "idle" }
  | { status: "loading" }
  | { status: "success"; data: T }
  | { status: "error"; error: Error };

function render<T>(state: AsyncState<T>) {
  switch (state.status) {
    case "idle":
      return "Waiting...";
    case "loading":
      return "Loading...";
    case "success":
      return state.data;  // T
    case "error":
      return state.error.message;
  }
}

// Form wizard
type WizardState =
  | { step: "personal"; name: string; email: string }
  | { step: "address"; street: string; city: string }
  | { step: "confirm"; agreed: boolean };

function nextStep(state: WizardState): WizardState {
  switch (state.step) {
    case "personal":
      return { step: "address", street: "", city: "" };
    case "address":
      return { step: "confirm", agreed: false };
    case "confirm":
      return state;
  }
}
```

## Branded Types

Create nominal types to prevent mixing primitives that share the same shape.

```typescript
// Brand: nominal typing for primitives
type UserId = string & { readonly __brand: "UserId" };
type OrderId = string & { readonly __brand: "OrderId" };

function createUserId(id: string): UserId {
  return id as UserId;
}

function getOrder(id: OrderId) { /* ... */ }

const uid = createUserId("user-123");
getOrder(uid);  // Error: UserId not assignable to OrderId

// Branded object
interface Branded<T extends string> {
  readonly __brand: T;
}
type BrandedUserId = string & Branded<"UserId">;
type BrandedOrderId = string & Branded<"OrderId">;
```

## Generic Constraints

Restrict what types a generic can accept.

```typescript
// extends keyof
function get<T, K extends keyof T>(obj: T, key: K): T[K] {
  return obj[key];
}

// extends object
function merge<T extends object, U extends object>(a: T, b: U): T & U {
  return { ...a, ...b };
}

// extends with conditional
type RequireAtLeastOne<T, Keys extends keyof T = keyof T> = Pick<T, Exclude<keyof T, Keys>> &
  { [K in Keys]-?: Required<Pick<T, K>> & Partial<Pick<T, Exclude<Keys, K>>> }[Keys];

// Multiple constraints
function logAndReturn<T extends { id: string } & { name: string }>(item: T): T {
  console.log(item.id, item.name);
  return item;
}
```

## Function Overloads

Define multiple call signatures for one implementation.

```typescript
function createElement(tag: "div"): HTMLDivElement;
function createElement(tag: "span"): HTMLSpanElement;
function createElement(tag: "input"): HTMLInputElement;
function createElement(tag: string): HTMLElement {
  return document.createElement(tag) as HTMLElement;
}

const div = createElement("div");   // HTMLDivElement
const span = createElement("span"); // HTMLSpanElement

// API with different shapes
function fetchData<T>(url: string): Promise<T>;
function fetchData(url: string, asArray: true): Promise<unknown[]>;
function fetchData<T>(url: string, asArray?: boolean): Promise<T | unknown[]> {
  return fetch(url).then((r) => r.json()) as Promise<T | unknown[]>;
}

const users = await fetchData<User[]>("/api/users");
const raw = await fetchData("/api/users", true);
```
