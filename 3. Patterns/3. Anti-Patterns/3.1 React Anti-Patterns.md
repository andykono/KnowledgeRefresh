# React Anti-Patterns

[[3. Anti-Patterns.md]]

Специфичные для React ошибки и плохие практики.

## 1. Missing useEffect Dependencies

### ❌ Bad
```typescript
function UserProfile({ userId }: Props) {
  const [user, setUser] = useState(null);

  // BUG: userId not in dependency array
  // If userId changes, this will still fetch OLD userId!
  useEffect(() => {
    api.getUser(userId).then(setUser);
  }, []);

  return <div>{user?.name}</div>;
}
```

### ✅ Good
```typescript
function UserProfile({ userId }: Props) {
  const [user, setUser] = useState(null);

  // Correctly includes userId in dependencies
  useEffect(() => {
    api.getUser(userId).then(setUser);
  }, [userId]);

  return <div>{user?.name}</div>;
}
```

### ESLint Rule
Enable `eslint-plugin-react-hooks`:
```json
{
  "extends": ["plugin:react-hooks/recommended"]
}
```

## 2. Memory Leaks in useEffect

### ❌ Bad
```typescript
function ChatComponent() {
  useEffect(() => {
    // Subscribe to websocket
    const ws = new WebSocket('ws://...');
    ws.onmessage = (event) => {
      console.log(event.data);
    };
    // NO CLEANUP! → ws keeps running after unmount
  }, []);

  return <div>Chat...</div>;
}
```

### ✅ Good
```typescript
function ChatComponent() {
  useEffect(() => {
    const ws = new WebSocket('ws://...');
    ws.onmessage = (event) => {
      console.log(event.data);
    };

    // Cleanup function
    return () => {
      ws.close();
    };
  }, []);

  return <div>Chat...</div>;
}
```

### Common Cleanup Scenarios
```typescript
// Timer
useEffect(() => {
  const timer = setTimeout(() => { /* ... */ }, 1000);
  return () => clearTimeout(timer);
}, []);

// Interval
useEffect(() => {
  const interval = setInterval(() => { /* ... */ }, 1000);
  return () => clearInterval(interval);
}, []);

// Event listener
useEffect(() => {
  window.addEventListener('resize', handleResize);
  return () => window.removeEventListener('resize', handleResize);
}, []);

// Subscription
useEffect(() => {
  const unsubscribe = eventBus.subscribe(callback);
  return () => unsubscribe();
}, []);
```

## 3. Mutating State Directly

### ❌ Bad
```typescript
const [user, setUser] = useState({ name: 'John', age: 30 });

function updateName(newName: string) {
  user.name = newName; // MUTATING STATE!
  setUser(user); // React won't detect change - same reference
}

function addHobby(hobby: string) {
  user.hobbies = [...user.hobbies, hobby]; // MUTATING ARRAY!
  setUser(user);
}
```

### ✅ Good
```typescript
const [user, setUser] = useState({ name: 'John', age: 30 });

function updateName(newName: string) {
  setUser({ ...user, name: newName }); // New object
}

function addHobby(hobby: string) {
  setUser({
    ...user,
    hobbies: [...user.hobbies, hobby], // New array
  });
}
```

### Immer for Complex Updates
```typescript
import produce from 'immer';

const [user, setUser] = useState({
  profile: {
    name: 'John',
    hobbies: ['reading'],
  },
});

// Complex nested update
setUser(produce(user => {
  user.profile.name = 'Jane';
  user.profile.hobbies.push('coding');
}));
```

## 4. Prop Drilling

### ❌ Bad
```typescript
function App() {
  const [user, setUser] = useState(null);

  return <Page user={user} />;
}

function Page({ user }: Props) {
  return <Header user={user} />;
}

function Header({ user }: Props) {
  return <Nav user={user} />;
}

function Nav({ user }: Props) {
  return <UserMenu user={user} />;
}

function UserMenu({ user }: Props) {
  return <div>{user?.name}</div>;
}
```

### ✅ Good
```typescript
function App() {
  const [user, setUser] = useState(null);

  return (
    <UserContext.Provider value={user}>
      <Page />
    </UserContext.Provider>
  );
}

function Page() {
  return <Header />; // No props!
}

function Header() {
  return <Nav />;
}

function Nav() {
  return <UserMenu />;
}

function UserMenu() {
  const user = useContext(UserContext);
  return <div>{user?.name}</div>;
}
```

## 5. useCallback/useMemo Overuse

### ❌ Bad
```typescript
function SearchUsers() {
  // Every single function wrapped in useCallback!
  const handleChange = useCallback((e) => {
    // ...
  }, []);

  const handleSubmit = useCallback((e) => {
    // ...
  }, []);

  const renderUser = useCallback((user) => {
    // ...
  }, []);

  // Every object wrapped in useMemo!
  const config = useMemo(() => ({
    timeout: 5000,
  }), []);
}
```

### ✅ Good
```typescript
function SearchUsers() {
  // Simple functions don't need useCallback
  const handleChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    // ...
  };

  const handleSubmit = (e: React.FormEvent) => {
    // ...
  };

  // Only memoize if passing to optimized child or complex dependency
  const renderUser = useCallback((user: User) => {
    return <UserCard user={user} />;
  }, []);

  // Don't memoize constants
  const config = { timeout: 5000 };
}
```

### When to Use
```typescript
// ✅ useCallback: Passing to memoized child or effect dependency
const onClick = useCallback(() => handleDelete(), [handleDelete]);
<MemoizedButton onClick={onClick} />

// ✅ useMemo: Expensive calculation
const sortedUsers = useMemo(() => {
  return users.sort((a, b) => a.name.localeCompare(b.name));
}, [users]);

// ❌ useMemo: Simple objects/arrays (just recreate)
const config = { timeout: 5000 }; // Fine, no need to memoize
const colors = ['red', 'blue']; // Fine, no need to memoize
```

## 6. Conditional Hooks

### ❌ Bad
```typescript
function Component({ fetchData }: Props) {
  // ILLEGAL: Calling hook conditionally!
  if (fetchData) {
    const data = useAsync(() => api.getData());
  }

  return <div></div>;
}
```

### ✅ Good
```typescript
function Component({ fetchData }: Props) {
  // Always call hook, control behavior inside
  const { data, loading } = useAsync(
    fetchData ? () => api.getData() : () => Promise.resolve(null)
  );

  if (!fetchData) return <div>Disabled</div>;

  return <div>{data}</div>;
}
```

## 7. Inline Object/Function Props

### ❌ Bad
```typescript
function Parent() {
  return (
    // New object created every render!
    <Child config={{ timeout: 5000, retry: 3 }} />
  );
}

function Child({ config }: Props) {
  useEffect(() => {
    // This dependency is constantly changing
  }, [config]); // config is new object every render!
}
```

### ✅ Good
```typescript
function Parent() {
  const config = useMemo(() => ({ timeout: 5000, retry: 3 }), []);

  return <Child config={config} />;
}
```

## 8. Infinite Loops in useEffect

### ❌ Bad
```typescript
// INFINITE LOOP!
function Component() {
  const [data, setData] = useState([]);

  useEffect(() => {
    fetchData().then(newData => {
      // setData creates new array reference
      // which is added to dependency array
      // which triggers effect again...
      setData(newData);
    });
  }, [data]); // BUG: data in dependencies causes infinite loop

  return <div>{data.length}</div>;
}
```

### ✅ Good
```typescript
function Component() {
  const [data, setData] = useState([]);

  useEffect(() => {
    fetchData().then(setData); // Just pass setter
  }, []); // Empty deps - fetch once

  return <div>{data.length}</div>;
}
```

## 9. Not Using Key Prop

### ❌ Bad
```typescript
function UserList({ users }: Props) {
  // Without key, React reuses DOM nodes
  // Can cause state corruption and bugs
  return (
    <ul>
      {users.map(user => (
        <li>{user.name}</li>
      ))}
    </ul>
  );
}
```

### ✅ Good
```typescript
function UserList({ users }: Props) {
  return (
    <ul>
      {users.map(user => (
        <li key={user.id}>{user.name}</li> {/* Unique, stable key */}
      ))}
    </ul>
  );
}
```

### Key Rules
```typescript
// ✅ Good keys
<li key={user.id}>...</li>        // Stable ID
<li key={`user-${user.id}`}>...</li> // Prefixed ID
<li key={uuid()}>...</li>         // Generated once (not recommended but ok)

// ❌ Bad keys
<li key={index}>...</li>          // Index changes when list reorders
<li key={Math.random()}>...</li>  // New key every render!
<li key={user.name}>...</li>      // Not unique if names repeat
```

## 10. Forgetting setState Callback

### ❌ Bad (Old Class Component Pattern)
```typescript
function Component() {
  const [count, setCount] = useState(0);

  const increment = () => {
    setCount(count + 1);
    console.log(count); // Still logs old value!
  };

  return <button onClick={increment}>Count: {count}</button>;
}
```

### ✅ Good
```typescript
function Component() {
  const [count, setCount] = useState(0);

  useEffect(() => {
    console.log(count); // Logs new value after state update
  }, [count]);

  const increment = () => {
    setCount(c => c + 1); // Use updater function
  };

  return <button onClick={increment}>Count: {count}</button>;
}
```

## 11. Using Array Index as Key in Dynamic Lists

### ❌ Bad
```typescript
function TodoList() {
  const [todos, setTodos] = useState(['item1', 'item2']);

  return (
    <ul>
      {todos.map((todo, index) => (
        <li key={index}>{todo}</li> // BUG: If you reorder, keys change!
      ))}
    </ul>
  );
}

// User reorders items:
// Before: ['item1', 'item2']
// After: ['item2', 'item1']
// Keys 0 and 1 now point to DIFFERENT items!
```

### ✅ Good
```typescript
interface Todo {
  id: string;
  text: string;
}

function TodoList() {
  const [todos, setTodos] = useState<Todo[]>([
    { id: 'a', text: 'item1' },
    { id: 'b', text: 'item2' },
  ]);

  return (
    <ul>
      {todos.map(todo => (
        <li key={todo.id}>{todo.text}</li> // Stable ID
      ))}
    </ul>
  );
}
```

## 12. Storing Derived State

### ❌ Bad
```typescript
function Component({ firstName, lastName }: Props) {
  // Derived state - will get out of sync!
  const [fullName, setFullName] = useState(`${firstName} ${lastName}`);

  // If props change, fullName doesn't update automatically
  // Must manually sync with useEffect (extra code!)
  useEffect(() => {
    setFullName(`${firstName} ${lastName}`);
  }, [firstName, lastName]);

  return <div>{fullName}</div>;
}
```

### ✅ Good
```typescript
function Component({ firstName, lastName }: Props) {
  // Calculate on the fly - always in sync
  const fullName = `${firstName} ${lastName}`;

  return <div>{fullName}</div>;
}
```

## Best Practices Checklist

- [ ] All useEffect dependencies included
- [ ] Cleanup functions for subscriptions/timers
- [ ] Never mutate state - always create new objects
- [ ] Use Context to avoid prop drilling
- [ ] Only memoize when necessary (actual performance issue)
- [ ] Never call hooks conditionally
- [ ] Always provide stable key in lists
- [ ] No console.log in production
- [ ] Use React DevTools to find performance issues
- [ ] Enable Strict Mode in development
