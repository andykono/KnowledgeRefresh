# State Management Anti-Patterns

[[3. Anti-Patterns.md]]

Ошибки при управлении состоянием приложения.

---

## 1. Global Mutable State

### ❌ Bad
```typescript
// Global mutable object
let appState = {
  user: null,
  notifications: [],
};

// Anyone can modify directly!
appState.user = newUser;
appState.notifications.push(notification);

// No way to track changes or debug
```

### ✅ Good
```typescript
// Use Context + useState
const AppContext = createContext();

function AppProvider({ children }) {
  const [state, dispatch] = useReducer(reducer, initialState);

  return (
    <AppContext.Provider value={{ state, dispatch }}>
      {children}
    </AppContext.Provider>
  );
}
```

## 2. Overly Nested Redux State

### ❌ Bad
```typescript
const state = {
  entities: {
    users: {
      byId: {
        '1': { id: '1', profile: { name: 'John', details: { age: 30 } } },
      },
      allIds: ['1'],
    },
  },
};

// Accessing nested value:
const age = state.entities.users.byId['1'].profile.details.age; // Too deep!

// Updating:
state.entities.users.byId['1'].profile.details.age = 31; // Painful!
```

### ✅ Good
```typescript
// Flatten structure
const state = {
  users: {
    '1': { id: '1', name: 'John', age: 30 },
  },
};

// Simple access
const age = state.users['1'].age; // Simple!

// Use selector functions
const selectUserAge = (state, userId) => state.users[userId].age;
```

## 3. Over-Fetching Data

### ❌ Bad
```typescript
// Fetching ALL users on page load
async function fetchUsers() {
  const users = await api.get('/users'); // 10,000 users!
  return users;
}

// Storing ALL data
const [allUsers, setAllUsers] = useState([]);
```

### ✅ Good
```typescript
// Paginate or lazy load
async function fetchUsers(page: number, limit: number = 20) {
  return api.get(`/users?page=${page}&limit=${limit}`);
}

// Or use infinite scroll
const { data: users, fetchMore } = useInfiniteQuery(
  ['users'],
  ({ pageParam = 0 }) => fetchUsers(pageParam)
);
```

## 4. Circular Dependencies

### ❌ Bad
```typescript
// Module A imports from Module B
// Module B imports from Module A
// Circular dependency!

// a.ts
import { createBFromA } from './b';
export function createA() { return createBFromA(); }

// b.ts
import { createA } from './a';
export function createBFromA() { return createA(); }
```

### ✅ Good
```typescript
// Extract common logic
// common.ts
export function createBase() { /* ... */ }

// a.ts
import { createBase } from './common';
export function createA() { return createBase(); }

// b.ts
import { createBase } from './common';
export function createB() { return createBase(); }
```

## 5. Storing API Responses as-is

### ❌ Bad
```typescript
// Store API response directly
const [userData, setUserData] = useState(null);

useEffect(() => {
  api.getUser().then(setUserData); // Store raw response
}, []);

// Now component tightly coupled to API structure
// If API changes, component breaks
<span>{userData.data.user.profile.name}</span>
```

### ✅ Good
```typescript
// Normalize and transform
interface AppUser {
  id: string;
  name: string;
  email: string;
}

interface ApiResponse {
  data: { user: { id: string; fullName: string; emailAddress: string } };
}

function normalizeUser(apiData: ApiResponse): AppUser {
  return {
    id: apiData.data.user.id,
    name: apiData.data.user.fullName,
    email: apiData.data.user.emailAddress,
  };
}

const [user, setUser] = useState<AppUser | null>(null);

useEffect(() => {
  api.getUser().then(response => {
    setUser(normalizeUser(response));
  });
}, []);

// Now component uses consistent structure
<span>{user.name}</span>
```

## 6. Updating State Without Immutability

### ❌ Bad
```typescript
// Redux reducer with mutation
const reducer = (state, action) => {
  if (action.type === 'ADD_TODO') {
    state.todos.push(action.payload); // MUTATING!
    return state; // Same reference - Redux won't detect
  }
};
```

### ✅ Good
```typescript
// Redux reducer immutable
const reducer = (state, action) => {
  if (action.type === 'ADD_TODO') {
    return {
      ...state,
      todos: [...state.todos, action.payload],
    };
  }
};
```

## 7. Tight Coupling Between State and UI

### ❌ Bad
```typescript
// State mirrors UI state
const [selectedTab, setSelectedTab] = useState(0);
const [filteredUsers, setFilteredUsers] = useState([]);
const [sortedUsers, setSortedUsers] = useState([]);
const [isLoading, setIsLoading] = useState(false);

// Multiple state updates for related changes
// Easy to get out of sync
```

### ✅ Good
```typescript
// Derive derived state, keep single source of truth
const [users, setUsers] = useState<User[]>([]);
const [selectedTab, setSelectedTab] = useState(0);

// Compute derived values
const filteredUsers = users.filter(/* ... */);
const sortedUsers = filteredUsers.sort(/* ... */);

// Use loading state globally
const [loading, setLoading] = useState(false);
```

## 8. Storing User Input Before Validation

### ❌ Bad
```typescript
// Store unvalidated input
const [email, setEmail] = useState('');

<input
  value={email}
  onChange={(e) => setEmail(e.target.value)} // Any value accepted
/>

// Validate on submit
const handleSubmit = () => {
  if (!isValidEmail(email)) {
    setError('Invalid email');
  }
};
```

### ✅ Good
```typescript
// Use form library with validation
import { useForm } from 'react-hook-form';

const { register, watch, formState } = useForm({
  defaultValues: { email: '' },
  mode: 'onChange', // Validate on change
});

<input
  {...register('email', {
    required: 'Email required',
    pattern: { value: /^.+@.+\..+$/, message: 'Invalid email' },
  })}
/>

{formState.errors.email && <span>{formState.errors.email.message}</span>}
```

## 9. Not Normalizing State

### ❌ Bad
```typescript
// Denormalized: same data in multiple places
const state = {
  users: {
    '1': { id: '1', name: 'John', posts: [{ id: 'p1', title: 'Post 1' }] },
  },
  posts: {
    'p1': { id: 'p1', title: 'Post 1', author: { id: '1', name: 'John' } },
  },
};

// If John's name changes, must update in 2 places!
// Easy to get out of sync
```

### ✅ Good
```typescript
// Normalized: single source of truth for each entity
const state = {
  users: {
    '1': { id: '1', name: 'John' },
  },
  posts: {
    'p1': { id: 'p1', title: 'Post 1', authorId: '1' },
  },
  userPosts: {
    '1': ['p1'], // User 1 has post p1
  },
};

// Use selectors to reconstruct data
const selectUserWithPosts = (state, userId) => ({
  ...state.users[userId],
  posts: state.userPosts[userId].map(postId => state.posts[postId]),
});
```

## 10. Not Using Selectors

### ❌ Bad
```typescript
// Component accesses deeply nested state
const user = store.getState().app.entities.users.byId['1'];
const posts = store.getState().app.entities.posts.allIds
  .map(id => store.getState().app.entities.posts.byId[id])
  .filter(p => p.authorId === '1');
```

### ✅ Good
```typescript
// Use selector functions
const selectUser = (state, userId) => state.users[userId];
const selectUserPosts = (state, userId) => {
  return state.userPosts[userId]
    .map(postId => state.posts[postId]);
};

// Memoize selectors for performance
import { createSelector } from 'reselect';

const selectUserPosts = createSelector(
  [(state, userId) => state.posts, (state, userId) => state.userPosts[userId]],
  (posts, postIds) => postIds.map(id => posts[id])
);

// Component
const user = useSelector(state => selectUser(state, userId));
const posts = useSelector(state => selectUserPosts(state, userId));
```

## Best Practices Checklist

- [ ] No global mutable state
- [ ] Immutable updates only
- [ ] Flat, normalized structure
- [ ] Single source of truth per entity
- [ ] Use selector functions
- [ ] Memoize derived selectors
- [ ] Transform API responses
- [ ] Keep UI state separate from domain state
- [ ] Validate on input, not storage
- [ ] Use DevTools (Redux, Zustand, etc)
