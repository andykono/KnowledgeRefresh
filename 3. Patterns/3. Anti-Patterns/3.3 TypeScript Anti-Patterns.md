# TypeScript Anti-Patterns

[[3. Anti-Patterns.md]]

Ошибки при использовании TypeScript.

---

## 1. Using `any` Type

### ❌ Bad
```typescript
// Loses all type safety
function handleData(data: any) {
  data.forEach(x => {
    // No autocomplete, no type checking!
    console.log(x.nonExistentProperty);
  });
}

// Can pass anything
handleData('not an array'); // No error!
```

### ✅ Good
```typescript
// Specify actual type
function handleData(data: unknown) {
  if (Array.isArray(data)) {
    data.forEach(x => {
      // Now x is typed
      console.log((x as any).property); // Only cast when necessary
    });
  }
}

// Or use generics
function handleData<T>(data: T[]) {
  data.forEach(x => {
    console.log(x); // T is properly typed
  });
}
```

## 2. Over-General Types

### ❌ Bad
```typescript
interface User {
  id: any;
  name: any;
  email: any;
  createdAt: any;
}

// Might as well not use TypeScript!
```

### ✅ Good
```typescript
interface User {
  id: string;
  name: string;
  email: string;
  createdAt: Date;
}

// Or use literal types for restricted values
interface UserRole {
  role: 'admin' | 'user' | 'guest';
  permissions: readonly string[]; // Readonly array
}
```

## 3. Not Using `readonly`

### ❌ Bad
```typescript
interface User {
  id: string;
  name: string;
  permissions: string[];
}

function removePermission(user: User, perm: string) {
  user.permissions.splice(user.permissions.indexOf(perm), 1); // Mutating!
}
```

### ✅ Good
```typescript
interface User {
  readonly id: string;
  readonly name: string;
  readonly permissions: readonly string[];
}

function removePermission(user: User, perm: string): string[] {
  // Can't mutate, must return new array
  return user.permissions.filter(p => p !== perm);
}
```

## 4. Missing Type Annotations

### ❌ Bad
```typescript
// Type inference can be wrong
const user = { id: 1, name: 'John' };
// TypeScript infers: { id: number; name: string }
// But if you meant id: string, too late!

// Functions without return types
function processUser(user: User) {
  return user.name.toUpperCase(); // Inferred as string
}
```

### ✅ Good
```typescript
// Explicit types
const user: User = { id: '1', name: 'John' };

// Return types explicit
function processUser(user: User): string {
  return user.name.toUpperCase();
}

// Complex return types
function fetchUser(id: string): Promise<User> {
  return api.getUser(id);
}
```

## 5. Not Using Literal Types

### ❌ Bad
```typescript
interface Response {
  status: string; // Could be "success", "error", "pending"...
}

// Later...
if (response.status === 'sukcess') { // Typo, but no error!
  // ...
}
```

### ✅ Good
```typescript
interface Response {
  status: 'success' | 'error' | 'pending';
}

// Now typo is caught
if (response.status === 'sukcess') { // ERROR: not a valid status
  // ...
}
```

## 6. Poor Union Type Handling

### ❌ Bad
```typescript
interface Cat {
  type: 'cat';
  meow: string;
}

interface Dog {
  type: 'dog';
  bark: string;
}

type Animal = Cat | Dog;

function getSound(animal: Animal): string {
  // TypeScript doesn't know which type!
  // @ts-ignore to make it compile
  return animal.meow || animal.bark;
}
```

### ✅ Good
```typescript
function getSound(animal: Animal): string {
  if (animal.type === 'cat') {
    return animal.meow; // TypeScript knows it's Cat
  } else {
    return animal.bark; // TypeScript knows it's Dog
  }
}

// Or use discriminated unions
function getSound(animal: Animal): string {
  switch (animal.type) {
    case 'cat':
      return animal.meow;
    case 'dog':
      return animal.bark;
  }
}
```

## 7. Not Using `unknown` for Unsafe Values

### ❌ Bad
```typescript
// Parsing JSON without type safety
const data: any = JSON.parse(jsonString);
const name = data.user.name; // Could be undefined!
```

### ✅ Good
```typescript
// Use unknown and guard
const data: unknown = JSON.parse(jsonString);

if (
  typeof data === 'object' &&
  data !== null &&
  'user' in data &&
  'name' in (data as any).user
) {
  const name = (data as any).user.name;
}

// Or use a validation library
import { z } from 'zod';

const UserSchema = z.object({
  user: z.object({
    name: z.string(),
  }),
});

const data = UserSchema.parse(JSON.parse(jsonString));
const name = data.user.name; // Fully typed and validated!
```

## 8. Not Using Generics

### ❌ Bad
```typescript
// Repeated patterns
interface ApiResponseUser {
  data: User | null;
  error: Error | null;
  loading: boolean;
}

interface ApiResponsePost {
  data: Post | null;
  error: Error | null;
  loading: boolean;
}

interface ApiResponseComment {
  data: Comment | null;
  error: Error | null;
  loading: boolean;
}
```

### ✅ Good
```typescript
// Generic version
interface ApiResponse<T> {
  data: T | null;
  error: Error | null;
  loading: boolean;
}

type ApiResponseUser = ApiResponse<User>;
type ApiResponsePost = ApiResponse<Post>;
type ApiResponseComment = ApiResponse<Comment>;

// Or use directly
function useApi<T>(url: string): ApiResponse<T> {
  // ...
}
```

## 9. Overly Complex Types

### ❌ Bad
```typescript
// Hard to understand
type DeepPartial<T> = T extends object
  ? {
      [P in keyof T]?: DeepPartial<T[P]>;
    }
  : T;

// vs
type User = {
  name?: string;
  age?: number;
  address?: {
    street?: string;
    city?: string;
  };
};
```

### ✅ Good
```typescript
// Keep types readable
type Optional<T> = {
  [K in keyof T]?: T[K];
};

// Or use built-in utilities
type PartialUser = Partial<User>;

// Document complex types
/** Recursively makes all properties optional */
type DeepPartial<T> = T extends object
  ? {
      [P in keyof T]?: DeepPartial<T[P]>;
    }
  : T;
```

## 10. Not Using Type Guards

### ❌ Bad
```typescript
function process(value: string | number) {
  return value.toUpperCase(); // Error: number doesn't have toUpperCase
  // Must use type assertions (bad practice)
  return (value as string).toUpperCase();
}
```

### ✅ Good
```typescript
// Type guard function
function isString(value: string | number): value is string {
  return typeof value === 'string';
}

function process(value: string | number) {
  if (isString(value)) {
    return value.toUpperCase(); // TypeScript knows it's string
  } else {
    return value.toFixed(2); // TypeScript knows it's number
  }
}
```

## 11. Overly Permissive Function Types

### ❌ Bad
```typescript
// Too permissive
type EventHandler = (event: any) => void;
type Callback = (data: any, status?: any) => any;

// Callers don't know what to pass
onEvent((e) => { /* what is e? */ });
```

### ✅ Good
```typescript
// Specific types
type ClickEventHandler = (event: React.MouseEvent<HTMLButtonElement>) => void;
type DataCallback = (data: User, status: 'success' | 'error') => void;

// Callers know what to expect
onClick={(e) => { /* e is MouseEvent, autocomplete works! */ });
```

## 12. Not Using Const Assertions

### ❌ Bad
```typescript
const colors = ['red', 'green', 'blue']; // Type: string[]
const color = colors[0]; // Type: string
// color could be any string!

const config = { timeout: 5000 }; // Type: { timeout: number }
config.timeout = 'invalid'; // No error!
```

### ✅ Good
```typescript
const colors = ['red', 'green', 'blue'] as const; // Type: readonly ['red', 'green', 'blue']
const color = colors[0]; // Type: 'red' (not just string)

const config = { timeout: 5000 } as const; // Type: { readonly timeout: 5000 }
config.timeout = 'invalid'; // Error!
```

## Best Practices Checklist

- [ ] No `any` types - use `unknown` or proper types
- [ ] All types are specific (not just `string`, `object`)
- [ ] Use `readonly` for immutable data
- [ ] Return types are explicit
- [ ] Use literal types for restricted values
- [ ] Use generics to avoid repetition
- [ ] Type guards for narrowing unions
- [ ] Const assertions for literal types
- [ ] Validate runtime data (JSON, API)
- [ ] Use TypeScript strict mode
