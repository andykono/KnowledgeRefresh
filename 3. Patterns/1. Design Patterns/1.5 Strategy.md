# Strategy Pattern

[[1. Patterns.md]]

Encapsulates algorithms in interchangeable objects, allowing runtime selection without changing client code. Perfect for sorting, filtering, validation, and payment methods.

## Classic Strategy Pattern

```typescript
interface SortStrategy<T> {
  execute(items: T[]): T[];
}

class AscendingSort implements SortStrategy<number> {
  execute(items: number[]): number[] {
    return [...items].sort((a, b) => a - b);
  }
}

class DescendingSort implements SortStrategy<number> {
  execute(items: number[]): number[] {
    return [...items].sort((a, b) => b - a);
  }
}

class DataTable {
  private strategy: SortStrategy<number>;

  constructor(strategy: SortStrategy<number>) {
    this.strategy = strategy;
  }

  setStrategy(strategy: SortStrategy<number>) {
    this.strategy = strategy;
  }

  sort(data: number[]): number[] {
    return this.strategy.execute(data);
  }
}

// Usage
const table = new DataTable(new AscendingSort());
table.sort([3, 1, 2]); // [1, 2, 3]
table.setStrategy(new DescendingSort());
table.sort([3, 1, 2]); // [3, 2, 1]
```

## React Sorting Strategies (TanStack Table)

```typescript
// types.ts
export type SortDirection = 'asc' | 'desc';

export interface SortStrategy<T> {
  compare(a: T, b: T): number;
}

// strategies/sortStrategies.ts
export class NumericSort implements SortStrategy<number> {
  compare(a: number, b: number): number {
    return a - b;
  }
}

export class StringSort implements SortStrategy<string> {
  compare(a: string, b: string): number {
    return a.localeCompare(b);
  }
}

export class DateSort implements SortStrategy<Date> {
  compare(a: Date, b: Date): number {
    return a.getTime() - b.getTime();
  }
}

export class CaseInsensitiveSort implements SortStrategy<string> {
  compare(a: string, b: string): number {
    return a.toLowerCase().localeCompare(b.toLowerCase());
  }
}

// hooks/useTableSort.ts
import { useState, useMemo } from 'react';

interface UseSortOptions<T> {
  initialStrategy: SortStrategy<T>;
  data: T[];
  keySelector: (item: T) => any;
}

export function useTableSort<T>({
  initialStrategy,
  data,
  keySelector,
}: UseSortOptions<T>) {
  const [strategy, setStrategy] = useState(initialStrategy);
  const [direction, setDirection] = useState<SortDirection>('asc');

  const sorted = useMemo(() => {
    const copy = [...data];
    copy.sort((a, b) => {
      const result = strategy.compare(keySelector(a), keySelector(b));
      return direction === 'asc' ? result : -result;
    });
    return copy;
  }, [data, strategy, direction, keySelector]);

  return {
    sorted,
    setStrategy,
    setDirection,
    direction,
  };
}

// components/UsersTable.tsx
interface User {
  id: number;
  name: string;
  email: string;
  joinDate: Date;
}

function UsersTable({ users }: { users: User[] }) {
  const { sorted, setStrategy, direction, setDirection } = useTableSort({
    initialStrategy: new StringSort(),
    data: users,
    keySelector: (user) => user.name,
  });

  const handleSortByName = () => {
    setStrategy(new StringSort());
  };

  const handleSortByDate = () => {
    setStrategy(new DateSort());
  };

  const toggleDirection = () => {
    setDirection(direction === 'asc' ? 'desc' : 'asc');
  };

  return (
    <table>
      <thead>
        <tr>
          <th onClick={handleSortByName}>Name {direction === 'asc' ? '↑' : '↓'}</th>
          <th>Email</th>
          <th onClick={handleSortByDate}>Join Date</th>
        </tr>
      </thead>
      <tbody>
        {sorted.map(user => (
          <tr key={user.id}>
            <td>{user.name}</td>
            <td>{user.email}</td>
            <td>{user.joinDate.toLocaleDateString()}</td>
          </tr>
        ))}
      </tbody>
    </table>
  );
}
```

## Validation Strategy Pattern

```typescript
// strategies/validationStrategies.ts
export interface ValidationStrategy {
  validate(value: string): { isValid: boolean; error?: string };
}

export class EmailValidation implements ValidationStrategy {
  validate(value: string) {
    const isValid = /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(value);
    return {
      isValid,
      error: isValid ? undefined : 'Invalid email format',
    };
  }
}

export class PasswordValidation implements ValidationStrategy {
  validate(value: string) {
    const isValid = value.length >= 8 && /[A-Z]/.test(value);
    return {
      isValid,
      error: isValid
        ? undefined
        : 'Password must be 8+ chars with uppercase',
    };
  }
}

export class PhoneValidation implements ValidationStrategy {
  validate(value: string) {
    const isValid = /^\+?1?\d{9,15}$/.test(value);
    return {
      isValid,
      error: isValid ? undefined : 'Invalid phone number',
    };
  }
}

export class URLValidation implements ValidationStrategy {
  validate(value: string) {
    try {
      new URL(value);
      return { isValid: true };
    } catch {
      return { isValid: false, error: 'Invalid URL' };
    }
  }
}

// hooks/useValidation.ts
import { useState, useCallback } from 'react';

export function useValidation(strategy: ValidationStrategy) {
  const [value, setValue] = useState('');
  const [error, setError] = useState<string | undefined>();

  const validate = useCallback((val: string) => {
    const result = strategy.validate(val);
    setValue(val);
    setError(result.error);
    return result.isValid;
  }, [strategy]);

  const reset = useCallback(() => {
    setValue('');
    setError(undefined);
  }, []);

  return { value, error, validate, reset, isValid: !error };
}

// components/RegistrationForm.tsx
import { useState } from 'react';

function RegistrationForm() {
  const email = useValidation(new EmailValidation());
  const password = useValidation(new PasswordValidation());
  const phone = useValidation(new PhoneValidation());
  const [submitted, setSubmitted] = useState(false);

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    const emailValid = email.validate(email.value);
    const passwordValid = password.validate(password.value);
    const phoneValid = phone.validate(phone.value);

    if (emailValid && passwordValid && phoneValid) {
      setSubmitted(true);
      // Submit form
    }
  };

  return (
    <form onSubmit={handleSubmit}>
      <div>
        <label>Email</label>
        <input
          value={email.value}
          onChange={(e) => email.validate(e.target.value)}
          aria-invalid={!!email.error}
        />
        {email.error && <span className="error">{email.error}</span>}
      </div>

      <div>
        <label>Password</label>
        <input
          type="password"
          value={password.value}
          onChange={(e) => password.validate(e.target.value)}
          aria-invalid={!!password.error}
        />
        {password.error && <span className="error">{password.error}</span>}
      </div>

      <div>
        <label>Phone</label>
        <input
          type="tel"
          value={phone.value}
          onChange={(e) => phone.validate(e.target.value)}
          aria-invalid={!!phone.error}
        />
        {phone.error && <span className="error">{phone.error}</span>}
      </div>

      <button disabled={!email.isValid || !password.isValid || !phone.isValid}>
        Register
      </button>

      {submitted && <p>✓ Registration successful!</p>}
    </form>
  );
}
```

## Payment Strategy Pattern

```typescript
// strategies/paymentStrategies.ts
export interface PaymentStrategy {
  pay(amount: number): Promise<{ success: boolean; transactionId: string }>;
  refund(transactionId: string): Promise<boolean>;
}

export class CreditCardPayment implements PaymentStrategy {
  constructor(private cardNumber: string, private cvv: string) {}

  async pay(amount: number) {
    // Simulate payment processing
    const transactionId = `CC-${Date.now()}`;
    return { success: true, transactionId };
  }

  async refund(transactionId: string) {
    console.log(`Refunding ${transactionId} to credit card`);
    return true;
  }
}

export class PayPalPayment implements PaymentStrategy {
  constructor(private email: string) {}

  async pay(amount: number) {
    const transactionId = `PP-${Date.now()}`;
    return { success: true, transactionId };
  }

  async refund(transactionId: string) {
    console.log(`Refunding ${transactionId} via PayPal`);
    return true;
  }
}

export class CryptoPayment implements PaymentStrategy {
  constructor(private walletAddress: string) {}

  async pay(amount: number) {
    const transactionId = `CRYPTO-${Date.now()}`;
    return { success: true, transactionId };
  }

  async refund(transactionId: string) {
    console.log(`Refunding ${transactionId} to wallet`);
    return true;
  }
}

// hooks/usePayment.ts
import { useState } from 'react';

export function usePayment(strategy: PaymentStrategy) {
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<Error | null>(null);
  const [transactionId, setTransactionId] = useState<string | null>(null);

  const pay = async (amount: number) => {
    setLoading(true);
    setError(null);
    try {
      const result = await strategy.pay(amount);
      setTransactionId(result.transactionId);
      return result;
    } catch (err) {
      setError(err instanceof Error ? err : new Error('Payment failed'));
      throw err;
    } finally {
      setLoading(false);
    }
  };

  const refund = async () => {
    if (!transactionId) return;
    try {
      await strategy.refund(transactionId);
      setTransactionId(null);
    } catch (err) {
      setError(err instanceof Error ? err : new Error('Refund failed'));
    }
  };

  return { pay, refund, loading, error, transactionId };
}

// components/Checkout.tsx
function Checkout({ total }: { total: number }) {
  const [paymentMethod, setPaymentMethod] = useState<PaymentStrategy>(
    new CreditCardPayment('****-****-****-1234', '***')
  );

  const payment = usePayment(paymentMethod);

  const handlePaymentMethodChange = (method: string) => {
    if (method === 'paypal') {
      setPaymentMethod(new PayPalPayment('user@example.com'));
    } else if (method === 'crypto') {
      setPaymentMethod(new CryptoPayment('0x1234...'));
    } else {
      setPaymentMethod(new CreditCardPayment('****-****-****-1234', '***'));
    }
  };

  const handleCheckout = async () => {
    try {
      await payment.pay(total);
      alert('Payment successful!');
    } catch {
      alert('Payment failed');
    }
  };

  return (
    <div className="checkout">
      <h2>Select Payment Method</h2>
      <select onChange={(e) => handlePaymentMethodChange(e.target.value)}>
        <option value="credit">Credit Card</option>
        <option value="paypal">PayPal</option>
        <option value="crypto">Cryptocurrency</option>
      </select>

      <div className="total">Total: ${total}</div>

      <button onClick={handleCheckout} disabled={payment.loading}>
        {payment.loading ? 'Processing...' : 'Pay Now'}
      </button>

      {payment.error && <div className="error">{payment.error.message}</div>}
      {payment.transactionId && (
        <div className="success">
          Transaction ID: {payment.transactionId}
          <button onClick={payment.refund}>Refund</button>
        </div>
      )}
    </div>
  );
}
```

## When to Use Strategy Pattern

✅ **Good use cases:**
- Sorting/filtering algorithms
- Validation rules
- Payment methods
- Data export formats (CSV, JSON, XML)
- Caching strategies
- Authentication strategies

❌ **Avoid when:**
- Only one strategy exists
- Strategy logic is simple (use ternary)
- Strategies change rarely
