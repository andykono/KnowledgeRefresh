# Factory Method Pattern

[[1. Patterns.md]]

Centralizes object creation logic, making it easy to create different types of objects without exposing creation details. Great for dynamic components or validators.

## Class-Based Factory

```typescript
interface Chart {
  render(containerId: string): void;
}

class LineChart implements Chart {
  render(containerId: string) {
    console.log('🔵 Rendering line chart in', containerId);
  }
}

class BarChart implements Chart {
  render(containerId: string) {
    console.log('📊 Rendering bar chart in', containerId);
  }
}

class PieChart implements Chart {
  render(containerId: string) {
    console.log('🥧 Rendering pie chart in', containerId);
  }
}

class ChartFactory {
  static create(type: 'line' | 'bar' | 'pie'): Chart {
    switch (type) {
      case 'line': return new LineChart();
      case 'bar': return new BarChart();
      case 'pie': return new PieChart();
      default: throw new Error(`Unknown chart type: ${type}`);
    }
  }
}
```

## React Component Factory

```typescript
// types/charts.ts
export type ChartType = 'line' | 'bar' | 'pie' | 'area';

export interface ChartProps {
  data: number[];
  title: string;
}

// components/LineChart.tsx
export function LineChart({ data, title }: ChartProps) {
  return (
    <div className="chart">
      <h3>{title}</h3>
      <p>Line: {data.join(', ')}</p>
    </div>
  );
}

// components/BarChart.tsx
export function BarChart({ data, title }: ChartProps) {
  return (
    <div className="chart">
      <h3>{title}</h3>
      <div className="bars">
        {data.map((val, i) => (
          <div key={i} style={{ height: `${val}px` }} className="bar" />
        ))}
      </div>
    </div>
  );
}

// components/PieChart.tsx
export function PieChart({ data, title }: ChartProps) {
  return (
    <div className="chart">
      <h3>{title}</h3>
      <p>Pie segments: {data.length}</p>
    </div>
  );
}

// factories/ChartComponentFactory.tsx
import { ComponentType } from 'react';
import { LineChart } from './LineChart';
import { BarChart } from './BarChart';
import { PieChart } from './PieChart';
import { ChartType, ChartProps } from '../types/charts';

export class ChartComponentFactory {
  private static components: Record<ChartType, ComponentType<ChartProps>> = {
    line: LineChart,
    bar: BarChart,
    pie: PieChart,
    area: LineChart, // Reuse LineChart for area
  };

  static getComponent(type: ChartType): ComponentType<ChartProps> {
    const Component = this.components[type];
    if (!Component) {
      throw new Error(`Chart type "${type}" not found`);
    }
    return Component;
  }

  static create(type: ChartType, props: ChartProps) {
    const Component = this.getComponent(type);
    return <Component {...props} />;
  }
}

// hooks/useChartFactory.ts
import { useMemo } from 'react';
import { ChartComponentFactory } from '../factories/ChartComponentFactory';
import { ChartType, ChartProps } from '../types/charts';

export function useChartFactory(type: ChartType, props: ChartProps) {
  return useMemo(() => {
    try {
      return ChartComponentFactory.create(type, props);
    } catch (error) {
      console.error('Chart creation failed:', error);
      return <div>Failed to create chart</div>;
    }
  }, [type, props]);
}

// components/Dashboard.tsx
interface DashboardProps {
  charts: Array<{ type: ChartType; data: number[]; title: string }>;
}

export function Dashboard({ charts }: DashboardProps) {
  return (
    <div className="dashboard">
      {charts.map((config, i) => (
        <div key={i}>
          {ChartComponentFactory.create(config.type, {
            data: config.data,
            title: config.title,
          })}
        </div>
      ))}
    </div>
  );
}
```

## Validator Factory Pattern

```typescript
// validators/validators.ts
export interface Validator {
  validate(value: string): { valid: boolean; error?: string };
}

export class EmailValidator implements Validator {
  validate(value: string) {
    const valid = /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(value);
    return {
      valid,
      error: valid ? undefined : 'Invalid email format',
    };
  }
}

export class PhoneValidator implements Validator {
  validate(value: string) {
    const valid = /^\d{10}$/.test(value);
    return {
      valid,
      error: valid ? undefined : 'Phone must be 10 digits',
    };
  }
}

export class URLValidator implements Validator {
  validate(value: string) {
    try {
      new URL(value);
      return { valid: true };
    } catch {
      return { valid: false, error: 'Invalid URL' };
    }
  }
}

// factories/ValidatorFactory.ts
export type ValidatorType = 'email' | 'phone' | 'url';

export class ValidatorFactory {
  private static validators: Record<ValidatorType, Validator> = {
    email: new EmailValidator(),
    phone: new PhoneValidator(),
    url: new URLValidator(),
  };

  static getValidator(type: ValidatorType): Validator {
    const validator = this.validators[type];
    if (!validator) {
      throw new Error(`Validator type "${type}" not found`);
    }
    return validator;
  }
}

// hooks/useFormValidation.ts
import { useState } from 'react';
import { ValidatorFactory, ValidatorType } from '../factories/ValidatorFactory';

interface ValidationState {
  value: string;
  error?: string;
  valid: boolean;
}

export function useFormValidation(validatorType: ValidatorType) {
  const [state, setState] = useState<ValidationState>({
    value: '',
    valid: false,
  });

  const validator = ValidatorFactory.getValidator(validatorType);

  const handleChange = (value: string) => {
    const result = validator.validate(value);
    setState({
      value,
      ...result,
    });
  };

  return { ...state, handleChange };
}

// components/Form.tsx
function MyForm() {
  const email = useFormValidation('email');
  const phone = useFormValidation('phone');

  return (
    <form>
      <input
        type="email"
        value={email.value}
        onChange={(e) => email.handleChange(e.target.value)}
        aria-invalid={!email.valid}
      />
      {email.error && <span className="error">{email.error}</span>}

      <input
        type="tel"
        value={phone.value}
        onChange={(e) => phone.handleChange(e.target.value)}
        aria-invalid={!phone.valid}
      />
      {phone.error && <span className="error">{phone.error}</span>}

      <button disabled={!email.valid || !phone.valid}>Submit</button>
    </form>
  );
}
```

## Function-Based Factory (Modern React)

```typescript
// types.ts
export type ModalType = 'confirm' | 'alert' | 'form';

export interface ModalConfig {
  title: string;
  message?: string;
  onConfirm?: () => void;
}

// components/ConfirmModal.tsx
function ConfirmModal({ title, message, onConfirm }: ModalConfig) {
  return (
    <dialog open>
      <h2>{title}</h2>
      {message && <p>{message}</p>}
      <button onClick={onConfirm}>Confirm</button>
    </dialog>
  );
}

// modalFactory.tsx
import { ReactNode } from 'react';

export function createModal(
  type: ModalType,
  config: ModalConfig
): ReactNode {
  switch (type) {
    case 'confirm':
      return <ConfirmModal {...config} />;
    case 'alert':
      return <div className="alert">{config.title}</div>;
    case 'form':
      return <form>{config.title}</form>;
    default:
      return null;
  }
}
```

## When to Use Factory Pattern

✅ **Good use cases:**
- Rendering different component types dynamically
- Creating validators based on field type
- Instantiating API response adapters
- Creating different modal/dialog types
- Plugin systems

❌ **Over-engineering:**
- Don't use for simple conditional rendering (use ternary)
- Don't add factory for single type
- Keep it simple: `if (type) return <Component />`
