# Decorator Pattern

[[1. Patterns.md]]

Dynamically adds behavior to objects without modifying their structure. In React, this manifests as Higher-Order Components (HOCs), custom hooks, or wrapper services.

## Classic Class-Based Decorator

```typescript
// Logger interface and implementation
interface Logger {
  log(message: string): void;
}

class ConsoleLogger implements Logger {
  log(message: string) {
    console.log(message);
  }
}

// Decorator: adds timestamp
class TimestampDecorator implements Logger {
  constructor(private logger: Logger) {}

  log(message: string) {
    this.logger.log(`[${new Date().toISOString()}] ${message}`);
  }
}

// Decorator: adds log level
class LevelDecorator implements Logger {
  constructor(private logger: Logger, private level: string) {}

  log(message: string) {
    this.logger.log(`[${this.level}] ${message}`);
  }
}

// Chaining decorators
const logger = new TimestampDecorator(
  new LevelDecorator(new ConsoleLogger(), 'INFO')
);

logger.log('User logged in');
// Output: [INFO] [2026-02-24T...] User logged in
```

## React HOC Pattern (Decorator in React)

```typescript
// types.ts
export interface UserContextType {
  user: { id: number; name: string } | null;
  isLoading: boolean;
}

// context/AuthContext.tsx
import { createContext } from 'react';

export const AuthContext = createContext<UserContextType | undefined>(
  undefined
);

// hocs/withAuth.tsx
import { ComponentType } from 'react';
import { useContext } from 'react';
import { AuthContext } from '../context/AuthContext';

interface WithAuthProps {
  user: NonNullable<UserContextType['user']>;
}

export function withAuth<P extends WithAuthProps>(
  Component: ComponentType<P>
) {
  return function ProtectedComponent(
    props: Omit<P, keyof WithAuthProps>
  ) {
    const context = useContext(AuthContext);

    if (!context) {
      throw new Error('withAuth must be used inside AuthProvider');
    }

    const { user, isLoading } = context;

    if (isLoading) {
      return <div>Loading...</div>;
    }

    if (!user) {
      return <div>Please log in</div>;
    }

    return <Component {...(props as P)} user={user} />;
  };
}

// hocs/withLoading.tsx
interface WithLoadingProps {
  isLoading: boolean;
}

export function withLoading<P extends WithLoadingProps>(
  Component: ComponentType<P>,
  loadingComponent: ComponentType = () => <div>Loading...</div>
) {
  return function LoadingWrapper(props: Omit<P, keyof WithLoadingProps>) {
    return props.isLoading ? (
      <loadingComponent />
    ) : (
      <Component {...(props as P)} />
    );
  };
}

// hocs/withErrorBoundary.tsx
import { Component, ComponentType, ErrorInfo, ReactNode } from 'react';

interface ErrorBoundaryState {
  hasError: boolean;
  error: Error | null;
}

export function withErrorBoundary<P>(Component: ComponentType<P>) {
  return class ErrorBoundary extends Component<P, ErrorBoundaryState> {
    constructor(props: P) {
      super(props);
      this.state = { hasError: false, error: null };
    }

    static getDerivedStateFromError(error: Error): ErrorBoundaryState {
      return { hasError: true, error };
    }

    componentDidCatch(error: Error, info: ErrorInfo) {
      console.error('Error caught:', error, info);
    }

    render(): ReactNode {
      if (this.state.hasError) {
        return (
          <div className="error">
            <p>Something went wrong: {this.state.error?.message}</p>
          </div>
        );
      }

      return <Component {...this.props} />;
    }
  };
}

// components/Dashboard.tsx
interface DashboardProps {
  user: { id: number; name: string };
}

function Dashboard({ user }: DashboardProps) {
  return (
    <div>
      <h1>Welcome, {user.name}!</h1>
      <p>Dashboard content...</p>
    </div>
  );
}

// Apply decorators (HOCs)
const ProtectedDashboard = withAuth(Dashboard);
const SafeDashboard = withErrorBoundary(ProtectedDashboard);

// Usage
export default SafeDashboard;
```

## Function Decorator Pattern (Middleware)

```typescript
// types.ts
export type HttpMethod = 'GET' | 'POST' | 'PUT' | 'DELETE';

export interface HttpRequest {
  method: HttpMethod;
  url: string;
  body?: any;
  headers?: Record<string, string>;
}

export interface HttpResponse<T = any> {
  status: number;
  data: T;
}

// services/httpClient.ts
export class HttpClient {
  async request<T>(request: HttpRequest): Promise<HttpResponse<T>> {
    const response = await fetch(request.url, {
      method: request.method,
      headers: request.headers,
      body: request.body ? JSON.stringify(request.body) : undefined,
    });
    const data = await response.json();
    return { status: response.status, data };
  }
}

// decorators for HttpClient
type ClientDecorator = (
  client: HttpClient
) => HttpClient;

// Logging decorator
export function withLogging(client: HttpClient): HttpClient {
  return {
    async request<T>(request: HttpRequest): Promise<HttpResponse<T>> {
      console.log('→ Request:', request.method, request.url);
      const response = await client.request<T>(request);
      console.log('← Response:', response.status);
      return response;
    },
  };
}

// Caching decorator
export function withCache(client: HttpClient): HttpClient {
  const cache = new Map<string, any>();

  return {
    async request<T>(request: HttpRequest): Promise<HttpResponse<T>> {
      const key = `${request.method}:${request.url}`;
      if (cache.has(key)) {
        console.log('💾 Cache hit:', key);
        return cache.get(key);
      }

      const response = await client.request<T>(request);
      cache.set(key, response);
      return response;
    },
  };
}

// Retry decorator
export function withRetry(
  client: HttpClient,
  maxRetries = 3
): HttpClient {
  return {
    async request<T>(request: HttpRequest): Promise<HttpResponse<T>> {
      for (let attempt = 1; attempt <= maxRetries; attempt++) {
        try {
          return await client.request<T>(request);
        } catch (error) {
          if (attempt === maxRetries) throw error;
          console.log(`Retry ${attempt}/${maxRetries}...`);
          await new Promise(resolve => setTimeout(resolve, 1000 * attempt));
        }
      }
      throw new Error('Max retries exceeded');
    },
  };
}

// Usage: Compose decorators
let client = new HttpClient();
client = withLogging(client);
client = withCache(client);
client = withRetry(client, 2);

// Now all requests are logged, cached, and retried
const response = await client.request({
  method: 'GET',
  url: '/api/users',
});
```

## Custom Hooks as Decorators

```typescript
// hooks/useAsync.ts
import { useEffect, useState } from 'react';

export function useAsync<T>(asyncFn: () => Promise<T>) {
  const [state, setState] = useState({
    loading: true,
    data: null as T | null,
    error: null as Error | null,
  });

  useEffect(() => {
    let mounted = true;

    asyncFn()
      .then(data => {
        if (mounted) setState({ loading: false, data, error: null });
      })
      .catch(error => {
        if (mounted) setState({ loading: false, data: null, error });
      });

    return () => {
      mounted = false;
    };
  }, [asyncFn]);

  return state;
}

// hooks/useDebounce.ts
export function useDebounce<T>(value: T, delay: number): T {
  const [debouncedValue, setDebouncedValue] = useState(value);

  useEffect(() => {
    const handler = setTimeout(() => {
      setDebouncedValue(value);
    }, delay);

    return () => clearTimeout(handler);
  }, [value, delay]);

  return debouncedValue;
}

// hooks/useMemoizedCallback.ts
import { useCallback, useRef } from 'react';

export function useMemoizedCallback<T extends (...args: any[]) => any>(
  callback: T,
  deps: any[]
): T {
  const ref = useRef(callback);

  useEffect(() => {
    ref.current = callback;
  }, deps);

  return useCallback((...args) => ref.current(...args), []) as T;
}

// Usage: Compose hooks

      />
      {loading && <div>Searching...</div>}
      {error && <div>Error: {error.message}</div>}
    </div>
  );
}
```

## Comparison: HOCs vs Hooks vs Components

| Approach | Pro | Con |
| :-- | :-- | :-- |
| **HOCs** | Flexible, composable | Wrapper hell, prop drilling |
| **Custom Hooks** | Simpler, cleaner | Can't extend non-React code |
| **Decorator Classes** | Works with any object | Verbose, OOP-heavy |

## Best Practices

✅ **Use Decorators for:**
- Cross-cutting concerns (logging, caching, retry)
- Authentication/authorization
- Error handling
- Performance monitoring

❌ **Avoid:**
- Deep HOC nesting → use Composition + Hooks
- Decorators on everything → keep it simple
- Tight coupling through decorators
