# Facade Pattern

[[1. Patterns.md]]

Provides a simplified unified interface to a complex subsystem. Hides complexity of orchestrating multiple services (API, storage, auth, logging).

## Classic Facade Pattern

```typescript
// Complex subsystems
class ApiClient {
  async request(path: string) {
    return fetch(path).then(r => r.json());
  }
}

class StorageService {
  get(key: string) {
    return localStorage.getItem(key);
  }
  set(key: string, value: string) {
    localStorage.setItem(key, value);
  }
}

class AuthService {
  validateToken(token: string): boolean {
    return token.length > 0; // Simplified
  }
}

class LoggerService {
  log(message: string) {
    console.log(`[${new Date().toISOString()}] ${message}`);
  }
}

// Facade: simplifies all the above
class UserFacade {
  constructor(
    private api = new ApiClient(),
    private storage = new StorageService(),
    private auth = new AuthService(),
    private logger = new LoggerService()
  ) {}

  async getProfile(): Promise<User> {
    this.logger.log('Fetching profile...');
    const token = this.storage.get('token');

    if (!token || !this.auth.validateToken(token)) {
      throw new Error('Not authenticated');
    }

    const user = await this.api.request('/profile');
    this.logger.log('Profile fetched successfully');
    return user;
  }

  async login(email: string, password: string): Promise<User> {
    this.logger.log(`Login attempt for ${email}`);
    const user = await this.api.request('/login');
    this.storage.set('token', user.token);
    this.storage.set('user', JSON.stringify(user));
    this.logger.log('Login successful');
    return user;
  }

  logout(): void {
    this.logger.log('Logging out...');
    this.storage.set('token', '');
    this.storage.set('user', '');
  }
}

// Simple usage
const userFacade = new UserFacade();
const profile = await userFacade.getProfile();
```

## React Authentication Facade

```typescript
// services/authFacade.ts
interface User {
  id: number;
  email: string;
  name: string;
  token: string;
}

interface LoginCredentials {
  email: string;
  password: string;
}

export class AuthFacade {
  private static readonly TOKEN_KEY = 'auth_token';
  private static readonly USER_KEY = 'auth_user';

  constructor(
    private apiClient: ApiClient,
    private storage: typeof localStorage
  ) {}

  async login(credentials: LoginCredentials): Promise<User> {
    // API call
    const response = await this.apiClient.post<{ user: User; token: string }>(
      '/auth/login',
      credentials
    );

    const { user, token } = response;

    // Store token
    this.storage.setItem(AuthFacade.TOKEN_KEY, token);
    this.storage.setItem(AuthFacade.USER_KEY, JSON.stringify(user));

    return user;
  }

  async signup(email: string, password: string): Promise<User> {
    // API call + validation
    const response = await this.apiClient.post<{ user: User; token: string }>(
      '/auth/signup',
      { email, password }
    );

    const { user, token } = response;

    // Store
    this.storage.setItem(AuthFacade.TOKEN_KEY, token);
    this.storage.setItem(AuthFacade.USER_KEY, JSON.stringify(user));

    return user;
  }

  async logout(): Promise<void> {
    // API call
    await this.apiClient.post('/auth/logout', {});

    // Clear storage
    this.storage.removeItem(AuthFacade.TOKEN_KEY);
    this.storage.removeItem(AuthFacade.USER_KEY);
  }

  async refreshToken(): Promise<string> {
    const response = await this.apiClient.post<{ token: string }>(
      '/auth/refresh',
      {}
    );

    this.storage.setItem(AuthFacade.TOKEN_KEY, response.token);
    return response.token;
  }

  isAuthenticated(): boolean {
    return !!this.storage.getItem(AuthFacade.TOKEN_KEY);
  }

  getToken(): string | null {
    return this.storage.getItem(AuthFacade.TOKEN_KEY);
  }

  getCurrentUser(): User | null {
    const userJson = this.storage.getItem(AuthFacade.USER_KEY);
    return userJson ? JSON.parse(userJson) : null;
  }
}

// Create singleton instance
export const authFacade = new AuthFacade(apiClient, localStorage);

// hooks/useAuth.ts
import { useEffect, useState } from 'react';

export function useAuth() {
  const [user, setUser] = useState<User | null>(authFacade.getCurrentUser());
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<Error | null>(null);

  const login = async (credentials: LoginCredentials) => {
    setLoading(true);
    setError(null);
    try {
      const user = await authFacade.login(credentials);
      setUser(user);
      return user;
    } catch (err) {
      const error = err instanceof Error ? err : new Error('Login failed');
      setError(error);
      throw error;
    } finally {
      setLoading(false);
    }
  };

  const logout = async () => {
    setLoading(true);
    setError(null);
    try {
      await authFacade.logout();
      setUser(null);
    } catch (err) {
      const error = err instanceof Error ? err : new Error('Logout failed');
      setError(error);
    } finally {
      setLoading(false);
    }
  };

  const signup = async (email: string, password: string) => {
    setLoading(true);
    setError(null);
    try {
      const user = await authFacade.signup(email, password);
      setUser(user);
      return user;
    } catch (err) {
      const error = err instanceof Error ? err : new Error('Signup failed');
      setError(error);
      throw error;
    } finally {
      setLoading(false);
    }
  };

  return { user, loading, error, login, logout, signup };
}

// components/LoginForm.tsx
function LoginForm() {
  const { loading, error, login } = useAuth();
  const [email, setEmail] = useState('');
  const [password, setPassword] = useState('');

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    try {
      await login({ email, password });
    } catch {
      // Error handled by hook
    }
  };

  return (
    <form onSubmit={handleSubmit}>
      <input
        type="email"
        value={email}
        onChange={(e) => setEmail(e.target.value)}
        placeholder="Email"
      />
      <input
        type="password"
        value={password}
        onChange={(e) => setPassword(e.target.value)}
        placeholder="Password"
      />
      <button disabled={loading}>
        {loading ? 'Logging in...' : 'Login'}
      </button>
      {error && <div className="error">{error.message}</div>}
    </form>
  );
}
```

## Data Export Facade

```typescript
// services/exportFacade.ts
export interface ExportOptions {
  format: 'csv' | 'json' | 'excel';
  filename?: string;
}

export class DataExportFacade {
  private validateData(data: any[]): void {
    if (!Array.isArray(data) || data.length === 0) {
      throw new Error('Invalid data for export');
    }
  }

  private generateFilename(format: string): string {
    return `export-${Date.now()}.${format === 'excel' ? 'xlsx' : format}`;
  }

  async export(data: any[], options: ExportOptions): Promise<Blob> {
    this.validateData(data);

    switch (options.format) {
      case 'csv':
        return this.exportCSV(data);
      case 'json':
        return this.exportJSON(data);
      case 'excel':
        return this.exportExcel(data);
      default:
        throw new Error(`Unknown format: ${options.format}`);
    }
  }

  private exportCSV(data: any[]): Blob {
    const headers = Object.keys(data[0]);
    const csv = [
      headers.join(','),
      ...data.map(row =>
        headers.map(h => JSON.stringify(row[h])).join(',')
      ),
    ].join('\n');
    return new Blob([csv], { type: 'text/csv' });
  }

  private exportJSON(data: any[]): Blob {
    const json = JSON.stringify(data, null, 2);
    return new Blob([json], { type: 'application/json' });
  }

  private async exportExcel(data: any[]): Promise<Blob> {
    // Would use a library like xlsx
    const json = JSON.stringify(data, null, 2);
    return new Blob([json], { type: 'application/vnd.ms-excel' });
  }

  downloadFile(blob: Blob, filename: string): void {
    const url = URL.createObjectURL(blob);
    const link = document.createElement('a');
    link.href = url;
    link.download = filename;
    link.click();
    URL.revokeObjectURL(url);
  }
}

// hooks/useDataExport.ts
export function useDataExport() {
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<Error | null>(null);
  const exporter = new DataExportFacade();

  const exportData = async (
    data: any[],
    options: ExportOptions
  ) => {
    setLoading(true);
    setError(null);
    try {
      const blob = await exporter.export(data, options);
      const filename = options.filename || exporter.generateFilename(options.format);
      exporter.downloadFile(blob, filename);
    } catch (err) {
      const error = err instanceof Error ? err : new Error('Export failed');
      setError(error);
    } finally {
      setLoading(false);
    }
  };

  return { exportData, loading, error };
}

// components/ExportButtons.tsx
function ExportButtons({ data }: { data: any[] }) {
  const { exportData, loading } = useDataExport();

  return (
    <div className="export-buttons">
      <button
        onClick={() => exportData(data, { format: 'csv', filename: 'data.csv' })}
        disabled={loading}
      >
        Export CSV
      </button>
      <button
        onClick={() => exportData(data, { format: 'json', filename: 'data.json' })}
        disabled={loading}
      >
        Export JSON
      </button>
      <button
        onClick={() => exportData(data, { format: 'excel', filename: 'data.xlsx' })}
        disabled={loading}
      >
        Export Excel
      </button>
    </div>
  );
}
```

## Payment Facade

```typescript
// services/paymentFacade.ts
export class PaymentFacade {
  constructor(
    private stripeService: StripeService,
    private paypalService: PayPalService,
    private orderService: OrderService,
    private notificationService: NotificationService
  ) {}

  async processPayment(
    orderId: string,
    amount: number,
    paymentMethod: 'stripe' | 'paypal'
  ): Promise<PaymentResult> {
    // Orchestrate all services
    let transactionId: string;

    try {
      if (paymentMethod === 'stripe') {
        transactionId = await this.stripeService.charge(amount);
      } else {
        transactionId = await this.paypalService.charge(amount);
      }

      // Update order
      await this.orderService.updatePaymentStatus(orderId, 'completed');

      // Send notification
      await this.notificationService.sendEmail(
        'payment_success',
        { orderId, amount }
      );

      return { success: true, transactionId };
    } catch (error) {
      await this.notificationService.sendEmail('payment_failed', { orderId });
      throw error;
    }
  }
}
```

## When to Use Facade

✅ **Good use cases:**
- Complex API integration (payment, auth, data export)
- Combining multiple services into one interface
- Hiding implementation details
- Providing a stable public API

❌ **Over-engineering:**
- Simple operations that don't need orchestration
- When facade adds no real simplification
- Single operation that just calls one service
