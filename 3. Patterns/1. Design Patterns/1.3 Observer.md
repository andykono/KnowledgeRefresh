# Observer Pattern

[[1. Patterns.md]]

Defines one-to-many dependency where a subject notifies multiple observers of state changes. Core pattern for reactivity, state management, and event-driven architectures.

## Classic Observable Implementation

```typescript
type Listener<T> = (data: T) => void;

class Subject<T> {
  private observers = new Set<Listener<T>>();

  subscribe(callback: Listener<T>): () => void {
    this.observers.add(callback);
    // Return unsubscribe function
    return () => this.observers.delete(callback);
  }

  notify(data: T): void {
    this.observers.forEach(callback => callback(data));
  }

  getObserverCount(): number {
    return this.observers.size;
  }
}

// Usage
const userSubject = new Subject<{ id: number; name: string }>();

const unsubscribe1 = userSubject.subscribe(user => {
  console.log('Observer 1:', user);
});

const unsubscribe2 = userSubject.subscribe(user => {
  console.log('Observer 2:', user);
});

userSubject.notify({ id: 1, name: 'Alice' }); // Both observers notified

unsubscribe1(); // Stop listening
userSubject.notify({ id: 2, name: 'Bob' }); // Only Observer 2 notified
```

## React Observer Pattern with Custom Hook

```typescript
// hooks/useObservable.ts
import { useEffect, useState } from 'react';

type Listener<T> = (data: T) => void;

export class Observable<T> {
  private listeners = new Set<Listener<T>>();

  subscribe(callback: Listener<T>): () => void {
    this.listeners.add(callback);
    return () => this.listeners.delete(callback);
  }

  emit(value: T): void {
    this.listeners.forEach(callback => callback(value));
  }
}

export function useObservable<T>(
  observable: Observable<T>,
  initialValue: T
): T {
  const [value, setValue] = useState(initialValue);

  useEffect(() => {
    const unsubscribe = observable.subscribe(setValue);
    return unsubscribe; // Cleanup
  }, [observable]);

  return value;
}

// stores/userStore.ts
export interface User {
  id: number;
  name: string;
  email: string;
}

export const userStore = new Observable<User | null>(null);

export function updateUser(user: User) {
  userStore.emit(user);
}

// components/UserProfile.tsx
import { useObservable } from '../hooks/useObservable';
import { userStore, updateUser } from '../stores/userStore';

function UserProfile() {
  const user = useObservable(userStore, null);

  if (!user) return <div>No user logged in</div>;

  return (
    <div>
      <h1>{user.name}</h1>
      <p>{user.email}</p>
      <button onClick={() => updateUser({ ...user, name: 'Updated' })}>
        Update
      </button>
    </div>
  );
}
```

## Event Bus Pattern (Observer for Events)

```typescript
// utils/EventBus.ts
type EventHandler<T = any> = (data: T) => void;

export class EventBus {
  private events: Map<string, Set<EventHandler>> = new Map();

  on<T>(eventName: string, handler: EventHandler<T>): () => void {
    if (!this.events.has(eventName)) {
      this.events.set(eventName, new Set());
    }
    this.events.get(eventName)!.add(handler);

    // Return unsubscribe function
    return () => {
      const handlers = this.events.get(eventName);
      if (handlers) {
        handlers.delete(handler);
      }
    };
  }

  emit(eventName: string, data?: any): void {
    const handlers = this.events.get(eventName);
    if (handlers) {
      handlers.forEach(handler => handler(data));
    }
  }

  off(eventName: string, handler: EventHandler): void {
    const handlers = this.events.get(eventName);
    if (handlers) {
      handlers.delete(handler);
    }
  }
}

export const eventBus = new EventBus();

// hooks/useEventBus.ts
export function useEventBusListener<T>(
  eventName: string,
  handler: (data: T) => void
): void {
  useEffect(() => {
    const unsubscribe = eventBus.on<T>(eventName, handler);
    return unsubscribe;
  }, [eventName, handler]);
}

// components/LoginForm.tsx
function LoginForm() {
  const handleSubmit = async (credentials: Credentials) => {
    try {
      const user = await api.post('/login', credentials);
      // Notify all listeners
      eventBus.emit('user:login', user);
    } catch (error) {
      eventBus.emit('auth:error', error);
    }
  };

  return <form onSubmit={handleSubmit}>...</form>;
}

// components/Navbar.tsx
function Navbar() {
  const [user, setUser] = useState<User | null>(null);

  useEventBusListener('user:login', (user: User) => {
    setUser(user);
  });

  useEventBusListener('user:logout', () => {
    setUser(null);
  });

  return <nav>{user ? <span>{user.name}</span> : <span>Login</span>}</nav>;
}
```

## React Context + Observer (Modern Approach)

```typescript
// context/UserContext.tsx
import { createContext, useState, ReactNode, useCallback } from 'react';

export interface User {
  id: number;
  name: string;
  email: string;
}

interface UserContextType {
  user: User | null;
  login: (user: User) => void;
  logout: () => void;
}

export const UserContext = createContext<UserContextType | undefined>(
  undefined
);

export function UserProvider({ children }: { children: ReactNode }) {
  const [user, setUser] = useState<User | null>(null);

  const login = useCallback((user: User) => {
    setUser(user);
  }, []);

  const logout = useCallback(() => {
    setUser(null);
  }, []);

  const value: UserContextType = { user, login, logout };

  return (
    <UserContext.Provider value={value}>
      {children}
    </UserContext.Provider>
  );
}

// hooks/useUser.ts
import { useContext } from 'react';

export function useUser() {
  const context = useContext(UserContext);
  if (!context) {
    throw new Error('useUser must be used within UserProvider');
  }
  return context;
}

// App.tsx
function App() {
  return (
    <UserProvider>
      <Navbar />
      <MainContent />
    </UserProvider>
  );
}

// components/Navbar.tsx
function Navbar() {
  const { user, logout } = useUser();

  return (
    <nav>
      {user ? (
        <>
          <span>{user.name}</span>
          <button onClick={logout}>Logout</button>
        </>
      ) : (
        <span>Not logged in</span>
      )}
    </nav>
  );
}
```

## RxJS Observable (Advanced React)

```typescript
// stores/userStore.ts
import { BehaviorSubject } from 'rxjs';

export interface User {
  id: number;
  name: string;
}

const initialState: User | null = null;
export const user$ = new BehaviorSubject(initialState);

// hooks/useRxObservable.ts
import { useEffect, useState } from 'react';
import { Observable } from 'rxjs';

export function useRxObservable<T>(
  source$: Observable<T>,
  initialValue: T
): T {
  const [value, setValue] = useState(initialValue);

  useEffect(() => {
    const subscription = source$.subscribe(setValue);
    return () => subscription.unsubscribe();
  }, [source$]);

  return value;
}

// components/UserProfile.tsx
function UserProfile() {
  const user = useRxObservable(user$, null);

  return <div>{user?.name || 'No user'}</div>;
}
```

## Comparison: When to Use Each

| Approach | Use Case | Pros | Cons |
| :-- | :-- | :-- | :-- |
| Custom Observable | Simple state sharing | Lightweight, custom | Manual management |
| EventBus | Cross-component communication | Decoupled, flexible | Can get messy |
| React Context | App-wide shared state | Built-in, integrated | Can cause re-renders |
| RxJS | Complex async flows | Powerful, composable | Steep learning curve |

## Memory Leak Prevention

⚠️ **Always unsubscribe!**

```typescript
useEffect(() => {
  const unsubscribe = observable.subscribe(handler);
  return unsubscribe; // Cleanup function
}, [observable]);
```

Without cleanup → memory leaks → app slowdown
