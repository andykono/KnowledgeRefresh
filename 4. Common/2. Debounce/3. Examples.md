## 1. Поиск в реальном времени

### React + API

```tsx
import { useState, useCallback } from 'react';
import { useDebounce } from './useDebounce';

function SearchUsers() {
  const [query, setQuery] = useState('');
  const [results, setResults] = useState<User[]>([]);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const fetchUsers = useCallback(async (searchQuery: string) => {
    if (!searchQuery.trim()) {
      setResults([]);
      return;
    }

    setLoading(true);
    setError(null);

    try {
      const res = await fetch(`/api/users/search?q=${searchQuery}`);
      const data = await res.json();
      setResults(data);
    } catch (err) {
      setError('Ошибка поиска');
      setResults([]);
    } finally {
      setLoading(false);
    }
  }, []);

  const debouncedSearch = useDebounce(fetchUsers, 500);

  const handleChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    const value = e.target.value;
    setQuery(value);
    debouncedSearch(value);
  };

  return (
    <div>
      <input
        placeholder="Поиск пользователей..."
        value={query}
        onChange={handleChange}
      />
      {loading && <p>Загрузка...</p>}
      {error && <p style={{ color: 'red' }}>{error}</p>}
      <ul>
        {results.map((user) => (
          <li key={user.id}>{user.name}</li>
        ))}
      </ul>
    </div>
  );
}
```

**Результат:** запрос отправляется только через 500мс после остановки печати.

## 2. Валидация форм

### React Hook Form + debounce

```tsx
import { useForm } from 'react-hook-form';
import { useDebounce } from './useDebounce';

interface FormData {
  email: string;
  username: string;
}

function RegistrationForm() {
  const { register, watch, formState: { errors } } = useForm<FormData>();
  const [validationErrors, setValidationErrors] = useState<Record<string, string>>({});

  const validateEmail = useCallback(async (email: string) => {
    if (!email) {
      setValidationErrors((prev) => ({ ...prev, email: '' }));
      return;
    }

    try {
      const res = await fetch(`/api/check-email?email=${email}`);
      const { available } = await res.json();

      if (!available) {
        setValidationErrors((prev) => ({ ...prev, email: 'Email уже занят' }));
      } else {
        setValidationErrors((prev) => ({ ...prev, email: '' }));
      }
    } catch (err) {
      setValidationErrors((prev) => ({ ...prev, email: 'Ошибка проверки' }));
    }
  }, []);

  const debouncedEmailValidation = useDebounce(validateEmail, 600);

  const email = watch('email');

  useEffect(() => {
    debouncedEmailValidation(email);
  }, [email, debouncedEmailValidation]);

  return (
    <form>
      <input {...register('email')} placeholder="Email" />
      {validationErrors.email && (
        <p style={{ color: 'red' }}>{validationErrors.email}</p>
      )}

      <input {...register('username')} placeholder="Username" />

      <button type="submit">Регистрация</button>
    </form>
  );
}
```

**Результат:** email проверяется на сервере только после паузы в вводе.

## 3. Автосохранение документа

### React + localStorage

```tsx
import { useState, useCallback, useEffect } from 'react';
import { useDebounce } from './useDebounce';

interface Document {
  id: string;
  content: string;
  lastSaved: Date;
}

function TextEditor({ docId }: { docId: string }) {
  const [content, setContent] = useState('');
  const [saving, setSaving] = useState(false);
  const [lastSaved, setLastSaved] = useState<Date | null>(null);

  // Загружаем документ при монтировании
  useEffect(() => {
    const saved = localStorage.getItem(`doc-${docId}`);
    if (saved) setContent(saved);
  }, [docId]);

  const saveToServer = useCallback(async (text: string) => {
    setSaving(true);
    try {
      await fetch(`/api/documents/${docId}`, {
        method: 'PUT',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ content: text }),
      });
      setLastSaved(new Date());
    } catch (err) {
      console.error('Ошибка сохранения');
    } finally {
      setSaving(false);
    }
  }, [docId]);

  const debouncedSave = useDebounce(saveToServer, 2000);

  const handleChange = (e: React.ChangeEvent<HTMLTextAreaElement>) => {
    const newContent = e.target.value;
    setContent(newContent);
    // Сохраняем локально сразу
    localStorage.setItem(`doc-${docId}`, newContent);
    // А на сервер — с debounce
    debouncedSave(newContent);
  };

  return (
    <div>
      <textarea value={content} onChange={handleChange} />
      <footer>
        {saving && <span>💾 Сохранение...</span>}
        {lastSaved && <span>✅ Сохранено в {lastSaved.toLocaleTimeString()}</span>}
      </footer>
    </div>
  );
}
```

**Результат:** пока пользователь печатает, содержимое хранится в localStorage. На сервер сохраняется только через 2 секунды после остановки.

## 4. Фильтрация с debounce

### Таблица с поиском

```tsx
import { useState, useMemo } from 'react';
import { useDebounce } from './useDebounce';

function DataTable({ data }: { data: Item[] }) {
  const [searchTerm, setSearchTerm] = useState('');
  const [debouncedTerm, setDebouncedTerm] = useState('');

  const debouncedSearch = useDebounce((term: string) => {
    setDebouncedTerm(term);
  }, 300);

  const handleSearchChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    const term = e.target.value;
    setSearchTerm(term);
    debouncedSearch(term);
  };

  const filteredData = useMemo(() => {
    if (!debouncedTerm) return data;
    return data.filter((item) =>
      item.name.toLowerCase().includes(debouncedTerm.toLowerCase())
    );
  }, [data, debouncedTerm]);

  return (
    <div>
      <input
        type="text"
        placeholder="Поиск..."
        value={searchTerm}
        onChange={handleSearchChange}
      />
      <table>
        <thead>
          <tr>
            <th>Имя</th>
            <th>Тип</th>
          </tr>
        </thead>
        <tbody>
          {filteredData.map((item) => (
            <tr key={item.id}>
              <td>{item.name}</td>
              <td>{item.type}</td>
            </tr>
          ))}
        </tbody>
      </table>
    </div>
  );
}
```

**Результат:** таблица фильтруется только после паузы в поиске, а не на каждый символ.
