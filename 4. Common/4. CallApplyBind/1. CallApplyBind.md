# Call, Apply, Bind

[[5. FED/3. JavaScript/1. Core concepts/1. this]] · [[4. Common/3. Hoisting/1. Hoisting]]

## Определение

**call**, **apply**, **bind** — методы функций, которые позволяют явно установить контекст `this` при вызове функции. Необходимы для правильной работы с контекстом и решения проблем с потерей `this`.

---

## 1. call()

**Синтаксис:** `func.call(thisArg, arg1, arg2, ...)`

Вызывает функцию **немедленно**, передавая контекст `this` и аргументы через запятую.

### Пример 1: Базовое использование

```javascript
function greet(greeting) {
  console.log(`${greeting}, ${this.name}!`);
}

const person = { name: 'Alice' };

greet.call(person, 'Hello');  // Hello, Alice!
```

Без `call` контекст был бы `window` (или `undefined` в strict mode).

### Пример 2: Заимствование методов (method borrowing)

```javascript
const person1 = {
  name: 'Alice',
  getName() {
    return this.name;
  }
};

const person2 = { name: 'Bob' };

console.log(person1.getName.call(person2));  // 'Bob'
// Берём метод person1 и применяем контекст person2
```

### Пример 3: call с несколькими аргументами

```javascript
function calculate(a, b, operation) {
  if (operation === 'add') {
    console.log(this.value + a + b);
  } else if (operation === 'multiply') {
    console.log(this.value * a * b);
  }
}

const obj = { value: 10 };

calculate.call(obj, 5, 3, 'add');       // 18
calculate.call(obj, 5, 3, 'multiply');  // 150
```

### Пример 4: Преобразование массива-подобного объекта

```javascript
function printArgs() {
  // arguments это не настоящий массив
  const argsArray = Array.prototype.slice.call(arguments);
  console.log(argsArray);  // [1, 2, 3]
}

printArgs(1, 2, 3);
```

## 2. apply()

**Синтаксис:** `func.apply(thisArg, [arg1, arg2, ...])`

Вызывает функцию **немедленно**, передавая контекст `this` и аргументы в виде **массива**.

### Пример 1: Базовое использование

```javascript
function greet(greeting, punctuation) {
  console.log(`${greeting}, ${this.name}${punctuation}`);
}

const person = { name: 'Alice' };

greet.apply(person, ['Hello', '!']);  // Hello, Alice!
```

### Пример 2: Поиск максимума в массиве

```javascript
const numbers = [5, 6, 2, 3, 7];

// Math.max ожидает аргументы, не массив
console.log(Math.max(...numbers));  // 7 (с spread оператором)

// Или с apply (старый способ):
console.log(Math.max.apply(null, numbers));  // 7
```

### Пример 3: Применение функции с переменным числом аргументов

```javascript
function sum(a, b, c) {
  return a + b + c;
}

const args = [1, 2, 3];
console.log(sum.apply(null, args));  // 6
```

### Пример 4: Декоратор с apply (сохранение контекста)

```javascript
function timer(func) {
  return function(...args) {
    console.time('Execution');
    const result = func.apply(this, args);
    console.timeEnd('Execution');
    return result;
  };
}

const person = {
  name: 'Alice',
  greet(greeting) {
    return `${greeting}, ${this.name}!`;
  }
};

const timedGreet = timer(person.greet);
console.log(timedGreet.call(person, 'Hello'));  // Execution: 0.1ms, Hello, Alice!
```

## 3. bind()

**Синтаксис:** `func.bind(thisArg, arg1, arg2, ...)`

Создаёт **новую функцию** с установленным контекстом `this`. **Не вызывает функцию**, а возвращает её для позднейшего вызова.

### Пример 1: Базовое использование

```javascript
function greet(greeting) {
  console.log(`${greeting}, ${this.name}!`);
}

const person = { name: 'Alice' };

const boundGreet = greet.bind(person);
boundGreet('Hello');  // Hello, Alice!

// Или с аргументами:
const boundGreet2 = greet.bind(person, 'Hi');
boundGreet2();  // Hi, Alice!
```

### Пример 2: Обработчики событий (частое применение)

```javascript
class Button {
  constructor(label) {
    this.label = label;
    this.element = document.createElement('button');
    this.element.textContent = label;

    // Без bind — this будет элементом, не instance класса
    this.element.addEventListener('click', this.onClick.bind(this));
  }

  onClick() {
    console.log(`Button "${this.label}" clicked!`);
  }
}

const btn = new Button('Submit');
// При клике выведет: Button "Submit" clicked!
```

### Пример 3: Partial application (частичное применение аргументов)

```javascript
function multiply(a, b, c) {
  return a * b * c;
}

const multiplyBy2 = multiply.bind(null, 2);
console.log(multiplyBy2(3, 4));  // 2 * 3 * 4 = 24

const multiplyBy2By3 = multiply.bind(null, 2, 3);
console.log(multiplyBy2By3(4));  // 2 * 3 * 4 = 24
```

### Пример 4: setTimeout с правильным контекстом

```javascript
class Timer {
  constructor(name) {
    this.name = name;
    this.count = 0;
  }

  increment() {
    this.count++;
    console.log(`${this.name}: ${this.count}`);
  }

  start() {
    // Без bind — this потеряется в setTimeout
    // setTimeout(this.increment, 1000);  // ERROR

    // С bind — контекст сохранится
    setInterval(this.increment.bind(this), 1000);
  }
}

const timer = new Timer('MyTimer');
timer.start();  // MyTimer: 1, MyTimer: 2, MyTimer: 3, ...
```

### Пример 5: React (функциональные компоненты вместо класса)

```javascript
// Старый способ с классом (нужен bind):
class LoginForm extends React.Component {
  constructor(props) {
    super(props);
    this.state = { email: '' };
    // Нужен bind для сохранения this
    this.handleChange = this.handleChange.bind(this);
  }

  handleChange(e) {
    this.setState({ email: e.target.value });
  }

  render() {
    return <input onChange={this.handleChange} />;
  }
}

// Современный способ с функциональным компонентом:
function LoginForm() {
  const [email, setEmail] = useState('');

  const handleChange = (e) => {
    setEmail(e.target.value);  // arrow function уже сохраняет this
  };

  return <input onChange={handleChange} />;
}
```

## 4. Сравнение call, apply, bind

| | **call** | **apply** | **bind** |
|-----|---------|---------|--------|
| **Вызов** | Немедленно | Немедленно | Возвращает функцию |
| **Аргументы** | Через запятую: `func.call(this, a, b)` | Массивом: `func.apply(this, [a, b])` | Через запятую: `func.bind(this, a, b)` |
| **Возврат** | Результат функции | Результат функции | Новая функция |
| **Использование** | Один вызов с контекстом | Один вызов с контекстом | Для повторного использования |

### Пример сравнения:

```javascript
function introduce(greeting, punctuation) {
  return `${greeting}, ${this.name}${punctuation}`;
}

const person = { name: 'Alice' };

// call — сразу
console.log(introduce.call(person, 'Hello', '!'));
// Hello, Alice!

// apply — сразу с массивом
console.log(introduce.apply(person, ['Hi', '?']));
// Hi, Alice?

// bind — функция на потом
const boundIntroduce = introduce.bind(person, 'Hey');
console.log(boundIntroduce('...'));
// Hey, Alice...
```

## 5. Практические примеры

### Пример 1: Проверка типа (Array.isArray альтернатива)

```javascript
function isArray(obj) {
  return Object.prototype.toString.call(obj) === '[object Array]';
}

console.log(isArray([1, 2, 3]));       // true
console.log(isArray('string'));        // false
console.log(isArray({ length: 3 }));   // false
```

### Пример 2: Создание дефолтного контекста

```javascript
class API {
  constructor(baseURL) {
    this.baseURL = baseURL;
  }

  fetch(endpoint) {
    return `Fetching from ${this.baseURL}${endpoint}`;
  }
}

const api = new API('https://api.example.com');

// Заимствуем метод для другого объекта
const anotherAPI = { baseURL: 'https://other.com' };
console.log(api.fetch.call(anotherAPI, '/users'));
// Fetching from https://other.com/users
```

### Пример 3: Функция для работы с любым this

```javascript
function greetAll(greeting) {
  this.people.forEach(person => {
    console.log(`${greeting}, ${person}!`);
  });
}

const group1 = { people: ['Alice', 'Bob'] };
const group2 = { people: ['Charlie', 'Diana'] };

greetAll.call(group1, 'Hello');
// Hello, Alice!
// Hello, Bob!

greetAll.call(group2, 'Hi');
// Hi, Charlie!
// Hi, Diana!
```

### Пример 4: Цепочка вызовов с bind

```javascript
const calculator = {
  value: 0,

  add(num) {
    this.value += num;
    return this;  // возвращаем this для chaining
  },

  multiply(num) {
    this.value *= num;
    return this;
  },

  result() {
    return this.value;
  }
};

// Bind для использования без контекста объекта
const calc = calculator.add.bind(calculator, 5)
  .add.bind(calculator, 3)
  .multiply.bind(calculator, 2);

// Или проще — использовать контекст:
console.log(
  calculator.add(5).multiply(2).add(3).result()
);  // (5 * 2) + 3 = 13
```

### Пример 5: Защита от случайного изменения контекста

```javascript
class User {
  constructor(name) {
    this.name = name;
  }

  // Привязываем контекст в конструкторе
  greet = () => {
    console.log(`Hi, I'm ${this.name}`);
  };

  // Или через bind:
  goodbye() {
    console.log(`Bye from ${this.name}`);
  }
}

const user = new User('Alice');

// greet сохранит контекст даже при передаче как callback
const handler = user.greet;
handler();  // Hi, I'm Alice

// goodbye потеряет контекст без bind
const handler2 = user.goodbye;  // потеряет this
// handler2();  // Error или window.name

// С bind:
const handler3 = user.goodbye.bind(user);
handler3();  // Bye from Alice
```

## 6. Когда использовать что?

- **call** — когда нужен один вызов и удобно передавать аргументы отдельно
- **apply** — когда аргументы уже есть в массиве (особенно полезно для функций вроде `Math.max`)
- **bind** — когда нужно сохранить функцию для позднего использования (обработчики событий, setTimeout, callbacks)

### Правило большого пальца:

```javascript
// Есть функция, которую нужно вызвать ЖИЗ и сейчас?
func.call(context, args);      // call
func.apply(context, [args]);   // apply

// Нужно передать функцию куда-то для позднего использования?
const boundFunc = func.bind(context, args);
```

## 7. Резюме

- ✅ **call** — вызов с явным `this`, аргументы через запятую
- ✅ **apply** — вызов с явным `this`, аргументы массивом
- ✅ **bind** — создание новой функции с фиксированным `this`
- ✅ **Используй call/apply** для одноразовых вызовов
- ✅ **Используй bind** для callbacks, обработчиков событий, методов класса
