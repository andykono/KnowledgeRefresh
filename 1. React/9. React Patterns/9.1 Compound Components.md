# Compound Components Pattern

[[9. React Patterns.md]]

**Compound Components** — это паттерн, где несколько компонентов работают вместе, разделяя состояние через Context API.

---

## What Problem Does It Solve?

```typescript
// ❌ Bad: Props drilling, inflexible
<Dialog
  title="Confirm"
  message="Are you sure?"
  showHeader={true}
  showFooter={true}
  onConfirm={handleConfirm}
  buttons={[{ label: 'Yes', onClick: confirm }]}
/>

// ✅ Good: Flexible, composable structure
<Dialog>
  <Dialog.Header>Confirm</Dialog.Header>
  <Dialog.Content>Are you sure?</Dialog.Content>
  <Dialog.Footer>
    <Dialog.Button onClick={handleConfirm}>Yes</Dialog.Button>
  </Dialog.Footer>
</Dialog>
```

---

## Core Concept

**Compound Components** = Multiple related components that share internal state via Context.

Benefits:
- ✅ **Flexible** — reorder elements as needed
- ✅ **Readable** — clear structure in JSX
- ✅ **Maintainable** — each sub-component is simple
- ✅ **Composable** — combine freely

## Basic Example: Tabs

```typescript
// Context for sharing state
interface TabsContextType {
  activeTab: number;
  setActiveTab: (index: number) => void;
}

const TabsContext = createContext<TabsContextType | undefined>(undefined);

// Helper hook
const useTabs = () => {
  const context = useContext(TabsContext);
  if (!context) {
    throw new Error('useTabs must be used inside <Tabs>');
  }
  return context;
};

// Main compound component
export function Tabs({ children }: { children: React.ReactNode }) {
  const [activeTab, setActiveTab] = useState(0);

  return (
    <TabsContext.Provider value={{ activeTab, setActiveTab }}>
      <div className="tabs">
        {children}
      </div>
    </TabsContext.Provider>
  );
}

// Sub-component: List
Tabs.List = function TabsList({ children }: { children: React.ReactNode }) {
  return <div className="tabs-list">{children}</div>;
};

// Sub-component: Item (tab header)
Tabs.Item = function TabsItem({
  index,
  children,
}: {
  index: number;
  children: React.ReactNode;
}) {
  const { activeTab, setActiveTab } = useTabs();

  return (
    <button
      onClick={() => setActiveTab(index)}
      className={`tab-item ${activeTab === index ? 'active' : ''}`}
    >
      {children}
    </button>
  );
};

// Sub-component: Content (tab panel)
Tabs.Content = function TabsContent({
  index,
  children,
}: {
  index: number;
  children: React.ReactNode;
}) {
  const { activeTab } = useTabs();

  return activeTab === index ? (
    <div className="tab-content">{children}</div>
  ) : null;
};

// Usage
export function TabsDemo() {
  return (
    <Tabs>
      <Tabs.List>
        <Tabs.Item index={0}>Home</Tabs.Item>
        <Tabs.Item index={1}>About</Tabs.Item>
        <Tabs.Item index={2}>Contact</Tabs.Item>
      </Tabs.List>

      <Tabs.Content index={0}>
        <h2>Home Content</h2>
      </Tabs.Content>

      <Tabs.Content index={1}>
        <h2>About Content</h2>
      </Tabs.Content>

      <Tabs.Content index={2}>
        <h2>Contact Content</h2>
      </Tabs.Content>
    </Tabs>
  );
}
```

## Accordion Example

```typescript
interface AccordionContextType {
  expandedIndex: number | null;
  setExpandedIndex: (index: number | null) => void;
}

const AccordionContext = createContext<AccordionContextType | undefined>(
  undefined
);

const useAccordion = () => {
  const context = useContext(AccordionContext);
  if (!context) {
    throw new Error('useAccordion must be used inside <Accordion>');
  }
  return context;
};

export function Accordion({ children }: { children: React.ReactNode }) {
  const [expandedIndex, setExpandedIndex] = useState<number | null>(0);

  return (
    <AccordionContext.Provider value={{ expandedIndex, setExpandedIndex }}>
      <div className="accordion">{children}</div>
    </AccordionContext.Provider>
  );
}

Accordion.Item = function AccordionItem({
  index,
  children,
}: {
  index: number;
  children: React.ReactNode;
}) {
  return <div className="accordion-item">{children}</div>;
};

Accordion.Header = function AccordionHeader({
  index,
  children,
}: {
  index: number;
  children: React.ReactNode;
}) {
  const { expandedIndex, setExpandedIndex } = useAccordion();
  const isExpanded = expandedIndex === index;

  return (
    <button
      onClick={() => setExpandedIndex(isExpanded ? null : index)}
      className={`accordion-header ${isExpanded ? 'expanded' : ''}`}
      aria-expanded={isExpanded}
    >
      {children}
      <span className="icon">{isExpanded ? '−' : '+'}</span>
    </button>
  );
};

Accordion.Content = function AccordionContent({
  index,
  children,
}: {
  index: number;
  children: React.ReactNode;
}) {
  const { expandedIndex } = useAccordion();

  return expandedIndex === index ? (
    <div className="accordion-content">{children}</div>
  ) : null;
};

// Usage
export function AccordionDemo() {
  return (
    <Accordion>
      <Accordion.Item index={0}>
        <Accordion.Header index={0}>Section 1</Accordion.Header>
        <Accordion.Content index={0}>Content 1</Accordion.Content>
      </Accordion.Item>

      <Accordion.Item index={1}>
        <Accordion.Header index={1}>Section 2</Accordion.Header>
        <Accordion.Content index={1}>Content 2</Accordion.Content>
      </Accordion.Item>

      <Accordion.Item index={2}>
        <Accordion.Header index={2}>Section 3</Accordion.Header>
        <Accordion.Content index={2}>Content 3</Accordion.Content>
      </Accordion.Item>
    </Accordion>
  );
}
```

## Dialog/Modal Example

```typescript
interface DialogContextType {
  isOpen: boolean;
  open: () => void;
  close: () => void;
}

const DialogContext = createContext<DialogContextType | undefined>(undefined);

const useDialog = () => {
  const context = useContext(DialogContext);
  if (!context) {
    throw new Error('useDialog must be used inside <Dialog>');
  }
  return context;
};

export function Dialog({ children }: { children: React.ReactNode }) {
  const [isOpen, setIsOpen] = useState(false);

  return (
    <DialogContext.Provider
      value={{ isOpen, open: () => setIsOpen(true), close: () => setIsOpen(false) }}
    >
      {children}
    </DialogContext.Provider>
  );
}

Dialog.Trigger = function DialogTrigger({
  children,
}: {
  children: React.ReactNode;
}) {
  const { open } = useDialog();
  return <button onClick={open}>{children}</button>;
};

Dialog.Content = function DialogContent({
  children,
}: {
  children: React.ReactNode;
}) {
  const { isOpen, close } = useDialog();

  if (!isOpen) return null;

  return (
    <div className="dialog-overlay" onClick={close}>
      <div className="dialog-content" onClick={(e) => e.stopPropagation()}>
        {children}
        <button className="dialog-close" onClick={close}>
          ✕
        </button>
      </div>
    </div>
  );
};

Dialog.Header = function DialogHeader({
  children,
}: {
  children: React.ReactNode;
}) {
  return <div className="dialog-header">{children}</div>;
};

Dialog.Body = function DialogBody({
  children,
}: {
  children: React.ReactNode;
}) {
  return <div className="dialog-body">{children}</div>;
};

Dialog.Footer = function DialogFooter({
  children,
}: {
  children: React.ReactNode;
}) {
  return <div className="dialog-footer">{children}</div>;
};

Dialog.Action = function DialogAction({
  onClick,
  children,
}: {
  onClick: () => void;
  children: React.ReactNode;
}) {
  const { close } = useDialog();

  return (
    <button
      onClick={() => {
        onClick();
        close();
      }}
    >
      {children}
    </button>
  );
};

// Usage
export function DialogDemo() {
  const handleConfirm = () => {
    console.log('Confirmed!');
  };

  return (
    <Dialog>
      <Dialog.Trigger>Open Dialog</Dialog.Trigger>

      <Dialog.Content>
        <Dialog.Header>
          <h2>Confirm Action</h2>
        </Dialog.Header>

        <Dialog.Body>
          <p>Are you sure you want to proceed?</p>
        </Dialog.Body>

        <Dialog.Footer>
          <button>Cancel</button>
          <Dialog.Action onClick={handleConfirm}>
            Confirm
          </Dialog.Action>
        </Dialog.Footer>
      </Dialog.Content>
    </Dialog>
  );
}
```

## Select/Dropdown Example

```typescript
interface SelectContextType {
  isOpen: boolean;
  toggleOpen: () => void;
  selectedValue: string | null;
  selectValue: (value: string) => void;
}

const SelectContext = createContext<SelectContextType | undefined>(undefined);

const useSelect = () => {
  const context = useContext(SelectContext);
  if (!context) {
    throw new Error('useSelect must be used inside <Select>');
  }
  return context;
};

export function Select({
  children,
  onValueChange,
}: {
  children: React.ReactNode;
  onValueChange?: (value: string) => void;
}) {
  const [isOpen, setIsOpen] = useState(false);
  const [selectedValue, setSelectedValue] = useState<string | null>(null);

  const handleSelectValue = (value: string) => {
    setSelectedValue(value);
    setIsOpen(false);
    onValueChange?.(value);
  };

  return (
    <SelectContext.Provider
      value={{
        isOpen,
        toggleOpen: () => setIsOpen(!isOpen),
        selectedValue,
        selectValue: handleSelectValue,
      }}
    >
      <div className="select">{children}</div>
    </SelectContext.Provider>
  );
}

Select.Trigger = function SelectTrigger({
  children,
}: {
  children?: React.ReactNode;
}) {
  const { isOpen, toggleOpen, selectedValue } = useSelect();

  return (
    <button
      className={`select-trigger ${isOpen ? 'open' : ''}`}
      onClick={toggleOpen}
    >
      {children || selectedValue || 'Select...'}
    </button>
  );
};

Select.Content = function SelectContent({
  children,
}: {
  children: React.ReactNode;
}) {
  const { isOpen } = useSelect();

  return isOpen ? (
    <div className="select-content">{children}</div>
  ) : null;
};

Select.Item = function SelectItem({
  value,
  children,
}: {
  value: string;
  children: React.ReactNode;
}) {
  const { selectValue, selectedValue } = useSelect();

  return (
    <button
      className={`select-item ${selectedValue === value ? 'selected' : ''}`}
      onClick={() => selectValue(value)}
    >
      {children}
    </button>
  );
};

// Usage
export function SelectDemo() {
  const [selected, setSelected] = useState('');

  return (
    <Select onValueChange={setSelected}>
      <Select.Trigger>Choose an option</Select.Trigger>
      <Select.Content>
        <Select.Item value="option1">Option 1</Select.Item>
        <Select.Item value="option2">Option 2</Select.Item>
        <Select.Item value="option3">Option 3</Select.Item>
      </Select.Content>
    </Select>
  );
}
```

## When to Use Compound Components

✅ **Use when:**
- Component has multiple related parts
- Parts can be reordered
- Flexibility is important
- Want to avoid "props explosion"

❌ **Don't use when:**
- Simple component with few options
- Structure is always the same
- Props are straightforward

---

## Best Practices

1. **Use Context carefully** — don't overuse for simple state
2. **Type your context** — prevent undefined errors
3. **Provide helpful errors** — when hooks used outside Context
4. **Keep components simple** — each sub-component does one thing
5. **Document structure** — show expected nesting in comments

## Common Libraries Using Compound Components

- **Radix UI** — unstyled components with compound pattern

## Anti-patterns

❌ **Don't:**
```typescript
// Too much nested context
<Tabs>
  <Tabs.List>
    <Tabs.Item>
      <Icon>
        <InnerContext>
          {/* Too deep */}
        </InnerContext>
      </Icon>
    </Tabs.Item>
  </Tabs.List>
</Tabs>

// Context for everything
const [everything] = useState(...);
// 50 values in one context provider
```

✅ **Do:**
```typescript
// Flat structure
<Tabs>
  <Tabs.List>
    <Tabs.Item>Home</Tabs.Item>
  </Tabs.List>
  <Tabs.Content>Content</Tabs.Content>
</Tabs>

// Focused context
// Each Context has one responsibility
```
