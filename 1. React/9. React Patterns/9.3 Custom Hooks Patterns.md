# Custom Hooks Patterns

[[9. React Patterns.md]]

Custom hooks — это способ **извлечь и переиспользовать логику** между компонентами. Паттерны для создания мощных, композируемых hooks.

---

## Golden Rules of Hooks

```typescript
// ✅ DO: Call at top level
function Component() {
  const data = useCustomHook();        // ✅ Top level
  return <div>{data}</div>;
}

// ❌ DON'T: Call conditionally
function Component() {
  if (condition) {
    const data = useCustomHook();      // ❌ Conditional!
  }
}

// ❌ DON'T: Call in loops
function Component() {
  for (let i = 0; i < 5; i++) {
    const data = useCustomHook();      // ❌ In loop!
  }
}
```

## Pattern 1: Simple Data Fetching Hook

```typescript
interface FetchState<T> {
  data: T | null;
  loading: boolean;
  error: Error | null;
}

function useFetch<T>(url: string): FetchState<T> {
  const [state, setState] = useState<FetchState<T>>({
    data: null,
    loading: false,
    error: null,
  });

  useEffect(() => {
    setState(prev => ({ ...prev, loading: true }));

    fetch(url)
      .then(res => res.json())
      .then(data => {
        setState({ data, loading: false, error: null });
      })
      .catch(error => {
        setState({ data: null, loading: false, error });
      });
  }, [url]);

  return state;
}

// Usage
function UsersList() {
  const { data: users, loading, error } = useFetch<User[]>('/api/users');

  if (loading) return <div>Loading...</div>;
  if (error) return <div>Error: {error.message}</div>;
  return <ul>{users?.map(u => <li key={u.id}>{u.name}</li>)}</ul>;
}
```

## Pattern 2: useAsync - Advanced Data Fetching

```typescript
interface AsyncState<T> {
  status: 'idle' | 'pending' | 'success' | 'error';
  data: T | null;
  error: Error | null;
}

interface UseAsyncOptions {
  immediate?: boolean;
  onSuccess?: (data: any) => void;
  onError?: (error: Error) => void;
}

function useAsync<T>(
  asyncFn: () => Promise<T>,
  options: UseAsyncOptions = {}
): AsyncState<T> & { execute: () => Promise<T> } {
  const { immediate = true, onSuccess, onError } = options;

  const [state, setState] = useState<AsyncState<T>>({
    status: 'idle',
    data: null,
    error: null,
  });

  const execute = useCallback(async () => {
    setState({ status: 'pending', data: null, error: null });
    try {
      const data = await asyncFn();
      setState({ status: 'success', data, error: null });
      onSuccess?.(data);
      return data;
    } catch (error) {
      const err = error instanceof Error ? error : new Error(String(error));
      setState({ status: 'error', data: null, error: err });
      onError?.(err);
      throw err;
    }
  }, [asyncFn, onSuccess, onError]);

  useEffect(() => {
    if (immediate) {
      execute();
    }
  }, [execute, immediate]);

  return { ...state, execute };
}

// Usage
function UserProfile({ userId }: { userId: string }) {
  const { status, data: user, execute } = useAsync(
    async () => {
      const res = await fetch(`/api/users/${userId}`);
      return res.json();
    },
    {
      onSuccess: (user) => console.log('User loaded:', user),
      onError: (error) => console.error('Failed:', error),
    }
  );

  return (
    <>
      {status === 'pending' && <div>Loading...</div>}
      {status === 'error' && <div>Error!</div>}
      {status === 'success' && <div>{user.name}</div>}
      <button onClick={() => execute()}>Retry</button>
    </>
  );
}
```

## Pattern 3: useToggle

```typescript
function useToggle(initialValue: boolean = false): [boolean, () => void] {
  const [value, setValue] = useState(initialValue);
  const toggle = useCallback(() => setValue(v => !v), []);
  return [value, toggle];
}

// Usage
function Menu() {
  const [isOpen, toggleMenu] = useToggle();

  return (
    <>
      <button onClick={toggleMenu}>Menu</button>
      {isOpen && <nav>Menu items...</nav>}
    </>
  );
}
```

## Pattern 4: useDebounce

```typescript
function useDebounce<T>(value: T, delay: number): T {
  const [debouncedValue, setDebouncedValue] = useState(value);

  useEffect(() => {
    const timer = setTimeout(() => {
      setDebouncedValue(value);
    }, delay);

    return () => clearTimeout(timer);
  }, [value, delay]);

  return debouncedValue;
}

// Usage: Search with debounce
function SearchUsers() {
  const [query, setQuery] = useState('');
  const debouncedQuery = useDebounce(query, 500);
  const { data: results } = useFetch(
    `/api/search?q=${debouncedQuery}`
  );

  return (
    <>
      <input
        value={query}
        onChange={(e) => setQuery(e.target.value)}
        placeholder="Search..."
      />
      {results?.map(r => <div key={r.id}>{r.name}</div>)}
    </>
  );
}
```

## Pattern 5: useThrottle

```typescript
function useThrottle<T>(value: T, interval: number): T {
  const [throttledValue, setThrottledValue] = useState(value);
  const lastUpdated = useRef<number | null>(null);

  useEffect(() => {
    const now = Date.now();

    if (lastUpdated.current === null || now >= lastUpdated.current + interval) {
      lastUpdated.current = now;
      setThrottledValue(value);
    } else {
      const timer = setTimeout(() => {
        lastUpdated.current = Date.now();
        setThrottledValue(value);
      }, interval - (now - (lastUpdated.current || now)));

      return () => clearTimeout(timer);
    }
  }, [value, interval]);

  return throttledValue;
}

// Usage: Window resize
function ResponsiveComponent() {
  const [windowSize, setWindowSize] = useState<[number, number]>([
    window.innerWidth,
    window.innerHeight,
  ]);

  const throttledWindowSize = useThrottle(windowSize, 200);

  useEffect(() => {
    const handleResize = () => {
      setWindowSize([window.innerWidth, window.innerHeight]);
    };

    window.addEventListener('resize', handleResize);
    return () => window.removeEventListener('resize', handleResize);
  }, []);

  return <div>Window: {throttledWindowSize.join('x')}</div>;
}
```

## Pattern 6: usePrevious

```typescript
function usePrevious<T>(value: T): T | undefined {
  const ref = useRef<T>();

  useEffect(() => {
    ref.current = value;
  }, [value]);

  return ref.current;
}

// Usage
function Counter() {
  const [count, setCount] = useState(0);
  const prevCount = usePrevious(count);

  return (
    <>
      <p>Now: {count}, before: {prevCount}</p>
      <button onClick={() => setCount(c => c + 1)}>Increment</button>
    </>
  );
}
```

## Pattern 7: useLocalStorage

```typescript
function useLocalStorage<T>(key: string, initialValue: T) {
  const [storedValue, setStoredValue] = useState<T>(() => {
    const item = window.localStorage.getItem(key);
    return item ? JSON.parse(item) : initialValue;
  });

  const setValue = useCallback(
    (value: T | ((val: T) => T)) => {
      const valueToStore = value instanceof Function ? value(storedValue) : value;
      setStoredValue(valueToStore);
      window.localStorage.setItem(key, JSON.stringify(valueToStore));
    },
    [key, storedValue]
  );

  return [storedValue, setValue] as const;
}

// Usage
function Theme() {
  const [theme, setTheme] = useLocalStorage<'light' | 'dark'>('theme', 'light');

  return (
    <>
      <p>Current: {theme}</p>
      <button onClick={() => setTheme(theme === 'light' ? 'dark' : 'light')}>
        Toggle
      </button>
    </>
  );
}
```

## Pattern 8: useCallback Optimization

```typescript
function useCallback<T extends (...args: any[]) => any>(
  callback: T,
  deps: React.DependencyList
): T {
  // Already exists in React! But here's the concept:
  // Return memoized version of callback
  const ref = useRef(callback);

  useEffect(() => {
    ref.current = callback;
  }, deps);

  return useCallback((...args: any[]) => ref.current(...args), []) as T;
}

// Usage
function Parent() {
  const handleClick = useCallback(() => {
    console.log('Clicked!');
  }, []);

  return <Child onClick={handleClick} />; // Won't re-render Child
}
```

## Pattern 9: useReducer (Complex State)

```typescript
interface State {
  count: number;
  user: User | null;
  error: Error | null;
}

type Action =
  | { type: 'INCREMENT' }
  | { type: 'DECREMENT' }
  | { type: 'SET_USER'; payload: User }
  | { type: 'SET_ERROR'; payload: Error };

function reducer(state: State, action: Action): State {
  switch (action.type) {
    case 'INCREMENT':
      return { ...state, count: state.count + 1 };
    case 'DECREMENT':
      return { ...state, count: state.count - 1 };
    case 'SET_USER':
      return { ...state, user: action.payload, error: null };
    case 'SET_ERROR':
      return { ...state, error: action.payload };
    default:
      return state;
  }
}

// Usage
function App() {
  const [state, dispatch] = useReducer(reducer, {
    count: 0,
    user: null,
    error: null,
  });

  return (
    <>
      <button onClick={() => dispatch({ type: 'INCREMENT' })}>+</button>
      {state.count}
    </>
  );
}
```

## Pattern 10: useMount / useUnmount

```typescript
function useMount(callback: () => void) {
  useEffect(() => {
    callback();
  }, []);
}

function useUnmount(callback: () => void) {
  useEffect(() => {
    return () => callback();
  }, []);
}

// Usage
function VideoPlayer() {
  useMount(() => {
    console.log('Video mounted');
  });

  useUnmount(() => {
    console.log('Video unmounted - cleanup');
  });

  return <video src="..." />;
}
```

## Composing Multiple Hooks

```typescript
// Hook 1
function useUser(userId: string) {
  return useFetch<User>(`/api/users/${userId}`);
}

// Hook 2: Builds on Hook 1
function useUserWithCache(userId: string) {
  const [cache, setCache] = useState<Record<string, User>>({});
  const { data, ...rest } = useUser(userId);

  return {
    ...rest,
    data: cache[userId] || data,
    cacheUser: (user: User) => setCache(prev => ({ ...prev, [userId]: user })),
  };
}

// Hook 3: Builds on Hook 2
function useUsersList(userIds: string[]) {
  return userIds.map(id => useUserWithCache(id));
}
```

## Best Practices

✅ **DO:**
- Name with `use` prefix
- Export hook separately from components
- Return data in predictable order
- Use TypeScript for better types
- Document expected behavior

❌ **DON'T:**
- Don't call hooks conditionally
- Don't call hooks in loops
- Don't call hooks in callbacks
- Don't ignore dependency arrays
- Don't create too many hooks (compose instead)

## Testing Custom Hooks

```typescript
import { renderHook, act, waitFor } from '@testing-library/react';

describe('useAsync', () => {
  it('loads data on mount', async () => {
    const mockFetch = jest.fn(() =>
      Promise.resolve({ id: 1, name: 'John' })
    );

    const { result } = renderHook(() => useAsync(mockFetch));

    expect(result.current.status).toBe('pending');

    await waitFor(() => {
      expect(result.current.status).toBe('success');
    });

    expect(result.current.data).toEqual({ id: 1, name: 'John' });
  });

  it('handles errors', async () => {
    const mockFetch = jest.fn(() =>
      Promise.reject(new Error('Network error'))
    );

    const { result } = renderHook(() => useAsync(mockFetch));

    await waitFor(() => {
      expect(result.current.status).toBe('error');
    });

    expect(result.current.error?.message).toBe('Network error');
  });
});
```
