# React Router Basics

[[3. Routing.md]]

---

React Router v6+ provides declarative routing for React apps. Use it for SPA navigation without full page reloads.

## Installation

```bash
npm install react-router-dom
```

## Basic Setup

```typescript
// main.tsx or App.tsx
import { BrowserRouter, Routes, Route } from 'react-router-dom';

function App() {
  return (
    <BrowserRouter>
      <Routes>
        <Route path="/" element={<HomePage />} />
        <Route path="/about" element={<AboutPage />} />
        <Route path="/users" element={<UsersPage />} />
      </Routes>
    </BrowserRouter>
  );
}
```

**When to use BrowserRouter:** Standard web apps with HTML5 History API (clean URLs like `/about`). Use `HashRouter` only for static hosting without server config.

---

## Routes and Route

```typescript
// Routes: picks first matching Route
// Route: path + element to render

<Routes>
  <Route path="/" element={<HomePage />} />
  <Route path="/products" element={<ProductsPage />} />
  <Route path="/products/:id" element={<ProductDetail />} />
  <Route path="*" element={<NotFoundPage />} />
</Routes>
```

**Path matching:** React Router v6 uses exact matching by default. `*` is a wildcard for 404.

## Link and NavLink

### Link

```typescript
import { Link } from 'react-router-dom';

function Navigation() {
  return (
    <nav>
      <Link to="/">Home</Link>
      <Link to="/about">About</Link>
      <Link to="/users/123">User 123</Link>
      <Link to="/search?q=react">Search</Link>
    </nav>
  );
}
```

**Prefer Link over `<a>`:** Link prevents full page reload and uses client-side navigation.

### NavLink

```typescript
import { NavLink } from 'react-router-dom';

function NavBar() {
  return (
    <nav>
      <NavLink
        to="/"
        className={({ isActive }) => (isActive ? 'nav-link active' : 'nav-link')}
      >
        Home
      </NavLink>
      <NavLink
        to="/about"
        style={({ isActive }) => ({ fontWeight: isActive ? 'bold' : 'normal' })}
      >
        About
      </NavLink>
    </nav>
  );
}
```

**When to use NavLink:** Navigation menus where you need to highlight the current route.

## useNavigate

Programmatic navigation (after form submit, login, etc.):

```typescript
import { useNavigate } from 'react-router-dom';

function LoginForm() {
  const navigate = useNavigate();

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    await login(credentials);
    navigate('/dashboard');
  };

  return (
    <form onSubmit={handleSubmit}>
      {/* ... */}
      <button type="submit">Login</button>
    </form>
  );
}
```

**Navigate with state:**

```typescript
navigate('/dashboard', { state: { from: 'login' } });
navigate(-1);  // Go back
navigate(1);   // Go forward
navigate('/users', { replace: true });  // Replace history entry
```

**When to use:** Any navigation triggered by code (not by user clicking a link).

## useParams

Access dynamic route parameters:

```typescript
import { useParams } from 'react-router-dom';

// Route: /users/:userId
function UserProfile() {
  const { userId } = useParams<{ userId: string }>();

  if (!userId) return <div>Invalid user</div>;

  return <div>User ID: {userId}</div>;
}

// Route: /posts/:postId/comments/:commentId
function CommentDetail() {
  const { postId, commentId } = useParams<{
    postId: string;
    commentId: string;
  }>();

  return (
    <div>
      Post: {postId}, Comment: {commentId}
    </div>
  );
}
```

**TypeScript tip:** Use generic `useParams<{ paramName: string }>()` for type safety. Params are always strings.

## useSearchParams

Read and update URL query string:

```typescript
import { useSearchParams } from 'react-router-dom';

function SearchPage() {
  const [searchParams, setSearchParams] = useSearchParams();

  const query = searchParams.get('q') ?? '';
  const page = searchParams.get('page') ?? '1';

  const handleSearch = (newQuery: string) => {
    setSearchParams({ q: newQuery, page: '1' });
  };

  return (
    <div>
      <input
        value={query}
        onChange={(e) => handleSearch(e.target.value)}
      />
      <p>Page: {page}</p>
    </div>
  );
}
```

**Common patterns:**

```typescript
// Get single param
const id = searchParams.get('id');

// Get all values for a key
const tags = searchParams.getAll('tag');

// Check if param exists
const hasFilter = searchParams.has('filter');

// Update params (replaces current)
setSearchParams({ sort: 'date', order: 'desc' });

// Merge with existing
setSearchParams((prev) => {
  prev.set('page', '2');
  return prev;
});
```

**When to use:** Filters, search, pagination, shareable URLs.

## useLocation

Access current location object:

```typescript
import { useLocation } from 'react-router-dom';

function Breadcrumb() {
  const location = useLocation();

  // location.pathname  → "/users/123"
  // location.search    → "?tab=posts"
  // location.hash      → "#section"
  // location.state     → { from: 'login' } (passed via navigate)

  const pathSegments = location.pathname.split('/').filter(Boolean);

  return (
    <nav>
      {pathSegments.map((seg, i) => (
        <span key={i}>{seg}</span>
      ))}
    </nav>
  );
}
```

**Passing state via navigate:**

```typescript
// Sender
navigate('/checkout', { state: { cartItems } });

// Receiver
function CheckoutPage() {
  const { state } = useLocation();
  const cartItems = state?.cartItems ?? [];
}
```

**When to use:** Breadcrumbs, analytics, reading `state` passed from `navigate()`.

## Navigate Component

Declarative redirect:

```typescript
import { Navigate } from 'react-router-dom';

function OldPage() {
  return <Navigate to="/new-page" replace />;
}

function ConditionalRedirect() {
  const isLoggedIn = useAuth();

  if (!isLoggedIn) {
    return <Navigate to="/login" state={{ from: location }} replace />;
  }

  return <Dashboard />;
}
```

**When to use:** Simple redirects, auth guards. Prefer `useNavigate` for redirects inside event handlers.

## Complete Example

```typescript
// App.tsx
import { BrowserRouter, Routes, Route, Link } from 'react-router-dom';

function App() {
  return (
    <BrowserRouter>
      <nav>
        <Link to="/">Home</Link>
        <Link to="/users">Users</Link>
      </nav>
      <Routes>
        <Route path="/" element={<HomePage />} />
        <Route path="/users" element={<UsersPage />} />
        <Route path="/users/:id" element={<UserDetailPage />} />
        <Route path="*" element={<NotFound />} />
      </Routes>
    </BrowserRouter>
  );
}

function UserDetailPage() {
  const { id } = useParams<{ id: string }>();
  const navigate = useNavigate();

  return (
    <div>
      <h1>User {id}</h1>
      <button onClick={() => navigate(-1)}>Back</button>
    </div>
  );
}
```

## Quick Reference

| Hook/Component | Purpose |
| :-- | :-- |
| **BrowserRouter** | Wraps app, enables routing |
| **Routes / Route** | Define route → component mapping |
| **Link** | Declarative navigation link |
| **NavLink** | Link with active state |
| **useNavigate** | Programmatic navigation |
| **useParams** | Dynamic route params |
| **useSearchParams** | Query string read/write |
| **useLocation** | Current path, search, state |
| **Navigate** | Declarative redirect |
