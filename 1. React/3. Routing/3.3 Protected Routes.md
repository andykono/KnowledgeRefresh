# Protected Routes

[[3. Routing.md]]

---

Protected routes restrict access based on authentication or roles. Use wrapper components and redirects to enforce access control.

## Basic Auth Guard

Redirect unauthenticated users to login:

```typescript
import { Navigate, useLocation } from 'react-router-dom';

interface ProtectedRouteProps {
  children: React.ReactNode;
  isAuthenticated: boolean;
}

function ProtectedRoute({ children, isAuthenticated }: ProtectedRouteProps) {
  const location = useLocation();

  if (!isAuthenticated) {
    return <Navigate to="/login" state={{ from: location }} replace />;
  }

  return <>{children}</>;
}
```

**Usage:**

```typescript
<Route
  path="/dashboard"
  element={
    <ProtectedRoute isAuthenticated={!!user}>
      <DashboardPage />
    </ProtectedRoute>
  }
/>
```

**Why `state={{ from: location }}`:** Lets the login page redirect back after successful auth.

## Redirect After Login

```typescript
import { useNavigate, useLocation } from 'react-router-dom';

function LoginPage() {
  const navigate = useNavigate();
  const location = useLocation();

  const from = (location.state as { from?: { pathname: string } })?.from?.pathname ?? '/';

  const handleLogin = async () => {
    await login(credentials);
    navigate(from, { replace: true });
  };

  return (
    <form onSubmit={handleLogin}>
      {/* ... */}
    </form>
  );
}
```

## Role-Based Access

Restrict routes by user role:

```typescript
import { Navigate, useLocation } from 'react-router-dom';

type Role = 'user' | 'admin';

interface RoleProtectedRouteProps {
  children: React.ReactNode;
  allowedRoles: Role[];
  userRole: Role | null;
}

function RoleProtectedRoute({
  children,
  allowedRoles,
  userRole,
}: RoleProtectedRouteProps) {
  const location = useLocation();

  if (!userRole) {
    return <Navigate to="/login" state={{ from: location }} replace />;
  }

  if (!allowedRoles.includes(userRole)) {
    return <Navigate to="/unauthorized" replace />;
  }

  return <>{children}</>;
}
```

**Usage:**

```typescript
<Route
  path="/admin"
  element={
    <RoleProtectedRoute allowedRoles={['admin']} userRole={user?.role ?? null}>
      <AdminDashboard />
    </RoleProtectedRoute>
  }
/>
```

## Combined Auth + Role Component

```typescript
interface ProtectedRouteProps {
  children: React.ReactNode;
  requireAuth?: boolean;
  allowedRoles?: Role[];
  user: User | null;
}

function ProtectedRoute({
  children,
  requireAuth = true,
  allowedRoles,
  user,
}: ProtectedRouteProps) {
  const location = useLocation();

  if (requireAuth && !user) {
    return <Navigate to="/login" state={{ from: location }} replace />;
  }

  if (allowedRoles && user && !allowedRoles.includes(user.role)) {
    return <Navigate to="/unauthorized" replace />;
  }

  return <>{children}</>;
}
```

## Layout-Based Protection

Protect entire layout and its children:

```typescript
<Routes>
  <Route path="/login" element={<LoginPage />} />
  <Route
    path="/"
    element={
      <ProtectedRoute isAuthenticated={!!user}>
        <AppLayout />
      </ProtectedRoute>
    }
  >
    <Route index element={<HomePage />} />
    <Route path="dashboard" element={<DashboardPage />} />
    <Route
      path="admin"
      element={
        <RoleProtectedRoute allowedRoles={['admin']} userRole={user?.role ?? null}>
          <AdminPage />
        </RoleProtectedRoute>
      }
    />
  </Route>
</Routes>
```

## Auth Context Integration

```typescript
import { createContext, useContext } from 'react';
import { Navigate, useLocation } from 'react-router-dom';

interface AuthContextValue {
  user: User | null;
  isLoading: boolean;
}

const AuthContext = createContext<AuthContextValue | null>(null);

function useAuth() {
  const ctx = useContext(AuthContext);
  if (!ctx) throw new Error('useAuth must be used within AuthProvider');
  return ctx;
}

function ProtectedRoute({ children }: { children: React.ReactNode }) {
  const { user, isLoading } = useAuth();
  const location = useLocation();

  if (isLoading) {
    return <div>Loading...</div>;
  }

  if (!user) {
    return <Navigate to="/login" state={{ from: location }} replace />;
  }

  return <>{children}</>;
}
```

## Public-Only Routes

Redirect logged-in users away from login/register:

```typescript
function PublicOnlyRoute({ children }: { children: React.ReactNode }) {
  const { user } = useAuth();
  const location = useLocation();

  if (user) {
    const from = (location.state as { from?: { pathname: string } })?.from?.pathname ?? '/';
    return <Navigate to={from} replace />;
  }

  return <>{children}</>;
}

// Usage
<Route
  path="/login"
  element={
    <PublicOnlyRoute>
      <LoginPage />
    </PublicOnlyRoute>
  }
/>
```

## useNavigate in Guards

Avoid using `useNavigate` in render for redirects. Use `<Navigate>` or a wrapper component:

```typescript
// ❌ BAD: Side effect in render
function ProtectedRoute({ children }) {
  const navigate = useNavigate();
  if (!user) {
    navigate('/login');  // Causes issues
    return null;
  }
  return children;
}

// ✅ GOOD: Declarative redirect
function ProtectedRoute({ children }) {
  if (!user) {
    return <Navigate to="/login" replace />;
  }
  return <>{children}</>;
}
```

## Loading State

Show a loading UI while auth is being checked:

```typescript
function ProtectedRoute({ children }: { children: React.ReactNode }) {
  const { user, isLoading } = useAuth();
  const location = useLocation();

  if (isLoading) {
    return <PageLoader />;
  }

  if (!user) {
    return <Navigate to="/login" state={{ from: location }} replace />;
  }

  return <>{children}</>;
}
```

## Unauthorized Page

```typescript
function UnauthorizedPage() {
  const navigate = useNavigate();

  return (
    <div>
      <h1>Access Denied</h1>
      <p>You don't have permission to view this page.</p>
      <button onClick={() => navigate(-1)}>Go Back</button>
      <Link to="/">Home</Link>
    </div>
  );
}
```

## Quick Reference

| Pattern | Use Case |
| :-- | :-- |
| **ProtectedRoute** | Require login |
| **RoleProtectedRoute** | Require specific role(s) |
| **PublicOnlyRoute** | Hide login/register when logged in |
| **state={{ from }}** | Redirect back after login |
| **replace** | Avoid back-button returning to protected page |
