### useTodos.ts

```typescript
// features/TodoList/hooks/useTodos.ts

import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { todosApi } from '../api';
import type { TodoFilter } from '@/entities/Todo';

const QUERY_KEY = ['todos'] as const;

export function useTodos(filter: TodoFilter) {
  const queryClient = useQueryClient();

  const { data: todos = [], isLoading, error } = useQuery({
    queryKey: [...QUERY_KEY, filter],
    queryFn: () => todosApi.fetchTodos(filter),
  });

  const toggleMutation = useMutation({
    mutationFn: (id: string) => todosApi.toggleTodo(id),
    onSuccess: () => queryClient.invalidateQueries({ queryKey: QUERY_KEY }),
  });

  const deleteMutation = useMutation({
    mutationFn: (id: string) => todosApi.deleteTodo(id),
    onSuccess: () => queryClient.invalidateQueries({ queryKey: QUERY_KEY }),
  });

  return {
    todos,
    isLoading,
    error,
    toggleTodo: toggleMutation.mutate,
    deleteTodo: deleteMutation.mutate,
  };
}
```

### useCreateTodo.ts

```typescript
// features/TodoCreate/hooks/useCreateTodo.ts

import { useMutation, useQueryClient } from '@tanstack/react-query';
import { createTodoApi } from '../api';
import type { CreateTodoInput } from '@/entities/Todo';

export function useCreateTodo() {
  const queryClient = useQueryClient();

  const mutation = useMutation({
    mutationFn: (input: CreateTodoInput) => createTodoApi.createTodo(input),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['todos'] });
    },
  });

  return {
    createTodo: mutation.mutate,
    createTodoAsync: mutation.mutateAsync,
    isPending: mutation.isPending,
    error: mutation.error,
  };
}
```

### CreateForm.tsx

```typescript
// features/TodoCreate/ui/CreateForm.tsx

import { useForm } from 'react-hook-form';
import { zodResolver } from '@hookform/resolvers/zod';
import { z } from 'zod';
import { useCreateTodo } from '../hooks/useCreateTodo';
import type { CreateTodoInput } from '@/entities/Todo';

const schema = z.object({
  title: z.string().min(1, 'Title is required').max(200),
});

type FormData = z.infer<typeof schema>;

export function CreateForm() {
  const { createTodo, isPending } = useCreateTodo();

  const { register, handleSubmit, reset, formState: { errors } } = useForm<FormData>({
    resolver: zodResolver(schema),
    defaultValues: { title: '' },
  });

  const onSubmit = (data: CreateTodoInput) => {
    createTodo(data);
    reset();
  };

  return (
    <form onSubmit={handleSubmit(onSubmit)}>
      <input
        {...register('title')}
        placeholder="What needs to be done?"
        disabled={isPending}
      />
      {errors.title && <span>{errors.title.message}</span>}
      <button type="submit" disabled={isPending}>
        Add
      </button>
    </form>
  );
}
```

### TodoList.tsx (List.tsx)

```typescript
// features/TodoList/ui/List.tsx

import { TodoItem } from '@/entities/Todo';
import { useTodos } from '../hooks/useTodos';
import { useFilter } from '@/features/TodoFilter/hooks/useFilter';

export function TodoList() {
  const { filter } = useFilter();
  const { todos, isLoading, error, toggleTodo, deleteTodo } = useTodos(filter);

  if (isLoading) return <div>Loading...</div>;
  if (error) return <div>Error loading todos</div>;

  return (
    <ul>
      {todos.map((todo) => (
        <TodoItem
          key={todo.id}
          todo={todo}
          onToggle={toggleTodo}
          onDelete={deleteTodo}
        />
      ))}
    </ul>
  );
}
```

### useFilter.ts

```typescript
// features/TodoFilter/hooks/useFilter.ts

import { create } from 'zustand';
import type { TodoFilter } from '@/entities/Todo';

interface FilterState {
  filter: TodoFilter;
  setFilter: (filter: TodoFilter) => void;
}

export const useFilter = create<FilterState>((set) => ({
  filter: 'all',
  setFilter: (filter) => set({ filter }),
}));
```

### Filter.tsx

```typescript
// features/TodoFilter/ui/Filter.tsx

import { useFilter } from '../hooks/useFilter';
import type { TodoFilter } from '@/entities/Todo';

const FILTERS: { value: TodoFilter; label: string }[] = [
  { value: 'all', label: 'All' },
  { value: 'active', label: 'Active' },
  { value: 'completed', label: 'Completed' },
];

export function Filter() {
  const { filter, setFilter } = useFilter();

  return (
    <div className="filter">
      {FILTERS.map(({ value, label }) => (
        <button
          key={value}
          onClick={() => setFilter(value)}
          className={filter === value ? 'active' : ''}
        >
          {label}
        </button>
      ))}
    </div>
  );
}
```

### App.tsx

```typescript
// app/App.tsx

import { QueryClient, QueryClientProvider } from '@tanstack/react-query';
import { CreateForm } from '@/features/TodoCreate';
import { TodoList } from '@/features/TodoList';
import { Filter } from '@/features/TodoFilter';

const queryClient = new QueryClient();

export function App() {
  return (
    <QueryClientProvider client={queryClient}>
      <div className="todo-app">
        <h1>Todos</h1>
        <CreateForm />
        <Filter />
        <TodoList />
      </div>
    </QueryClientProvider>
  );
}
```

## Key Patterns Used

- **FSD Architecture:** Organize by features
- **Custom Hooks:** Reuse data fetching logic
- **Compound Components:** Complex UI elements
- **Context API / Zustand:** Global filter state

## Patterns Demonstrated

✅ Factory Method — Creating different filter strategies
✅ Strategy Pattern — Different sorting algorithms
✅ Observer Pattern — Todo list updates
✅ Facade — Simple API interface
✅ Custom Hooks — Data fetching abstraction
