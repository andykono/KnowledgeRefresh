## Key Patterns Used

- **Strategy:** Chart rendering (line, bar, pie strategies)
- **Compound Components:** Dashboard widgets
- **Custom Hooks:** useChartData for data fetching
- **Memoization:** React.memo, useMemo for performance

## Chart Strategy Pattern

```typescript
// entities/Chart/model/Chart.types.ts
export interface ChartDataPoint {
  label: string;
  value: number;
  color?: string;
}

export interface ChartConfig {
  type: 'line' | 'bar' | 'pie';
  title: string;
  data: ChartDataPoint[];
}

// Strategy interface for different chart renderers
export interface ChartRenderer {
  render: (data: ChartDataPoint[], config: ChartConfig) => React.ReactNode;
}
```

```typescript
// entities/Chart/ui/ChartRenderer.tsx - Strategy selector
import { LineChart } from './LineChart';
import { BarChart } from './BarChart';
import { PieChart } from './PieChart';

const chartStrategies: Record<string, React.ComponentType<ChartProps>> = {
  line: LineChart,
  bar: BarChart,
  pie: PieChart,
};

export function ChartRenderer({ config }: { config: ChartConfig }) {
  const ChartComponent = chartStrategies[config.type];
  if (!ChartComponent) return null;
  return <ChartComponent data={config.data} title={config.title} />;
}
```

## useChartData Custom Hook (TanStack Query)

```typescript
// features/Dashboard/hooks/useChartData.ts
import { useQuery } from '@tanstack/react-query';
import { analyticsApi } from '@/features/Analytics/api/analyticsApi';

interface UseChartDataParams {
  metric: string;
  dateRange: { start: Date; end: Date };
  chartType: 'line' | 'bar' | 'pie';
}

export function useChartData({ metric, dateRange, chartType }: UseChartDataParams) {
  const queryKey = ['chartData', metric, dateRange.start.toISOString(), dateRange.end.toISOString(), chartType];

  const { data, isLoading, error, refetch } = useQuery({
    queryKey,
    queryFn: () => analyticsApi.getChartData(metric, dateRange),
    staleTime: 30 * 1000,        // 30 seconds
    refetchInterval: 60 * 1000,  // Real-time: refetch every minute
  });

  return { data, isLoading, error, refetch };
}
```

## Compound Components: Dashboard Widget

```typescript
// features/Dashboard/ui/DashboardWidget.tsx
import { memo } from 'react';
import { ChartRenderer } from '@/entities/Chart/ui/ChartRenderer';

interface DashboardWidgetProps {
  title: string;
  chartConfig: ChartConfig;
  children?: React.ReactNode;
}

const DashboardWidgetRoot = memo(function DashboardWidget({
  title,
  chartConfig,
  children,
}: DashboardWidgetProps) {
  return (
    <div className="dashboard-widget">
      <div className="widget-header">
        <h3>{title}</h3>
        {children}
      </div>
      <div className="widget-content">
        <ChartRenderer config={chartConfig} />
      </div>
    </div>

// Compound sub-components
DashboardWidgetRoot.Filters = function WidgetFilters({ children }: { children: React.ReactNode }) {
  return <div className="widget-filters">{children}</div>;
};

DashboardWidgetRoot.Loading = function WidgetLoading() {
  return <div className="widget-loading">Loading...</div>;
};

export const DashboardWidget = DashboardWidgetRoot;
```

## Memoized Chart Components

```typescript
// entities/Chart/ui/LineChart.tsx
import { memo, useMemo } from 'react';

interface LineChartProps {
  data: ChartDataPoint[];
  title: string;
}

export const LineChart = memo(function LineChart({ data, title }: LineChartProps) {
  const chartConfig = useMemo(() => ({
    labels: data.map((d) => d.label),
    datasets: [{
      label: title,
      data: data.map((d) => d.value),
      borderColor: 'rgb(75, 192, 192)',
      tension: 0.1,
    }],
  }), [data, title]);

  return (
    <div className="line-chart">
      <h4>{title}</h4>
      {/* Render with Chart.js, Recharts, or similar */}
      <canvas data-chart={JSON.stringify(chartConfig)} />
    </div>
  );
});
```

## Filter Controls & Date Range

```typescript
// features/Dashboard/ui/FilterControls.tsx
import { useState, useCallback } from 'react';
import { subDays } from 'date-fns';
import { DateRangePicker } from '@/shared/ui/DateRangePicker';

interface FilterControlsProps {
  onFilterChange: (filters: DashboardFilters) => void;
}

export function FilterControls({ onFilterChange }: FilterControlsProps) {
  const [dateRange, setDateRange] = useState({
    start: subDays(new Date(), 30),
    end: new Date(),
  });
  const [metric, setMetric] = useState('revenue');

  const handleApply = useCallback(() => {
    onFilterChange({ dateRange, metric });
  }, [dateRange, metric, onFilterChange]);

  return (
    <div className="filter-controls">
      <DateRangePicker
        start={dateRange.start}
        end={dateRange.end}
        onChange={setDateRange}
      />
      <select value={metric} onChange={(e) => setMetric(e.target.value)}>
        <option value="revenue">Revenue</option>
        <option value="users">Users</option>
        <option value="orders">Orders</option>
      </select>
      <button onClick={handleApply}>Apply</button>
    </div>
  );
}
```

## Responsive Dashboard Layout

```typescript
// widgets/Dashboard/DashboardLayout.tsx
import { useState } from 'react';
import { subDays } from 'date-fns';
import { useChartData } from '@/features/Dashboard/hooks/useChartData';
import { DashboardWidget } from '@/features/Dashboard/ui/DashboardWidget';
import { FilterControls } from '@/features/Dashboard/ui/FilterControls';

export function DashboardLayout() {
  const [filters, setFilters] = useState({
    dateRange: { start: subDays(new Date(), 30), end: new Date() },
    metric: 'revenue',
  });

  const revenueChart = useChartData({
    metric: filters.metric,
    dateRange: filters.dateRange,
    chartType: 'line',
  });

  const userChart = useChartData({
    metric: 'users',
    dateRange: filters.dateRange,
    chartType: 'bar',
  });

  const categoryChart = useChartData({
    metric: 'categories',
    dateRange: filters.dateRange,
    chartType: 'pie',
  });

  return (
    <div className="dashboard responsive-grid">
      <FilterControls onFilterChange={setFilters} />

      <div className="widget-row">
        <DashboardWidget
          title="Revenue Over Time"
          chartConfig={{
            type: 'line',
            title: 'Revenue',
            data: revenueChart.data ?? [],
          }}
        >
          <DashboardWidget.Filters>
            <span>Last 30 days</span>
          </DashboardWidget.Filters>
        </DashboardWidget>
      </div>

      <div className="widget-row two-columns">
        <DashboardWidget
          title="User Growth"
          chartConfig={{
            type: 'bar',
            title: 'Users',
            data: userChart.data ?? [],
          }}
        />
        <DashboardWidget
          title="Category Distribution"
          chartConfig={{
            type: 'pie',
            title: 'Categories',
            data: categoryChart.data ?? [],
          }}
        />
      </div>
    </div>
  );
}
```

## Analytics API

```typescript
// features/Analytics/api/analyticsApi.ts
import { httpClient } from '@/shared/api/httpClient';

export const analyticsApi = {
  async getChartData(
    metric: string,
    dateRange: { start: Date; end: Date }
  ): Promise<ChartDataPoint[]> {
    const { data } = await httpClient.get('/analytics/chart', {
      params: {
        metric,
        start: dateRange.start.toISOString(),
        end: dateRange.end.toISOString(),
      },
    });
    return data;
  },

  async getRealtimeMetrics(): Promise<Record<string, number>> {
    const { data } = await httpClient.get('/analytics/realtime');
    return data;
  },
};
```
