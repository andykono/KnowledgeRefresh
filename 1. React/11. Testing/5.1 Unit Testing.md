# 5.1 Unit Testing

[[5. Testing.md]]

Unit tests verify isolated units of code—pure functions, reducers, and custom hooks—without rendering UI or hitting external services.

## AAA Pattern (Arrange-Act-Assert)

Structure every test in three phases:

```typescript
describe('calculateDiscount', () => {
  it('applies 10% discount for orders over 100', () => {
    // Arrange
    const price = 150;
    const discountPercent = 10;

    // Act
    const result = calculateDiscount(price, discountPercent);

    // Assert
    expect(result).toBe(135);
  });
});
```

## Testing Pure Functions

```typescript
// utils/validation.ts
export function validateEmail(email: string): boolean {
  return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email);
}

export function formatCurrency(amount: number, currency = 'USD'): string {
  return new Intl.NumberFormat('en-US', {
    style: 'currency',
    currency,
  }).format(amount);
}
```

```typescript
// utils/validation.test.ts
import { describe, it, expect } from 'vitest';
import { validateEmail, formatCurrency } from './validation';

describe('validateEmail', () => {
  it('returns true for valid emails', () => {
    expect(validateEmail('user@example.com')).toBe(true);
    expect(validateEmail('a@b.co')).toBe(true);
  });

  it('returns false for invalid emails', () => {
    expect(validateEmail('')).toBe(false);
    expect(validateEmail('invalid')).toBe(false);
    expect(validateEmail('missing@domain')).toBe(false);
  });
});

describe('formatCurrency', () => {
  it('formats USD by default', () => {
    expect(formatCurrency(1234.56)).toBe('$1,234.56');
  });

  it('accepts custom currency', () => {
    expect(formatCurrency(100, 'EUR')).toContain('100');
  });
});
```

## Testing Reducers

```typescript
// store/cartReducer.ts
export type CartItem = { id: string; name: string; qty: number };
export type CartState = { items: CartItem[] };

export type CartAction =
  | { type: 'ADD_ITEM'; payload: CartItem }
  | { type: 'REMOVE_ITEM'; payload: string }
  | { type: 'UPDATE_QTY'; payload: { id: string; qty: number } };

export function cartReducer(state: CartState, action: CartAction): CartState {
  switch (action.type) {
    case 'ADD_ITEM':
      const existing = state.items.find((i) => i.id === action.payload.id);
      if (existing) {
        return {
          items: state.items.map((i) =>
            i.id === action.payload.id
              ? { ...i, qty: i.qty + action.payload.qty }
              : i
          ),
        };
      }
      return { items: [...state.items, action.payload] };
    case 'REMOVE_ITEM':
      return { items: state.items.filter((i) => i.id !== action.payload) };
    case 'UPDATE_QTY':
      return {
        items: state.items.map((i) =>
          i.id === action.payload.id ? { ...i, qty: action.payload.qty } : i
        ),
      };
    default:
      return state;
  }
}
```

```typescript
// store/cartReducer.test.ts
import { describe, it, expect } from 'vitest';
import { cartReducer, CartState } from './cartReducer';

describe('cartReducer', () => {
  const initialState: CartState = { items: [] };

  it('adds new item', () => {
    const state = cartReducer(initialState, {
      type: 'ADD_ITEM',
      payload: { id: '1', name: 'Widget', qty: 2 },
    });
    expect(state.items).toHaveLength(1);
    expect(state.items[0]).toEqual({ id: '1', name: 'Widget', qty: 2 });
  });

  it('increments qty when adding existing item', () => {
    const state = cartReducer(
      { items: [{ id: '1', name: 'Widget', qty: 1 }] },
      { type: 'ADD_ITEM', payload: { id: '1', name: 'Widget', qty: 3 } }
    );
    expect(state.items[0].qty).toBe(4);
  });

  it('removes item', () => {
    const state = cartReducer(
      { items: [{ id: '1', name: 'Widget', qty: 1 }] },
      { type: 'REMOVE_ITEM', payload: '1' }
    );
    expect(state.items).toHaveLength(0);
  });

  it('updates qty', () => {
    const state = cartReducer(
      { items: [{ id: '1', name: 'Widget', qty: 1 }] },
      { type: 'UPDATE_QTY', payload: { id: '1', qty: 5 } }
    );
    expect(state.items[0].qty).toBe(5);
  });
});
```

## Testing Custom Hooks with renderHook

```typescript
// hooks/useCounter.ts
import { useState, useCallback } from 'react';

export function useCounter(initial = 0) {
  const [count, setCount] = useState(initial);
  const increment = useCallback(() => setCount((c) => c + 1), []);
  const decrement = useCallback(() => setCount((c) => c - 1), []);
  const reset = useCallback(() => setCount(initial), [initial]);
  return { count, increment, decrement, reset };
}
```

```typescript
// hooks/useCounter.test.ts
import { describe, it, expect } from 'vitest';
import { renderHook, act } from '@testing-library/react';
import { useCounter } from './useCounter';

describe('useCounter', () => {
  it('starts at initial value', () => {
    const { result } = renderHook(() => useCounter(5));
    expect(result.current.count).toBe(5);
  });

  it('increments', () => {
    const { result } = renderHook(() => useCounter(0));
    act(() => result.current.increment());
    expect(result.current.count).toBe(1);
    act(() => result.current.increment());
    expect(result.current.count).toBe(2);
  });

  it('decrements', () => {
    const { result } = renderHook(() => useCounter(3));
    act(() => result.current.decrement());
    expect(result.current.count).toBe(2);
  });

  it('resets to initial', () => {
    const { result } = renderHook(() => useCounter(10));
    act(() => result.current.increment());
    act(() => result.current.reset());
    expect(result.current.count).toBe(10);
  });
});
```

## describe / it / expect Patterns

```typescript
describe('UserService', () => {
  describe('getFullName', () => {
    it('returns first and last name', () => {
      expect(getFullName({ first: 'John', last: 'Doe' })).toBe('John Doe');
    });

    it('handles missing last name', () => {
      expect(getFullName({ first: 'John', last: '' })).toBe('John');
    });
  });

  describe('validateUser', () => {
    it.each([
      [null, false],
      [{}, false],
      [{ id: '1', name: 'John' }, true],
    ])('validates %s as %s', (input, expected) => {
      expect(validateUser(input)).toBe(expected);
    });
  });
});
```

## Test Doubles: Mock, Stub, Spy, Fake

| Type | Purpose | Example |
| :-- | :-- | :-- |
| **Mock** | Replace dependency, verify calls | `jest.fn()` |
| **Stub** | Replace with fixed return value | `mockResolvedValue(...)` |
| **Spy** | Wrap real function, track calls | `jest.spyOn(obj, 'method')` |
| **Fake** | Working but simplified implementation | In-memory DB |

### Mock

```typescript
const fetchUser = vi.fn(); // or jest.fn()

it('calls fetchUser with id', async () => {
  fetchUser.mockResolvedValue({ id: '1', name: 'John' });
  await loadUser('1');
  expect(fetchUser).toHaveBeenCalledWith('1');
});
```

### Stub

```typescript
vi.mock('./api', () => ({
  fetchUser: vi.fn().mockResolvedValue({ id: '1', name: 'John' }),
}));
```

### Spy

```typescript
it('logs on error', () => {
  const logSpy = vi.spyOn(console, 'error').mockImplementation(() => {});
  doSomethingThatLogsError();
  expect(logSpy).toHaveBeenCalledWith('Expected error message');
  logSpy.mockRestore();
});
```

### Fake

```typescript
const fakeStorage: Record<string, string> = {};
const fakeLocalStorage = {
  getItem: (key: string) => fakeStorage[key] ?? null,
  setItem: (key: string, value: string) => { fakeStorage[key] = value; },
};
```
