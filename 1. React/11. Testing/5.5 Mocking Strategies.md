# 5.5 Mocking Strategies

[[5. Testing.md]]

Effective mocking isolates the code under test from external dependencies. Choose the right strategy for each scenario.

---

## jest.mock / vi.mock

Replace an entire module with a mock:

```typescript
// api/users.ts
export async function fetchUser(id: string) {
  const res = await fetch(`/api/users/${id}`);
  return res.json();
}

// api/users.test.ts
import { vi } from 'vitest';
import { fetchUser } from './users';

vi.mock('./users', () => ({
  fetchUser: vi.fn(),
}));

// Re-import after mock to get mocked version
import { fetchUser } from './users';

test('uses mocked fetchUser', async () => {
  (fetchUser as ReturnType<typeof vi.fn>).mockResolvedValue({
    id: '1',
    name: 'John',
  });

  const result = await fetchUser('1');
  expect(result.name).toBe('John');
  expect(fetchUser).toHaveBeenCalledWith('1');
});
```

### Partial Mock (Keep Some Exports)

```typescript
vi.mock('./utils', async (importOriginal) => {
  const actual = await importOriginal<typeof import('./utils')>();
  return {
    ...actual,
    formatDate: vi.fn().mockReturnValue('2024-01-01'),
  };
});
```

## jest.spyOn / vi.spyOn

Wrap a real method to track calls; optionally replace implementation:

```typescript
test('logs error on fetch failure', async () => {
  const consoleSpy = vi.spyOn(console, 'error').mockImplementation(() => {});

  vi.spyOn(global, 'fetch').mockRejectedValue(new Error('Network error'));
  await loadData();

  expect(consoleSpy).toHaveBeenCalledWith('Failed to load:', expect.any(Error));
  consoleSpy.mockRestore();
});

test('spy without replacing', () => {
  const obj = { greet: (name: string) => `Hello ${name}` };
  const spy = vi.spyOn(obj, 'greet');

  obj.greet('World');
  expect(spy).toHaveBeenCalledWith('World');
  expect(spy).toHaveReturnedWith('Hello World');

  spy.mockRestore();
});
```

## Manual Mocks

Create `__mocks__` folder next to the module:

```
src/
  api/
    client.ts
    __mocks__/
      client.ts
```

```typescript
// src/api/__mocks__/client.ts
export const fetchUser = vi.fn();
export const fetchPosts = vi.fn();
```

```typescript
// In test
vi.mock('./api/client');
import { fetchUser } from './api/client';
```

## MSW for HTTP

Intercept requests at the network layer. Works with fetch, axios, or any HTTP client.

```typescript
// src/mocks/handlers.ts
import { http, HttpResponse } from 'msw';

export const handlers = [
  http.get('/api/users/:id', ({ params }) => {
    return HttpResponse.json({
      id: params.id,
      name: 'John Doe',
    });
  }),

  http.post('/api/login', async ({ request }) => {
    const { email, password } = await request.json() as Record<string, string>;
    if (password === 'wrong') {
      return HttpResponse.json({ error: 'Invalid' }, { status: 401 });
    }
    return HttpResponse.json({ token: 'abc', user: { email } });
  }),
];
```

```typescript
// Override in specific test
test('handles 404', async () => {
  server.use(
    http.get('/api/users/:id', () =>
      HttpResponse.json({ error: 'Not found' }, { status: 404 })
    )
  );

  render(<UserProfile userId="999" />);
  expect(await screen.findByText(/not found/i)).toBeInTheDocument();
});
```

## Mocking react-router / next/navigation

### React Router

```typescript
const mockNavigate = vi.fn();

vi.mock('react-router-dom', async (importOriginal) => {
  const actual = await importOriginal<typeof import('react-router-dom')>();
  return {
    ...actual,
    useNavigate: () => mockNavigate,
    useParams: () => ({ id: '123' }),
  };
});

test('navigates on success', async () => {
  render(<SaveButton />);
  await userEvent.click(screen.getByRole('button', { name: /save/i }));
  expect(mockNavigate).toHaveBeenCalledWith('/items/123');
});
```

### Next.js App Router

```typescript
vi.mock('next/navigation', () => ({
  useRouter: () => ({
    push: vi.fn(),
    replace: vi.fn(),
    refresh: vi.fn(),
    back: vi.fn(),
    forward: vi.fn(),
    prefetch: vi.fn(),
  }),
  usePathname: () => '/dashboard',
  useSearchParams: () => new URLSearchParams(),
}));
```

## Mocking Timers

```typescript
import { vi, beforeEach, afterEach } from 'vitest';

describe('with fake timers', () => {
  beforeEach(() => {
    vi.useFakeTimers();
  });

  afterEach(() => {
    vi.useRealTimers();
  });

  test('debounce delays execution', async () => {
    const fn = vi.fn();
    const debounced = debounce(fn, 100);

    debounced();
    expect(fn).not.toHaveBeenCalled();

    vi.advanceTimersByTime(100);
    expect(fn).toHaveBeenCalledTimes(1);
  });

  test('setTimeout', () => {
    const callback = vi.fn();
    setTimeout(callback, 5000);

    vi.advanceTimersByTime(5000);
    expect(callback).toHaveBeenCalled();
  });
});
```

## Mocking localStorage

```typescript
const localStorageMock = (() => {
  let store: Record<string, string> = {};
  return {
    getItem: (key: string) => store[key] ?? null,
    setItem: (key: string, value: string) => { store[key] = value; },
    removeItem: (key: string) => { delete store[key]; },
    clear: () => { store = {}; },
    get length() { return Object.keys(store).length; },
    key: (i: number) => Object.keys(store)[i] ?? null,
  };
})();

Object.defineProperty(global, 'localStorage', { value: localStorageMock });

beforeEach(() => {
  localStorage.clear();
});
```

## Test Data Builders / Factories

Create consistent, customizable test data:

```typescript
// test/builders/userBuilder.ts
export interface User {
  id: string;
  name: string;
  email: string;
  role: 'admin' | 'user';
}

export function createUser(overrides: Partial<User> = {}): User {
  return {
    id: '1',
    name: 'John Doe',
    email: 'john@example.com',
    role: 'user',
    ...overrides,
  };
}

export function createAdminUser(overrides: Partial<User> = {}): User {
  return createUser({ role: 'admin', ...overrides });
}
```

```typescript
// Usage
test('admin sees delete button', () => {
  const admin = createAdminUser({ name: 'Jane' });
  render(<UserMenu user={admin} />);
  expect(screen.getByRole('button', { name: /delete/i })).toBeInTheDocument();
});

test('regular user does not see delete', () => {
  const user = createUser();
  render(<UserMenu user={user} />);
  expect(screen.queryByRole('button', { name: /delete/i })).not.toBeInTheDocument();
});
```
