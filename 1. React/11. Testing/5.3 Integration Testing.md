# 5.3 Integration Testing

[[5. Testing.md]]

Integration tests verify multiple units working together: components with Context, React Router, and API calls. They catch bugs at component boundaries.

## Testing with Context Providers

```typescript
// contexts/AuthContext.tsx
const AuthContext = createContext<AuthContextValue | null>(null);

export function AuthProvider({ children }: { children: React.ReactNode }) {
  const [user, setUser] = useState<User | null>(null);
  const login = async (email: string, password: string) => { /* ... */ };
  const logout = () => setUser(null);
  return (
    <AuthContext.Provider value={{ user, login, logout }}>
      {children}
    </AuthContext.Provider>
  );
}
```

```typescript
// UserMenu.test.tsx
import { render, screen } from '@testing-library/react';
import userEvent from '@testing-library/user-event';
import { AuthContext } from './AuthContext';
import { UserMenu } from './UserMenu';

function renderWithAuth(ui: React.ReactElement, { user = null }: { user?: User | null } = {}) {
  const mockLogin = vi.fn();
  const mockLogout = vi.fn();
  const value = { user, login: mockLogin, logout: mockLogout };

  return {
    ...render(
      <AuthContext.Provider value={value}>
        {ui}
      </AuthContext.Provider>
    ),
    mockLogin,
    mockLogout,
  };
}

test('shows login when user is null', () => {
  renderWithAuth(<UserMenu />);
  expect(screen.getByText(/log in/i)).toBeInTheDocument();
});

test('shows user name and logout when authenticated', () => {
  renderWithAuth(<UserMenu />, { user: { id: '1', name: 'John' } });
  expect(screen.getByText(/john/i)).toBeInTheDocument();
  expect(screen.getByRole('button', { name: /logout/i })).toBeInTheDocument();
});

test('calls logout on button click', async () => {
  const { mockLogout } = renderWithAuth(<UserMenu />, {
    user: { id: '1', name: 'John' },
  });
  const user = userEvent.setup();
  await user.click(screen.getByRole('button', { name: /logout/i }));
  expect(mockLogout).toHaveBeenCalled();
});
```

## Custom Render Wrappers

Create a reusable `render` that includes all common providers:

```typescript
// test/utils.tsx
import { render, RenderOptions } from '@testing-library/react';
import { ReactElement } from 'react';
import { BrowserRouter } from 'react-router-dom';
import { AuthProvider } from '../contexts/AuthContext';
import { ThemeProvider } from '../contexts/ThemeContext';

interface AllProvidersProps {
  children: React.ReactNode;
  authUser?: User | null;
}

function AllProviders({ children, authUser = null }: AllProvidersProps) {
  return (
    <BrowserRouter>
      <ThemeProvider>
        <AuthProvider initialUser={authUser}>
          {children}
        </AuthProvider>
      </ThemeProvider>
    </BrowserRouter>
  );
}

function customRender(
  ui: ReactElement,
  options?: Omit<RenderOptions, 'wrapper'> & { authUser?: User | null }
) {
  const { authUser, ...renderOptions } = options ?? {};
  return render(ui, {
    wrapper: ({ children }) => (
      <AllProviders authUser={authUser}>{children}</AllProviders>
    ),
    ...renderOptions,
  });
}

export * from '@testing-library/react';
export { customRender as render };
```

```typescript
// In tests
import { render, screen } from '../test/utils';

test('protected route redirects when not logged in', () => {
  render(<App />);
  expect(screen.getByText(/log in/i)).toBeInTheDocument();
});

test('protected route shows content when logged in', () => {
  render(<App />, { authUser: { id: '1', name: 'John' } });
  expect(screen.getByText(/dashboard/i)).toBeInTheDocument();
});
```

## Testing with React Router

```typescript
import { render, screen } from '@testing-library/react';
import userEvent from '@testing-library/user-event';
import { MemoryRouter, Routes, Route } from 'react-router-dom';
import { App } from './App';

function renderWithRouter(
  ui: React.ReactElement,
  { initialEntries = ['/'] }: { initialEntries?: string[] } = {}
) {
  return render(
    <MemoryRouter initialEntries={initialEntries}>
      <Routes>
        <Route path="/*" element={ui} />
      </Routes>
    </MemoryRouter>
  );
}

test('navigates to profile', async () => {
  const user = userEvent.setup();
  renderWithRouter(<App />);

  await user.click(screen.getByRole('link', { name: /profile/i }));

  expect(screen.getByText(/profile page/i)).toBeInTheDocument();
});

test('shows 404 for unknown route', () => {
  renderWithRouter(<App />, { initialEntries: ['/unknown-route'] });
  expect(screen.getByText(/not found/i)).toBeInTheDocument();
});

test('protected route redirects to login', () => {
  renderWithRouter(<App />, { initialEntries: ['/dashboard'] });
  expect(screen.getByText(/log in/i)).toBeInTheDocument();
});
```

## Testing User Flows: Login

test('user can log in', async () => {
  const user = userEvent.setup();

});

## Testing CRUD Flows

```typescript
test('user can create and delete item', async () => {
  const user = userEvent.setup();

  vi.spyOn(global, 'fetch')
    .mockResolvedValueOnce({ ok: true, json: async () => ({ items: [] }) } as Response)
    .mockResolvedValueOnce({ ok: true, json: async () => ({ id: '1', title: 'New Item' }) } as Response)
    .mockResolvedValueOnce({ ok: true } as Response);

  render(<ItemList />);

  // Create
  await user.click(screen.getByRole('button', { name: /add item/i }));
  await user.type(screen.getByLabelText(/title/i), 'New Item');
  await user.click(screen.getByRole('button', { name: /save/i }));

  await waitFor(() => {
    expect(screen.getByText('New Item')).toBeInTheDocument();
  });

  // Delete
  await user.click(screen.getByRole('button', { name: /delete/i }));
  await waitFor(() => {
    expect(screen.queryByText('New Item')).not.toBeInTheDocument();
  });
});
```

## MSW (Mock Service Worker) for API Mocking

MSW intercepts network requests at the network level—no need to mock `fetch` or axios.

```bash
npm install -D msw
npx msw init public/ --save
```

```typescript
// src/mocks/handlers.ts
import { http, HttpResponse } from 'msw';

export const handlers = [
  http.get('/api/user/:id', ({ params }) => {
    return HttpResponse.json({
      id: params.id,
      name: 'John Doe',
      email: 'john@example.com',
    });
  }),

  http.post('/api/login', async ({ request }) => {
    const body = await request.json() as { email: string; password: string };
    if (body.password === 'wrong') {
      return HttpResponse.json({ error: 'Invalid credentials' }, { status: 401 });
    }
    return HttpResponse.json({ user: { id: '1', name: 'John' } });
  }),
];
```

```typescript
// src/mocks/server.ts
import { setupServer } from 'msw/node';
import { handlers } from './handlers';

export const server = setupServer(...handlers);
```

```typescript
// src/test/setup.ts
import '@testing-library/jest-dom';
import { server } from '../mocks/server';

beforeAll(() => server.listen({ onUnhandledRequest: 'error' }));
afterEach(() => server.resetHandlers());
afterAll(() => server.close());
```

```typescript
// UserProfile.test.tsx
test('loads user from API', async () => {
  render(<UserProfile userId="123" />);

  const name = await screen.findByText('John Doe');
  expect(name).toBeInTheDocument();
});

test('handles API error', async () => {
  server.use(
    http.get('/api/user/:id', () => HttpResponse.json({ error: 'Not found' }, { status: 404 }))
  );

  render(<UserProfile userId="999" />);

  expect(await screen.findByText(/error/i)).toBeInTheDocument();
});
```
