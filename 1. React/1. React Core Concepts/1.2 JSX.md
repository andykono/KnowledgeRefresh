# JSX

[[1. React Core Concepts.md]]

---

JSX is a syntax extension that looks like HTML but is JavaScript.

## JSX Basics

```typescript
// JSX
const element = <h1>Hello, World!</h1>;

// Compiles to:
// const element = React.createElement('h1', null, 'Hello, World!');
```

## Embedding Expressions

```typescript
const name = 'John';
const age = 30;

// Curly braces for JavaScript
const greeting = <h1>Hello, {name}! You are {age} years old.</h1>;

// Any expression works
<p>Sum: {2 + 3}</p>
<p>Condition: {age >= 18 ? 'Adult' : 'Minor'}</p>
<img src={imageUrl} alt="Description" />
```

## Attributes

```typescript
// HTML attributes become camelCase
<div className="container">
  <label htmlFor="name">Name:</label>
  <input id="name" type="text" />
  <button onClick={handleClick} disabled={isDisabled}>
    Click
  </button>
  <div style={{ color: 'blue', fontSize: '16px' }}>
    Inline styles
  </div>
</div>
```

## Children

```typescript
// String children
<h1>Hello</h1>

// Multiple children
<div>
  <h1>Title</h1>
  <p>Paragraph</p>
</div>

// Expression children
<div>
  {items.map(item => (
    <span key={item.id}>{item.name}</span>
  ))}
</div>

// Props.children
interface CardProps {
  children: React.ReactNode;
}

function Card({ children }: CardProps) {
  return <div className="card">{children}</div>;
}
```

## Conditionals

```typescript
// If statement (not in JSX)
function Greeting({ age }: { age: number }) {
  let message;
  if (age >= 18) {
    message = 'Adult';
  } else {
    message = 'Minor';
  }
  return <h1>{message}</h1>;
}

// Ternary (in JSX)
<h1>{age >= 18 ? 'Adult' : 'Minor'}</h1>

// Logical AND
<button>{isLoading && 'Loading...'}</button>

// Element variable
function Dashboard() {
  let content;
  if (isLoading) {
    content = <Spinner />;
  } else if (error) {
    content = <Error error={error} />;
  } else {
    content = <Data data={data} />;
  }
  return <div>{content}</div>;
}
```

## Lists and Keys

```typescript
// Always use unique key for lists
function UserList({ users }) {
  return (
    <ul>
      {users.map(user => (
        <li key={user.id}>  {/* Unique key! */}
          {user.name}
        </li>
      ))}
    </ul>
  );
}

// DO NOT use index as key (bad performance)
// ❌ BAD
{items.map((item, index) => <div key={index}>{item}</div>)}

// ✅ GOOD
{items.map(item => <div key={item.id}>{item.name}</div>)}
```

## Fragments

```typescript
// Return multiple elements without wrapper
function Component() {
  return (
    <>
      <h1>Title</h1>
      <p>Content</p>
    </>
  );
}

// Or use React.Fragment with key (rare)
{items.map(item => (
  <React.Fragment key={item.id}>
    <h1>{item.name}</h1>
    <p>{item.description}</p>
  </React.Fragment>
))}
```

## JSX is Not HTML

```typescript
// ❌ DON'T: HTML attribute names
<div class="container">  {/* Wrong */}
  <label for="name">  {/* Wrong */}
    <img src="" alt="" />
  </label>
</div>

// ✅ DO: camelCase attributes
<div className="container">  {/* Right */}
  <label htmlFor="name">  {/* Right */}
    <img src="" alt="" />
  </label>
</div>

// Boolean attributes
<input disabled={true} />   // Or just <input disabled />
<button hidden={false} />

// Styles are objects, not strings
<div style={{ color: 'red', fontSize: '16px' }} />

// JSX children must be expressions or text, not objects
// ❌ <div>{anObject}</div>  // Error
// ✅ <div>{JSON.stringify(anObject)}</div>
```

## JSX Best Practices

✅ **DO:**
- Use PascalCase for component names in JSX
- Close all tags properly
- Use unique keys in lists
- Extract complex logic outside JSX
- Use TypeScript for type safety

❌ **DON'T:**
- Don't use index as key in lists
- Don't put complex logic in JSX
- Don't use lowercase names for components
- Don't forget to close self-closing tags
- Don't mutate props or state

## JSX Rules

1. **Return a single root element**
   ```typescript
   // ❌ Error: multiple elements
   function Bad() {
     return (
       <h1>Title</h1>
       <p>Content</p>
     );
   }

   // ✅ Correct: wrapped in fragment or div
   function Good() {
     return (
       <>
         <h1>Title</h1>
         <p>Content</p>
       </>
     );
   }
   ```

2. **Close all tags**
   ```typescript
   // ❌ Error
   <img src="..." />  // Wait, this is actually OK

   // Actually all tags must be closed
   <div>Content</div>  // Correct
   <br />               // Correct (self-closing)
   <input />           // Correct (self-closing)
   ```

3. **Use camelCase for most attributes**
   ```typescript
   // Special cases:
   className  // Not "class"
   htmlFor    // Not "for"
   onClick    // Event handlers in camelCase
   onChange   // Event handlers in camelCase
   data-*     // Keep as-is
   aria-*     // Keep as-is
   ```

## Complex Example

```typescript
interface Product {
  id: number;
  name: string;
  price: number;
  inStock: boolean;
}

interface ProductListProps {
  products: Product[];
  onPurchase: (id: number) => void;
}

function ProductList({ products, onPurchase }: ProductListProps) {
  return (
    <div className="product-list">
      {products.length === 0 ? (
        <p>No products available</p>
      ) : (
        <ul>
          {products.map(product => (
            <li key={product.id} className={!product.inStock ? 'out-of-stock' : ''}>
              <h3>{product.name}</h3>
              <p>${product.price.toFixed(2)}</p>
              <button
                onClick={() => onPurchase(product.id)}
                disabled={!product.inStock}
              >
                {product.inStock ? 'Buy' : 'Out of Stock'}
              </button>
            </li>
          ))}
        </ul>
      )}
    </div>
  );
}
```
