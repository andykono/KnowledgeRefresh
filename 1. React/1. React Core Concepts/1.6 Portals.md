# Portals

[[1. React Core Concepts.md]]

---

Portals render components outside the normal DOM hierarchy.

## Basic Usage

```typescript
import { createPortal } from 'react-dom';

function Modal() {
  return createPortal(
    <div className="modal">
      <p>I'm rendered outside the parent!</p>
    </div>,
    document.body  // Render into body instead of parent
  );
}
```

## Common Use Cases

```typescript
// Modal dialogs
createPortal(<Modal />, document.getElementById('modal-root'));

// Tooltips
createPortal(<Tooltip />, document.body);

// Dropdowns
createPortal(<Dropdown />, document.body);
```

---

## Event Bubbling Through Portals

**Critical:** Events bubble through the **React component tree**, not the DOM tree. A portal's parent in the React tree receives events even though the portal renders elsewhere in the DOM.

```typescript
function Parent() {
  const handleClick = () => console.log('Parent clicked!');

  return (
    <div onClick={handleClick} style={{ padding: 50 }}>
      <p>Parent div</p>
      {createPortal(
        <button>Portal button (in body)</button>,
        document.body
      )}
    </div>
  );
}
// Clicking the portal button logs "Parent clicked!" — event bubbles to React parent
```

This means you can handle portal events in the parent component. To stop propagation to the React parent, call `event.stopPropagation()` in the portal's handler.

## Modal Component with Portal Pattern

```typescript
import { createPortal } from 'react-dom';

interface ModalProps {
  isOpen: boolean;
  onClose: () => void;
  children: React.ReactNode;
}

function Modal({ isOpen, onClose, children }: ModalProps) {
  if (!isOpen) return null;

  return createPortal(
    <div
      className="modal-overlay"
      onClick={onClose}
      role="dialog"
      aria-modal="true"
    >
      <div
        className="modal-content"
        onClick={(e) => e.stopPropagation()}
      >
        <button onClick={onClose} aria-label="Close">×</button>
        {children}
      </div>
    </div>,
    document.body
  );
}
```

The overlay closes on outside click; the content div stops propagation so clicking inside doesn't close it.

## Tooltip with Portal and Positioning

```typescript
function Tooltip({ children, text }: { children: React.ReactNode; text: string }) {
  const [visible, setVisible] = useState(false);
  const [coords, setCoords] = useState({ x: 0, y: 0 });
  const triggerRef = useRef<HTMLSpanElement>(null);

  const updatePosition = useCallback(() => {
    const rect = triggerRef.current?.getBoundingClientRect();
    if (rect) {
      setCoords({
        x: rect.left + rect.width / 2,
        y: rect.top - 8,
      });
    }
  }, []);

  return (
    <>
      <span
        ref={triggerRef}
        onMouseEnter={() => { updatePosition(); setVisible(true); }}
        onMouseLeave={() => setVisible(false)}
      >
        {children}
      </span>
      {visible && createPortal(
        <div
          className="tooltip"
          style={{
            position: 'fixed',
            left: coords.x,
            top: coords.y,
            transform: 'translate(-50%, -100%)',
          }}
        >
          {text}
        </div>,
        document.body
      )}
    </>
  );
}
```

Portals avoid z-index and overflow issues; positioning with `getBoundingClientRect()` keeps the tooltip aligned with the trigger.

## Accessibility Considerations

### Focus Trap

Modals should trap focus so users can't tab outside:

```typescript
function useFocusTrap(isActive: boolean) {
  const ref = useRef<HTMLDivElement>(null);

  useEffect(() => {
    if (!isActive || !ref.current) return;
    const focusable = ref.current.querySelectorAll(
      'button, [href], input, select, textarea, [tabindex]:not([tabindex="-1"])'
    );
    const first = focusable[0] as HTMLElement;
    const last = focusable[focusable.length - 1] as HTMLElement;

    const handleKeyDown = (e: KeyboardEvent) => {
      if (e.key !== 'Tab') return;
      if (e.shiftKey) {
        if (document.activeElement === first) {
          e.preventDefault();
          last?.focus();
        }
      } else {
        if (document.activeElement === last) {
          e.preventDefault();
          first?.focus();
        }
      }
    };

    first?.focus();
    ref.current.addEventListener('keydown', handleKeyDown);
    return () => ref.current?.removeEventListener('keydown', handleKeyDown);
  }, [isActive]);

  return ref;
}
```

### ARIA Attributes

```typescript
<div
  role="dialog"
  aria-modal="true"
  aria-labelledby="modal-title"
  aria-describedby="modal-desc"
>
  <h2 id="modal-title">Confirm Action</h2>
  <p id="modal-desc">Are you sure you want to proceed?</p>
</div>
```

Also restore focus to the trigger element when the modal closes.

## Portal with Server-Side Rendering

Portals require a DOM node. On the server, that node doesn't exist. Handle SSR by rendering the portal content in place during SSR and only using the portal on the client:

```typescript
function ClientPortal({ children }: { children: React.ReactNode }) {
  const [mounted, setMounted] = useState(false);

  useEffect(() => {
    setMounted(true);
  }, []);

  if (!mounted) return null;
  return createPortal(children, document.body);
}

// Or render in-place during SSR (may cause layout shift)
function SafePortal({ children }: { children: React.ReactNode }) {
  const [mounted, setMounted] = useState(false);

  useEffect(() => {
    setMounted(true);
  }, []);

  if (!mounted) return <>{children}</>;  // SSR: render in tree
  return createPortal(children, document.body);
}
```

## When to Use vs When Not to Use Portals

### Use Portals For

- **Modals and dialogs** — avoid z-index and overflow clipping
- **Tooltips and popovers** — escape overflow:hidden ancestors
- **Dropdowns** — render above other content
- **Full-screen overlays** — e.g., loading screens
- **Third-party widget containers** — inject into a specific DOM node

### Avoid Portals When

- **Normal layout flow works** — no need for portals if there are no stacking/overflow issues
- **SEO matters** — content in portals is still in the React tree, but consider semantic structure
- **You need parent CSS** — portaled content is outside the parent in the DOM, so parent styles (e.g., `color: inherit`) don't apply
- **Event delegation from DOM parent** — DOM ancestors won't receive events from portaled content; React tree ancestors will
