# Error Boundaries

[[1. React Core Concepts.md]]

---

Error Boundaries catch errors in child components.

## Basic Error Boundary

```typescript
class ErrorBoundary extends React.Component<
  { children: React.ReactNode },
  { hasError: boolean }
> {
  constructor(props) {
    super(props);
    this.state = { hasError: false };
  }

  static getDerivedStateFromError(error) {
    return { hasError: true };
  }

  componentDidCatch(error, errorInfo) {
    console.error('Error caught:', error, errorInfo);
  }

  render() {
    if (this.state.hasError) {
      return <h1>Something went wrong</h1>;
    }
    return this.props.children;
  }
}

// Usage
<ErrorBoundary>
  <ProblematicComponent />
</ErrorBoundary>
```

## react-error-boundary Library

The `react-error-boundary` package provides a ready-made ErrorBoundary and hooks.

### ErrorBoundary Component

```typescript
import { ErrorBoundary } from 'react-error-boundary';

function ErrorFallback({ error, resetErrorBoundary }: {
  error: Error;
  resetErrorBoundary: () => void;
}) {
  return (
    <div>
      <h2>Something went wrong</h2>
      <pre>{error.message}</pre>
      <button onClick={resetErrorBoundary}>Try again</button>
    </div>
  );
}

<ErrorBoundary FallbackComponent={ErrorFallback}>
  <ProblematicComponent />
</ErrorBoundary>
```

### fallbackRender

```typescript
<ErrorBoundary
  fallbackRender={({ error, resetErrorBoundary }) => (
    <div>
      <p>Error: {error.message}</p>
      <button onClick={resetErrorBoundary}>Retry</button>
    </div>
  )}
>
  <App />
</ErrorBoundary>
```

### onReset and resetKeys

```typescript
<ErrorBoundary
  FallbackComponent={ErrorFallback}
  onReset={() => {
    // Cleanup or analytics
    logErrorRecovery();
  }}
  resetKeys={[userId, pageId]}  // Reset when these change
>
  <UserDashboard userId={userId} />
</ErrorBoundary>
```

When any value in `resetKeys` changes, the boundary resets and tries to render children again.

### useErrorBoundary Hook

```typescript
import { useErrorBoundary } from 'react-error-boundary';

function MyComponent() {
  const { showBoundary } = useErrorBoundary();

  const handleClick = async () => {
    try {
      await riskyOperation();
    } catch (error) {
      showBoundary(error);  // Trigger nearest ErrorBoundary
    }
  };

  return <button onClick={handleClick}>Do something</button>;
}
```

Use this to manually trigger the nearest error boundary from async code or event handlers.

## Recovery Patterns

### Retry

```typescript
function ErrorFallback({ error, resetErrorBoundary }: FallbackProps) {
  return (
    <div>
      <p>{error.message}</p>
      <button onClick={resetErrorBoundary}>Retry</button>
    </div>
  );
}
```

### Reset with Key

```typescript
const [retryCount, setRetryCount] = useState(0);

<ErrorBoundary
  key={retryCount}
  FallbackComponent={({ resetErrorBoundary }) => (
    <button onClick={() => {
      setRetryCount((c) => c + 1);
      resetErrorBoundary();
    }}>
      Try again
    </button>
  )}
>
  <UnstableComponent />
</ErrorBoundary>
```

### Navigate Away

```typescript
function ErrorFallback({ resetErrorBoundary }: FallbackProps) {
  const navigate = useNavigate();
  return (
    <div>
      <p>Something went wrong</p>
      <button onClick={() => navigate('/')}>Go home</button>
      <button onClick={resetErrorBoundary}>Retry</button>
    </div>
  );
}
```

## Nested Error Boundaries Strategy

Place boundaries at different levels so one failure doesn't take down the whole app:

```typescript
<ErrorBoundary fallback={<AppCrashFallback />}>
  <Header />
  <ErrorBoundary fallback={<SidebarFallback />}>
    <Sidebar />
  </ErrorBoundary>
  <ErrorBoundary fallback={<ContentFallback />}>
    <MainContent />
  </ErrorBoundary>
  <ErrorBoundary fallback={<FooterFallback />}>
    <Footer />
  </ErrorBoundary>
</ErrorBoundary>
```

If `MainContent` crashes, only that section shows the fallback; header, sidebar, and footer stay visible.

## Error Boundary Placement Patterns

- **Route level:** One boundary per route so a broken page doesn't crash the whole app
- **Feature level:** Wrap major features (e.g., checkout, dashboard)
- **Widget level:** Wrap isolated widgets (e.g., charts, third-party embeds)
- **App root:** Catch unhandled errors and show a global fallback

```typescript
// Route-level
<Routes>
  <Route path="/" element={
    <ErrorBoundary fallback={<HomeError />}>
      <HomePage />
    </ErrorBoundary>
  } />
  <Route path="/dashboard" element={
    <ErrorBoundary fallback={<DashboardError />}>
      <Dashboard />
    </ErrorBoundary>
  } />
</Routes>
```

## Logging Errors to Services (Sentry etc.)

```typescript
import * as Sentry from '@sentry/react';

class ErrorBoundary extends React.Component<Props, State> {
  componentDidCatch(error: Error, errorInfo: React.ErrorInfo) {
    Sentry.captureException(error, {
      extra: {
        componentStack: errorInfo.componentStack,
      },
    });
    this.setState({ hasError: true });
  }
  // ...
}
```

Or with `react-error-boundary`:

```typescript
<ErrorBoundary
  FallbackComponent={ErrorFallback}
  onError={(error, info) => {
    Sentry.captureException(error, { extra: info });
  }}
>
  <App />
</ErrorBoundary>
```

## What They Catch

✅ **Caught by Error Boundaries:**
- Errors during render
- Errors in lifecycle methods (e.g., `componentDidMount`)

❌ **NOT caught by Error Boundaries:**
- **Event handlers** — use try/catch inside the handler
- **Async code** (setTimeout, Promises, async/await) — use `.catch()` or try/catch
- **Server-side rendering** — errors in SSR are not caught by client boundaries
- **Errors in the boundary itself** — propagate to the nearest parent boundary

```typescript
// Event handler: use try/catch
const handleClick = () => {
  try {
    throw new Error('Click error');
  } catch (e) {
    console.error(e);
  }
};

// Async: use .catch or try/catch
useEffect(() => {
  fetchData().catch((err) => {
    logError(err);
    setError(err);
  });
}, []);
```
