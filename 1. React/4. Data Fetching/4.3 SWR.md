# SWR

[[4. Data Fetching.md]]

SWR (stale-while-revalidate) is a React Hooks library for data fetching by Vercel. The name comes from the HTTP cache invalidation strategy: show cached (stale) data immediately while revalidating in the background.

## Installation

```bash
npm install swr
```

## Basic Usage with useSWR

The `useSWR` hook accepts a key (usually the URL) and a fetcher function:

```typescript
import useSWR from 'swr';

const fetcher = (url: string) => fetch(url).then((res) => res.json());

function UserProfile({ userId }: { userId: string }) {
  const { data, error, isLoading, mutate } = useSWR(
    `/api/users/${userId}`,
    fetcher
  );

  if (error) return <div>Failed to load</div>;
  if (isLoading) return <div>Loading...</div>;
  return <div>Hello, {data.name}!</div>;
}
```

The hook returns:
- `data`: The fetched data (or undefined while loading)
- `error`: Error object if the request failed
- `isLoading`: True during initial load (no cached data)
- `isValidating`: True whenever a revalidation is in progress
- `mutate`: Function to manually revalidate or update the cache

## Global Configuration with SWRConfig

Wrap your app with `SWRConfig` to set defaults for all SWR hooks:

```typescript
import { SWRConfig } from 'swr';

const fetcher = (url: string) => fetch(url).then((res) => res.json());

function App() {
  return (
    <SWRConfig
      value={{
        fetcher,
        revalidateOnFocus: true,
        revalidateOnReconnect: true,
        dedupingInterval: 2000,
        errorRetryCount: 3,
      }}
    >
      <UserProfile userId="123" />
    </SWRConfig>
  );
}
```

## Cache Invalidation with mutate()

Use `mutate()` to invalidate and refetch data:

```typescript
const { data, mutate } = useSWR('/api/user', fetcher);

async function handleLogout() {
  await mutate(); // Revalidate
  // Or: mutate(undefined, { revalidate: true });
}

// Optimistic update
async function updateName(newName: string) {
  await mutate(
    { ...data, name: newName },
    { revalidate: false }
  );
  await fetch('/api/user', {
    method: 'PATCH',
    body: JSON.stringify({ name: newName }),
  });
  mutate(); // Refetch to sync with server
}
```

## useSWRMutation for Mutations

For POST, PUT, DELETE, use `useSWRMutation`:

```typescript
import useSWRMutation from 'swr/mutation';

async function createUser(url: string, { arg }: { arg: { name: string; email: string } }) {
  const res = await fetch(url, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(arg),
  });
  if (!res.ok) throw new Error('Failed to create user');
  return res.json();
}

function CreateUserForm() {
  const { trigger, isMutating, error } = useSWRMutation(
    '/api/users',
    createUser
  );

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    try {
      await trigger({ name: 'John', email: 'john@example.com' });
    } catch (err) {
      console.error(err);
    }
  };

  return (
    <form onSubmit={handleSubmit}>
      <button type="submit" disabled={isMutating}>
        {isMutating ? 'Creating...' : 'Create User'}
      </button>
      {error && <p>{error.message}</p>}
    </form>
  );
}
```

## Revalidation Strategies

| Strategy | Option | Description |
|----------|--------|-------------|
| On focus | `revalidateOnFocus: true` | Refetch when window regains focus |
| On interval | `refreshInterval: 3000` | Refetch every N milliseconds |
| On reconnect | `revalidateOnReconnect: true` | Refetch when network reconnects |
| On mount | `revalidateIfStale: true` | Refetch on mount if data is stale |

```typescript
useSWR('/api/data', fetcher, {
  revalidateOnFocus: true,
  refreshInterval: 5000,
  revalidateOnReconnect: true,
});
```

## useSWRInfinite for Pagination

Handle infinite lists and pagination:

```typescript
import useSWRInfinite from 'swr/infinite';

const getKey = (pageIndex: number, previousPageData: User[] | null) => {
  if (previousPageData && !previousPageData.length) return null;
  return `/api/users?page=${pageIndex}&limit=10`;
};

function UserList() {
  const { data, size, setSize, isLoading } = useSWRInfinite(
    getKey,
    fetcher
  );

  const users = data ? data.flat() : [];
  const isLoadingMore = isLoading || (size > 0 && data && typeof data[size - 1] === 'undefined');

  return (
    <div>
      {users.map((user) => (
        <div key={user.id}>{user.name}</div>
      ))}
      <button
        onClick={() => setSize(size + 1)}
        disabled={isLoadingMore}
      >
        Load More
      </button>
    </div>
  );
}
```

## Error Handling and Retries

```typescript
const { data, error, isLoading, mutate } = useSWR('/api/data', fetcher, {
  onErrorRetry: (error, key, config, revalidate, { retryCount }) => {
    if (error.status === 404) return;
    if (retryCount >= 10) return;
    setTimeout(() => revalidate({ retryCount }), 5000);
  },
  errorRetryCount: 3,
});

if (error) {
  return (
    <div>
      <p>Error: {error.message}</p>
      <button onClick={() => mutate()}>Retry</button>
    </div>
  );
}
```

## TypeScript Integration

```typescript
interface User {
  id: string;
  name: string;
  email: string;
}

const { data, error } = useSWR<User>('/api/user', fetcher);
// data is typed as User | undefined

// With useSWRMutation
const { trigger } = useSWRMutation<User, Error, string, { name: string }>(
  '/api/user',
  (url, { arg }) => fetch(url, {
    method: 'POST',
    body: JSON.stringify(arg),
  }).then((r) => r.json())
);
```

## SWR vs TanStack Query

| Feature | SWR | TanStack Query |
|---------|-----|----------------|
| Bundle size | ~4KB | ~13KB |
| Learning curve | Minimal | Moderate |
| DevTools | Basic | Full-featured |
| Mutations | useSWRMutation | useMutation |
| Infinite queries | useSWRInfinite | useInfiniteQuery |
| Cache persistence | Manual | Built-in |
| Optimistic updates | Manual | Built-in |
| Stale time control | dedupingInterval | staleTime |
| Best for | Simple apps, Vercel stack | Complex apps, enterprise |
