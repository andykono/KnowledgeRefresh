# Suspense for Data

[[4. Data Fetching.md]]

---

React Suspense allows you to declaratively handle loading states. Combined with the `use()` hook (React 19) and data fetching libraries, you can build streaming UIs without manual loading flags.

## Suspense Basics

```typescript
import { Suspense } from 'react';

function App() {
  return (
    <Suspense fallback={<UserSkeleton />}>
      <UserProfile userId="123" />
    </Suspense>
  );
}

// UserProfile suspends while loading - no isLoading checks needed
function UserProfile({ userId }: { userId: string }) {
  const user = use(fetchUser(userId)); // use() unwraps the promise
  return <div>{user.name}</div>;
}
```

## use() Hook (React 19)

The `use()` hook unwraps promises and context. Unlike `useEffect`, it can be called conditionally in some cases (e.g., inside try/catch for error boundaries).

```typescript
import { use } from 'react';

// use() with a promise - suspends until resolved
function UserName({ userId }: { userId: string }) {
  const user = use(fetchUser(userId));
  return <span>{user.name}</span>;
}

// use() with context
function ThemeToggle() {
  const theme = use(ThemeContext);
  return <button>{theme}</button>;
}
```

## Creating a Suspense-Compatible Fetcher

```typescript
// Cache to avoid duplicate requests
const userCache = new Map<string, Promise<User>>();

function fetchUser(userId: string): Promise<User> {
  if (!userCache.has(userId)) {
    userCache.set(
      userId,
      fetch(`/api/users/${userId}`)
        .then((r) => r.json())
        .then((data) => {
          if (!data) throw new Error('User not found');
          return data;
        })
    );
  }
  return userCache.get(userId)!;
}

// Usage - component suspends until data is ready
function UserProfile({ userId }: { userId: string }) {
  const user = use(fetchUser(userId));
  return (
    <div>
      <h1>{user.name}</h1>
      <p>{user.email}</p>
    </div>
  );
}
```

## Error Boundaries with Data

Wrap Suspense boundaries with error boundaries to catch fetch failures:

```typescript
import { Component, type ReactNode } from 'react';

class ErrorBoundary extends Component<
  { children: ReactNode; fallback: ReactNode },
  { hasError: boolean; error?: Error }
> {
  state = { hasError: false, error: undefined };

  static getDerivedStateFromError(error: Error) {
    return { hasError: true, error };
  }

  render() {
    if (this.state.hasError) {
      return this.props.fallback;
    }
    return this.props.children;
  }
}

// Usage
<ErrorBoundary fallback={<ErrorMessage />}>
  <Suspense fallback={<UserSkeleton />}>
    <UserProfile userId="123" />
  </Suspense>
</ErrorBoundary>
```

## Nested Suspense and Streaming

Multiple Suspense boundaries enable streaming: each boundary can resolve independently.

```typescript
function Dashboard() {
  return (
    <div>
      <Suspense fallback={<HeaderSkeleton />}>
        <Header />
      </Suspense>
      <div className="grid">
        <Suspense fallback={<CardSkeleton />}>
          <StatsCards />
        </Suspense>
        <Suspense fallback={<TableSkeleton />}>
          <RecentActivity />
        </Suspense>
      </div>
    </div>
  );
}

// Each child can fetch its own data - UI streams in as each resolves
function StatsCards() {
  const stats = use(fetchStats());
  return (
    <div className="cards">
      {stats.map((s) => (
        <Card key={s.id} {...s} />
      ))}
    </div>
  );
}
```

## use() in try/catch for Error Boundaries

In React 19, `use()` can be used inside try/catch. When the promise rejects, the error propagates to the nearest error boundary:

```typescript
function UserProfile({ userId }: { userId: string }) {
  try {
    const user = use(fetchUser(userId));
    return <div>{user.name}</div>;
  } catch (error) {
    // In React 19, this catches promise rejections from use()
    throw error; // Re-throw to let ErrorBoundary handle it
  }
}
```

## TanStack Query with Suspense

```typescript
import { useSuspenseQuery } from '@tanstack/react-query';

function UserProfile({ userId }: { userId: string }) {
  const { data } = useSuspenseQuery({
    queryKey: ['user', userId],
    queryFn: () => fetchUser(userId),
  });
  return <div>{data.name}</div>;
}

// No loading state - Suspense handles it
<Suspense fallback={<Skeleton />}>
  <UserProfile userId="123" />
</Suspense>
```

## Streaming SSR (React 18+)

With React Server Components and streaming, Suspense boundaries flush HTML as data resolves:

```tsx
// Server Component (Next.js App Router example)
async function UserPage({ params }: { params: { id: string } }) {
  return (
    <div>
      <Suspense fallback={<UserSkeleton />}>
        <UserProfile userId={params.id} />
      </Suspense>
    </div>
  );
}

// UserProfile fetches on server, streams when ready
async function UserProfile({ userId }: { userId: string }) {
  const user = await fetchUser(userId);
  return <div>{user.name}</div>;
}
```

## Custom use() Wrapper for Caching

```typescript
const cache = new Map<string, { data: unknown; promise: Promise<unknown> }>();

function useFetch<T>(key: string, fetcher: () => Promise<T>): T {
  if (!cache.has(key)) {
    const promise = fetcher().then((data) => {
      cache.set(key, { data, promise });
      return data;
    });
    cache.set(key, { data: null, promise });
  }
  const entry = cache.get(key)!;
  if (entry.data !== null) return entry.data as T;
  return use(entry.promise) as T;
}
```

## Best Practices

1. **Place fallbacks close to suspending content** - Granular boundaries improve perceived performance
2. **Use error boundaries** - Catch failed fetches and show user-friendly messages
3. **Cache fetchers** - Prevent duplicate in-flight requests with a cache
4. **Prefer libraries** - TanStack Query and SWR have built-in Suspense support
