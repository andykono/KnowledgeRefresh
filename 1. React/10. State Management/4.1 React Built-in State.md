# React Built-in State

[[4. State Management.md]]

---

React provides `useState`, `useReducer`, and `useContext` for state management. For many applications, these built-in tools are sufficient.

## useState

The primary hook for local component state.

```typescript
import { useState } from 'react';

function Counter() {
  const [count, setCount] = useState(0);

  return (
    <div>
      <p>Count: {count}</p>
      <button onClick={() => setCount((c) => c + 1)}>Increment</button>
    </div>
  );
}
```

### Functional Updates

Use when the new state depends on the previous state:

```typescript
// ✅ Correct — avoids stale closure
setCount((prev) => prev + 1);

// ❌ Risky — may use stale value in async/event handlers
setCount(count + 1);
```

### Lazy Initialization

Expensive initial state should use a function:

```typescript
const [state, setState] = useState(() => {
  return expensiveComputation();
});
```

## useReducer

For complex state logic with multiple sub-values or when next state depends on previous.

```typescript
interface TodoState {
  items: string[];
  filter: 'all' | 'active' | 'completed';
}

type TodoAction =
  | { type: 'ADD'; payload: string }
  | { type: 'REMOVE'; payload: number }
  | { type: 'SET_FILTER'; payload: TodoState['filter'] };

function todoReducer(state: TodoState, action: TodoAction): TodoState {
  switch (action.type) {
    case 'ADD':
      return { ...state, items: [...state.items, action.payload] };
    case 'REMOVE':
      return {
        ...state,
        items: state.items.filter((_, i) => i !== action.payload),
      };
    case 'SET_FILTER':
      return { ...state, filter: action.payload };
    default:
      return state;
  }
}

function TodoList() {
  const [state, dispatch] = useReducer(todoReducer, {
    items: [],
    filter: 'all',
  });

  return (
    <>
      <button onClick={() => dispatch({ type: 'ADD', payload: 'New item' })}>
        Add
      </button>
      <select
        value={state.filter}
        onChange={(e) =>
          dispatch({
            type: 'SET_FILTER',
            payload: e.target.value as TodoState['filter'],
          })
        }
      >
        <option value="all">All</option>
        <option value="active">Active</option>
        <option value="completed">Completed</option>
      </select>
    </>
  );
}
```

## useContext as State Management

Context can share state without prop drilling. Combine with `useState` or `useReducer` for a simple global state solution.

```typescript
import { createContext, useContext, useState, useCallback, ReactNode } from 'react';

interface ThemeContextValue {
  theme: 'light' | 'dark';
  toggleTheme: () => void;
}

const ThemeContext = createContext<ThemeContextValue | null>(null);

export function ThemeProvider({ children }: { children: ReactNode }) {
  const [theme, setTheme] = useState<'light' | 'dark'>('light');

  const toggleTheme = useCallback(() => {
    setTheme((t) => (t === 'light' ? 'dark' : 'light'));
  }, []);

  return (
    <ThemeContext.Provider value={{ theme, toggleTheme }}>
      {children}
    </ThemeContext.Provider>
  );
}

export function useTheme() {
  const ctx = useContext(ThemeContext);
  if (!ctx) throw new Error('useTheme must be used within ThemeProvider');
  return ctx;
}
```

## Lifting State Up

When sibling components need to share state, lift it to their common parent.

```typescript
function Parent() {
  const [selectedId, setSelectedId] = useState<string | null>(null);

  return (
    <div>
      <List onSelect={setSelectedId} selectedId={selectedId} />
      <Detail id={selectedId} />
    </div>
  );
}
```

## Derived State

Compute values from state instead of storing them. Avoids sync bugs.

```typescript
function TodoList() {
  const [todos, setTodos] = useState<Todo[]>([]);
  const [filter, setFilter] = useState<'all' | 'active' | 'done'>('all');

  // ✅ Derived — always in sync
  const filteredTodos = useMemo(() => {
    switch (filter) {
      case 'active':
        return todos.filter((t) => !t.done);
      case 'done':
        return todos.filter((t) => t.done);
      default:
        return todos;
    }
  }, [todos, filter]);

  const completedCount = todos.filter((t) => t.done).length;
}
```

## When Built-in Is Enough

Use React built-in state when:

- **Local UI state:** Modals, forms, toggles
- **Shared across a few components:** Lift state or use Context
- **Simple global values:** Theme, locale, auth user via Context
- **No heavy debugging needs:** React DevTools is sufficient

Consider external libraries when:

- **Prop drilling** becomes painful (3+ levels)
- **Context causes excessive re-renders** (frequently changing values)
- **Need time-travel debugging** or middleware
- **Server state** (use TanStack Query instead)

## Combining useState and useReducer

Use `useReducer` for complex state, `useState` for simple local UI:

```typescript
function CheckoutWizard() {
  const [step, setStep] = useState(0);
  const [formData, dispatch] = useReducer(checkoutReducer, initialFormData);

  return (
    <>
      <Stepper current={step} onStepChange={setStep} />
      {step === 0 && <ShippingForm data={formData} dispatch={dispatch} />}
      {step === 1 && <PaymentForm data={formData} dispatch={dispatch} />}
    </>
  );
}
```
