# Zustand

[[4. State Management.md]]

---

Zustand is a minimal state management library. No providers, minimal boilerplate, built-in support for persist, immer, and devtools.

## Installation

```bash
npm install zustand
```

## Store Creation

```typescript
import { create } from 'zustand';

interface CounterState {
  count: number;
  increment: () => void;
  decrement: () => void;
  reset: () => void;
}

export const useCounterStore = create<CounterState>((set) => ({
  count: 0,
  increment: () => set((state) => ({ count: state.count + 1 })),
  decrement: () => set((state) => ({ count: state.count - 1 })),
  reset: () => set({ count: 0 }),
}));

// Usage
function Counter() {
  const { count, increment, decrement } = useCounterStore();

  return (
    <div>
      <span>{count}</span>
      <button onClick={increment}>+</button>
      <button onClick={decrement}>-</button>
    </div>
  );
}
```

## Selectors

Select only what you need to avoid unnecessary re-renders.

```typescript
// ❌ Re-renders on any store change
const { count } = useCounterStore();

// ✅ Re-renders only when count changes
const count = useCounterStore((state) => state.count);

// ✅ Select multiple values (shallow compare)
const { count, increment } = useCounterStore(
  (state) => ({ count: state.count, increment: state.increment }),
  (a, b) => a.count === b.count && a.increment === b.increment
);
```

## Middleware: Persist

Persist state to localStorage or sessionStorage.

```typescript
import { create } from 'zustand';
import { persist } from 'zustand/middleware';

interface AuthState {
  token: string | null;
  setToken: (token: string | null) => void;
}

export const useAuthStore = create<AuthState>()(
  persist(
    (set) => ({
      token: null,
      setToken: (token) => set({ token }),
    }),
    { name: 'auth-storage' }
  )
);
```

## Middleware: Immer

Use mutable-style updates for nested state.

```typescript
import { create } from 'zustand';
import { immer } from 'zustand/middleware/immer';

interface TodoState {
  todos: { id: string; text: string; done: boolean }[];
  addTodo: (text: string) => void;
  toggleTodo: (id: string) => void;
}

export const useTodoStore = create<TodoState>()(
  immer((set) => ({
    todos: [],
    addTodo: (text) =>
      set((state) => {
        state.todos.push({
          id: crypto.randomUUID(),
          text,
          done: false,
        });
      }),
    toggleTodo: (id) =>
      set((state) => {
        const todo = state.todos.find((t) => t.id === id);
        if (todo) todo.done = !todo.done;
      }),
  }))
);
```

## Middleware: DevTools

Connect to Redux DevTools for debugging.

```typescript
import { devtools } from 'zustand/middleware';

export const useStore = create<State>()(
  devtools(
    (set) => ({
      /* ... */
    }),
    { name: 'MyStore' }
  )
);
```

## Slices Pattern

Split a large store into slices for maintainability.

```typescript
import { create } from 'zustand';

interface UserSlice {
  user: { name: string } | null;
  setUser: (user: { name: string } | null) => void;
}

interface UISlice {
  sidebarOpen: boolean;
  toggleSidebar: () => void;
}

type Store = UserSlice & UISlice;

const createUserSlice = (set: any): UserSlice => ({
  user: null,
  setUser: (user) => set({ user }),
});

const createUISlice = (set: any): UISlice => ({
  sidebarOpen: true,
  toggleSidebar: () => set((s: Store) => ({ sidebarOpen: !s.sidebarOpen })),
});

export const useStore = create<Store>()((...a) => ({
  ...createUserSlice(...a),
  ...createUISlice(...a),
}));
```

## TypeScript Integration

```typescript
interface BearState {
  bears: number;
  addBear: () => void;
}

const useBearStore = create<BearState>()((set) => ({
  bears: 0,
  addBear: () => set((state) => ({ bears: state.bears + 1 })),
}));

// Infer types from store
type BearStore = typeof useBearStore.getState;
```

## Comparison with Redux

| Aspect | Zustand | Redux Toolkit |
| :-- | :-- | :-- |
| Boilerplate | Minimal | Medium |
| Provider | None | Required |
| DevTools | Via middleware | Built-in |
| Async | Manual (or use TanStack Query) | createAsyncThunk / RTK Query |
| Learning curve | Low | Medium |

## When to Use Zustand

**Use when:**
- Global state without Redux complexity
- Hooks-first, no providers
- Fast prototyping
- Small to medium apps

**Avoid when:**
- Need Redux middleware ecosystem
- Team standardizes on Redux
- Very complex reducer logic with many actions
