# ESLint and Prettier

[[7. Tooling & Build.md]]

---

ESLint catches bugs and enforces code quality. Prettier enforces consistent formatting. Together they keep code clean and consistent.

## ESLint Flat Config

ESLint 9+ uses the flat config format (`eslint.config.js`).

```bash
npm init eslint@latest
# Choose: Flat config, TypeScript, React, Browser
```

### eslint.config.js

```javascript
import js from '@eslint/js';
import tseslint from 'typescript-eslint';
import reactPlugin from 'eslint-plugin-react';
import reactHooks from 'eslint-plugin-react-hooks';
import reactRefresh from 'eslint-plugin-react-refresh';

export default tseslint.config(
  { ignores: ['dist', 'node_modules', '*.config.js'] },
  {
    extends: [js.configs.recommended, ...tseslint.configs.recommended],
    files: ['**/*.{ts,tsx}'],
    plugins: {
      react: reactPlugin,
      'react-hooks': reactHooks,
      'react-refresh': reactRefresh,
    },
    languageOptions: {
      ecmaVersion: 2022,
      sourceType: 'module',
      parserOptions: {
        ecmaFeatures: { jsx: true },
      },
      globals: {
        window: 'readonly',
        document: 'readonly',
      },
    },
    settings: {
      react: {
        version: 'detect',
      },
    },
    rules: {
      ...reactPlugin.configs.recommended.rules,
      'react/react-in-jsx-scope': 'off',
      'react/prop-types': 'off',
      'react-hooks/rules-of-hooks': 'error',
      'react-hooks/exhaustive-deps': 'warn',
      'react-refresh/only-export-components': [
        'warn',
        { allowConstantExport: true },
      ],
    },
  }
);
```

## eslint-plugin-react

```bash
npm install -D eslint-plugin-react
```

Key rules:
- `react/jsx-uses-react` — Prevent React from being marked unused
- `react/jsx-uses-vars` — Prevent variables used in JSX from being marked unused
- `react/prop-types` — Enforce prop types (off if using TypeScript)

---

## eslint-plugin-react-hooks

```bash
npm install -D eslint-plugin-react-hooks
```

- `react-hooks/rules-of-hooks` — Hooks must be called at top level
- `react-hooks/exhaustive-deps` — Dependencies in useEffect/useMemo/useCallback

```typescript
// Exhaustive deps will warn
useEffect(() => {
  fetchUser(userId);
}, []); // Missing userId

// Fix
useEffect(() => {
  fetchUser(userId);
}, [userId]);
```

## Prettier Setup

```bash
npm install -D prettier eslint-config-prettier
```

`eslint-config-prettier` disables ESLint rules that conflict with Prettier.

### .prettierrc

```json
{
  "semi": true,
  "singleQuote": true,
  "tabWidth": 2,
  "trailingComma": "es5",
  "printWidth": 100,
  "bracketSpacing": true,
  "arrowParens": "always",
  "endOfLine": "lf"
}
```

### .prettierignore

```
dist
node_modules
.next
*.min.js
pnpm-lock.yaml
```

## Integration

### Extend Prettier in ESLint

```javascript
// eslint.config.js
import prettier from 'eslint-config-prettier';

export default tseslint.config(
  // ... other config
  prettier
);
```

### Format on Save (VS Code)

```json
// .vscode/settings.json
{
  "editor.formatOnSave": true,
  "editor.defaultFormatter": "esbenp.prettier-vscode",
  "editor.codeActionsOnSave": {
    "source.fixAll.eslint": "explicit"
  }
}
```

## Custom Rules

### Inline Disable

```typescript
// eslint-disable-next-line @typescript-eslint/no-explicit-any
const data: any = response;

/* eslint-disable react-hooks/exhaustive-deps */
useEffect(() => {
  doSomething();
}, []);
/* eslint-enable react-hooks/exhaustive-deps */
```

### Override for Specific Files

```javascript
// eslint.config.js
export default tseslint.config(
  // ... base config
  {
    files: ['**/*.test.{ts,tsx}'],
    rules: {
      '@typescript-eslint/no-explicit-any': 'off',
      '@typescript-eslint/no-non-null-assertion': 'off',
    },
  }
);
```

## Pre-commit Hooks

Use Husky + lint-staged to run ESLint and Prettier on staged files.

```bash
npm install -D husky lint-staged
npx husky init
```

```json
// package.json
{
  "lint-staged": {
    "*.{ts,tsx,js,jsx,json,css,md}": [
      "prettier --write",
      "eslint --fix"
    ]
  }
}
```

```javascript
// .husky/pre-commit
#!/usr/bin/env sh
. "$(dirname -- "$0")/_/husky.sh"
npx lint-staged
```

## Scripts

```json
{
  "scripts": {
    "lint": "eslint .",
    "lint:fix": "eslint . --fix",
    "format": "prettier --write \"src/**/*.{ts,tsx,js,jsx,json,css,md}\"",
    "format:check": "prettier --check \"src/**/*.{ts,tsx,js,jsx,json,css,md}\""
  }
}
```

## TypeScript ESLint

```bash
npm install -D typescript-eslint
```

```javascript
// eslint.config.js
import tseslint from 'typescript-eslint';

export default tseslint.config(
  ...tseslint.configs.recommended,
  ...tseslint.configs.recommendedTypeChecked
);
```

Recommended rules for strict TypeScript:
- `@typescript-eslint/no-unused-vars`
- `@typescript-eslint/no-explicit-any`
- `@typescript-eslint/strict-boolean-expressions`
