# Server Actions

[[6. Server Components.md]]

---

Server Actions are async functions that run on the server. They can be called from Client Components, used as form actions, and trigger revalidation. They enable mutations without separate API routes.

## Basic Server Action

```typescript
"use server";

export async function createPost(formData: FormData) {
  const title = formData.get('title') as string;
  const body = formData.get('body') as string;

  if (!title || !body) {
    return { error: 'Title and body required' };
  }

  await db.posts.create({
    data: { title, body },
  });

  revalidatePath('/posts');
  return { success: true };
}
```

## Form Actions

Use a Server Action as the `action` prop of a form. No JavaScript required for submit (progressive enhancement).

```typescript
// app/posts/new/page.tsx
import { createPost } from './actions';

export default function NewPostPage() {
  return (
    <form action={createPost}>
      <input name="title" placeholder="Title" required />
      <textarea name="body" placeholder="Body" required />
      <button type="submit">Create Post</button>
    </form>
  );
}
```

### With useFormStatus (Pending UI)

```typescript
"use client";

import { useFormStatus } from 'react-dom';

function SubmitButton() {
  const { pending } = useFormStatus();

  return (
    <button type="submit" disabled={pending}>
      {pending ? 'Creating...' : 'Create Post'}
    </button>
  );
}

export default function NewPostPage() {
  return (
    <form action={createPost}>
      <input name="title" required />
      <textarea name="body" required />
      <SubmitButton />
    </form>
  );
}
```

## Server Mutations

Call Server Actions from event handlers (requires Client Component):

```typescript
"use client";

import { likePost } from './actions';

function LikeButton({ postId }: { postId: string }) {
  return (
    <button
      onClick={async () => {
        await likePost(postId);
      }}
    >
      Like
    </button>
  );
}
```

### With useTransition (Pending State)

```typescript
"use client";

import { useTransition } from 'react';
import { likePost } from './actions';

function LikeButton({ postId }: { postId: string }) {
  const [isPending, startTransition] = useTransition();

  return (
    <button
      disabled={isPending}
      onClick={() => {
        startTransition(async () => {
          await likePost(postId);
        });
      }}
    >
      {isPending ? 'Liking...' : 'Like'}
    </button>
  );
}
```

## Revalidation

After mutating data, revalidate so the UI reflects changes:

```typescript
"use server";

import { revalidatePath, revalidateTag } from 'next/cache';

export async function updateUser(id: string, data: UserUpdate) {
  await db.user.update({ where: { id }, data });
  revalidatePath(`/users/${id}`);
  revalidatePath('/users');
}

export async function refreshProducts() {
  revalidateTag('products');  // If fetch used { next: { tags: ['products'] } }
}
```

## Error Handling

### Return Object Pattern

```typescript
"use server";

export async function createUser(formData: FormData) {
  try {
    const email = formData.get('email') as string;
    if (!email) return { error: 'Email required' };

    await db.user.create({ data: { email } });
    revalidatePath('/users');
    return { success: true };
  } catch (e) {
    return { error: 'Failed to create user' };
  }
}
```

### Client-Side Handling

```typescript
"use client";

import { createUser } from './actions';

function SignupForm() {
  const [result, setResult] = useState<{ error?: string; success?: boolean } | null>(null);

  return (
    <form
      action={async (formData) => {
        const res = await createUser(formData);
        setResult(res);
        if (res.success) redirect('/dashboard');
      }}
    >
      <input name="email" type="email" />
      {result?.error && <p className="error">{result.error}</p>}
      <button type="submit">Sign up</button>
    </form>
  );
}
```

### useActionState (React 19)

```typescript
"use client";

import { useActionState } from 'react';
import { createUser } from './actions';

function SignupForm() {
  const [state, formAction] = useActionState(createUser, null);

  return (
    <form action={formAction}>
      <input name="email" type="email" />
      {state?.error && <p className="error">{state.error}</p>}
      <button type="submit">Sign up</button>
    </form>
  );
}
```

## Progressive Enhancement

Forms with `action` work without JavaScript. If JS loads, you get:
- `useFormStatus` for pending state
- `useActionState` for state/errors
- Programmatic submission via `formAction`

```typescript
<form action={createPost}>
  {/* Works without JS: form submits, page navigates */}
  {/* With JS: can show loading, handle errors, prevent navigation */}
</form>
```

## Inline vs Module-Level Actions

### Module-Level (Recommended)

```typescript
// actions.ts
"use server";

export async function createPost(formData: FormData) {
  // ...
}
```

### Inline

```typescript
async function createPost(formData: FormData) {
  "use server";
  // ...
}

export default function Page() {
  return <form action={createPost}>...</form>;
}
```

Inline actions are recreated per render; module-level are stable.

## Passing Arguments

Server Actions can take serializable arguments:

```typescript
"use server";

export async function deletePost(postId: string) {
  await db.posts.delete({ where: { id: postId } });
  revalidatePath('/posts');
}

// Client
<button onClick={() => deletePost(post.id)}>Delete</button>
```
