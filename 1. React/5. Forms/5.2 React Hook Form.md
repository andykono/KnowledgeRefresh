# React Hook Form

[[5. Forms.md]]

---

React Hook Form minimizes re-renders by using uncontrolled inputs under the hood while exposing a controlled-like API. It integrates well with Zod, Yup, and TypeScript for validation and type safety.

## Setup

```bash
npm install react-hook-form
# With Zod validation
npm install @hookform/resolvers zod
# With Yup validation
npm install @hookform/resolvers yup
```

## Basic Usage

```typescript
import { useForm } from 'react-hook-form';

interface FormValues {
  email: string;
  password: string;
}

function LoginForm() {
  const {
    register,
    handleSubmit,
    formState: { errors },
  } = useForm<FormValues>();

  const onSubmit = (data: FormValues) => {
    console.log(data);  // { email: string, password: string }
  };

  return (
    <form onSubmit={handleSubmit(onSubmit)}>
      <input
        {...register('email', { required: 'Email is required' })}
        type="email"
      />
      {errors.email && <span>{errors.email.message}</span>}

      <input
        {...register('password', {
          required: 'Password is required',
          minLength: { value: 8, message: 'Min 8 characters' },
        })}
        type="password"
      />
      {errors.password && <span>{errors.password.message}</span>}

      <button type="submit">Submit</button>
    </form>
  );
}
```

## register()

`register` returns props to spread onto inputs: `name`, `onChange`, `onBlur`, `ref`. It wires the input to the form state without controlled re-renders.

```typescript
const { register } = useForm();

// Spread onto input
<input {...register('fieldName')} />

// Equivalent to:
<input
  name="fieldName"
  onChange={...}
  onBlur={...}
  ref={...}
/>
```

### Register Options

```typescript
register('email', {
  required: 'Email is required',
  minLength: { value: 5, message: 'Min 5 chars' },
  maxLength: { value: 100, message: 'Max 100 chars' },
  pattern: {
    value: /^[A-Z0-9._%+-]+@[A-Z0-9.-]+\.[A-Z]{2,}$/i,
    message: 'Invalid email',
  },
  validate: (value) => value !== 'admin' || 'Username not allowed',
});
```

## handleSubmit

`handleSubmit` wraps your submit handler and:
- Prevents default form submission
- Validates all fields
- Calls your handler only if valid; otherwise sets `formState.errors`

```typescript
const onSubmit = (data: FormValues) => {
  await fetch('/api/login', {
    method: 'POST',
    body: JSON.stringify(data),
  });
};

<form onSubmit={handleSubmit(onSubmit)}>
```

### Invalid Submit

When validation fails, `onSubmit` is not called. Access errors via `formState.errors`:

```typescript
const { formState: { errors } } = useForm();

// errors.email?.message, errors.password?.message, etc.
```

## Validation with Zod

```typescript
import { useForm } from 'react-hook-form';
import { zodResolver } from '@hookform/resolvers/zod';
import { z } from 'zod';

const schema = z.object({
  email: z.string().email('Invalid email'),
  password: z.string().min(8, 'Min 8 characters'),
  age: z.number().min(18).optional(),
});

type FormValues = z.infer<typeof schema>;

function Form() {
  const {
    register,
    handleSubmit,
    formState: { errors },
  } = useForm<FormValues>({
    resolver: zodResolver(schema),
  });

  return (
    <form onSubmit={handleSubmit((data) => console.log(data))}>
      <input {...register('email')} />
      {errors.email && <span>{errors.email.message}</span>}
      <input {...register('password')} type="password" />
      {errors.password && <span>{errors.password.message}</span>}
      <input {...register('age', { valueAsNumber: true })} type="number" />
      {errors.age && <span>{errors.age.message}</span>}
      <button type="submit">Submit</button>
    </form>
  );
}
```

## Validation with Yup

```typescript
import { useForm } from 'react-hook-form';
import { yupResolver } from '@hookform/resolvers/yup';
import * as yup from 'yup';

const schema = yup.object({
  email: yup.string().email().required(),
  password: yup.string().min(8).required(),
}).required();

type FormValues = yup.InferType<typeof schema>;

function Form() {
  const { register, handleSubmit, formState: { errors } } = useForm<FormValues>({
    resolver: yupResolver(schema),
  });

  return (
    <form onSubmit={handleSubmit((data) => console.log(data))}>
      <input {...register('email')} />
      {errors.email && <span>{errors.email.message}</span>}
      <input {...register('password')} type="password" />
      {errors.password && <span>{errors.password.message}</span>}
      <button type="submit">Submit</button>
    </form>
  );
}
```

## TypeScript Integration

```typescript
interface FormValues {
  email: string;
  password: string;
  remember: boolean;
  role: 'admin' | 'user';
}

const { register, handleSubmit } = useForm<FormValues>();

// register is type-safe
register('email');     // ✅
register('invalid');   // ❌ TypeScript error

// handleSubmit receives typed data
handleSubmit((data) => {
  data.email;    // string
  data.role;     // 'admin' | 'user'
});
```

## useFieldArray

For dynamic lists (add/remove rows):

```typescript
import { useForm, useFieldArray } from 'react-hook-form';

interface FormValues {
  items: { name: string; quantity: number }[];
}

function DynamicForm() {
  const { register, control, handleSubmit } = useForm<FormValues>({
    defaultValues: { items: [{ name: '', quantity: 0 }] },
  });

  const { fields, append, remove } = useFieldArray({
    control,
    name: 'items',
  });

  return (
    <form onSubmit={handleSubmit((data) => console.log(data))}>
      {fields.map((field, index) => (
        <div key={field.id}>
          <input {...register(`items.${index}.name`)} placeholder="Name" />
          <input
            {...register(`items.${index}.quantity`, { valueAsNumber: true })}
            type="number"
          />
          <button type="button" onClick={() => remove(index)}>Remove</button>
        </div>
      ))}
      <button type="button" onClick={() => append({ name: '', quantity: 0 })}>
        Add item
      </button>
      <button type="submit">Submit</button>
    </form>
  );
}
```

## useWatch

Subscribe to specific field values without re-rendering the whole form:

```typescript
import { useForm, useWatch } from 'react-hook-form';

function Form() {
  const { control, register } = useForm();

  const watchedEmail = useWatch({ control, name: 'email', defaultValue: '' });

  return (
    <form>
      <input {...register('email')} />
      <p>You typed: {watchedEmail}</p>
    </form>
  );
}
```

## Complex Forms: Nested Objects

```typescript
interface FormValues {
  user: {
    name: string;
    address: {
      street: string;
      city: string;
    };
  };
}

function NestedForm() {
  const { register, handleSubmit } = useForm<FormValues>();

  return (
    <form onSubmit={handleSubmit(console.log)}>
      <input {...register('user.name')} />
      <input {...register('user.address.street')} />
      <input {...register('user.address.city')} />
      <button type="submit">Submit</button>
    </form>
  );
}
```

## Default Values and Reset

```typescript
const { register, reset } = useForm<FormValues>({
  defaultValues: {
    email: 'user@example.com',
    remember: true,
  },
});

// Reset to defaults
reset();

// Reset to specific values
reset({ email: '', password: '' });
```

## Mode: When to Validate

```typescript
useForm({
  mode: 'onSubmit',   // Default: validate on submit
  mode: 'onBlur',    // Validate on blur
  mode: 'onChange',  // Validate on every change
  mode: 'onTouched', // After first blur, validate on change
  mode: 'all',       // onSubmit + onBlur + onChange
});
```
