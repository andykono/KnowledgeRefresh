# Validation Patterns

[[5. Forms.md]]

Form validation ensures data integrity before submission. This guide covers schema-based validation with Zod and Yup, integration with React Hook Form, validation timing strategies, and server-side error handling.

## Schema Validation with Zod

Zod is a TypeScript-first schema validation library. Define schemas and infer types automatically:

```typescript
import { z } from 'zod';

const userSchema = z.object({
  name: z.string().min(2, 'Name must be at least 2 characters'),
  email: z.string().email('Invalid email address'),
  age: z.number().min(18, 'Must be 18 or older').optional(),
  role: z.enum(['admin', 'user', 'guest']),
});

type UserFormData = z.infer<typeof userSchema>;
```

### Basic Zod Types

```typescript
z.string()           // String
z.number()           // Number
z.boolean()          // Boolean
z.date()             // Date
z.array(z.string())  // Array of strings
z.object({ ... })    // Object shape
z.enum(['a', 'b'])   // Union of literals
```

### Custom Validators

```typescript
const passwordSchema = z
  .string()
  .min(8, 'Password must be 8+ characters')
  .regex(/[A-Z]/, 'Must contain uppercase')
  .regex(/[0-9]/, 'Must contain number');

const phoneSchema = z.string().refine(
  (val) => /^\d{3}-\d{3}-\d{4}$/.test(val),
  { message: 'Phone must be XXX-XXX-XXXX' }
);
```

### Error Messages

```typescript
const schema = z.object({
  email: z.string({
    required_error: 'Email is required',
    invalid_type_error: 'Email must be a string',
  }).email('Invalid email format'),
});
```

## Yup Equivalent

Yup offers similar API with slightly different syntax:

```typescript
import * as yup from 'yup';

const userSchema = yup.object({
  name: yup.string().min(2, 'Name must be at least 2 characters').required(),
  email: yup.string().email('Invalid email').required(),
  age: yup.number().min(18).optional(),
  role: yup.string().oneOf(['admin', 'user', 'guest']).required(),
});

type UserFormData = yup.InferType<typeof userSchema>;
```

## Integrating Zod with React Hook Form

Use `@hookform/resolvers` to connect Zod (or Yup) with React Hook Form:

```typescript
import { useForm } from 'react-hook-form';
import { zodResolver } from '@hookform/resolvers/zod';
import { z } from 'zod';

const schema = z.object({
  email: z.string().email(),
  password: z.string().min(8),
});

type FormData = z.infer<typeof schema>;

function LoginForm() {
  const {
    register,
    handleSubmit,
    formState: { errors },
  } = useForm<FormData>({
    resolver: zodResolver(schema),
  });

  const onSubmit = (data: FormData) => console.log(data);

  return (
    <form onSubmit={handleSubmit(onSubmit)}>
      <input {...register('email')} />
      {errors.email && <span>{errors.email.message}</span>}
      <input {...register('password')} type="password" />
      {errors.password && <span>{errors.password.message}</span>}
      <button type="submit">Submit</button>
    </form>
  );
}
```

## Validation Timing Strategies

| Strategy | When | Pros | Cons |
|----------|------|------|------|
| **onChange** | Every keystroke | Immediate feedback | Can feel intrusive, more re-renders |
| **onBlur** | When field loses focus | Less noisy, good UX | Delayed feedback |
| **onSubmit** | On form submit | Simple, fewer validations | User sees errors late |

### React Hook Form Modes

```typescript
useForm<FormData>({
  resolver: zodResolver(schema),
  mode: 'onBlur',      // Validate on blur (default: onSubmit)
  // mode: 'onChange', // Validate on every change
  // mode: 'onTouched', // After first blur, then onChange
  // mode: 'all',      // onChange + onBlur
});
```

### Custom Timing

```typescript
// Validate specific field on blur
<input
  {...register('email', {
    onBlur: () => trigger('email'),
  })}
/>

// Manual validation
const { trigger } = useForm();
const isValid = await trigger(); // Validate all fields
const isEmailValid = await trigger('email'); // Single field
```

## Server-Side Validation and Error Mapping

Map server errors back to form fields:

```typescript
const { setError } = useForm<FormData>();

async function onSubmit(data: FormData) {
  try {
    const res = await fetch('/api/register', {
      method: 'POST',
      body: JSON.stringify(data),
    });
    const result = await res.json();

    if (!res.ok && result.errors) {
      result.errors.forEach((err: { field: keyof FormData; message: string }) => {
        setError(err.field, { type: 'server', message: err.message });
      });
    }
  } catch (error) {
    setError('root', { message: 'Network error' });
  }
}
```

## Displaying Field Errors

```typescript
function FormField({ name, label, ...props }: FormFieldProps) {
  const { register, formState: { errors } } = useFormContext<FormData>();
  const error = errors[name];

  return (
    <div>
      <label htmlFor={name}>{label}</label>
      <input
        id={name}
        {...register(name)}
        aria-invalid={!!error}
        aria-describedby={error ? `${name}-error` : undefined}
        {...props}
      />
      {error && (
        <span id={`${name}-error`} role="alert" className="error">
          {error.message}
        </span>
      )}
    </div>
  );
}
```

## Cross-Field Validation

Validate fields that depend on each other:

```typescript
const schema = z
  .object({
    password: z.string().min(8),
    confirmPassword: z.string(),
  })
  .refine((data) => data.password === data.confirmPassword, {
    message: "Passwords don't match",
    path: ['confirmPassword'],
  });

// Or with superRefine for multiple cross-field rules
const schema = z.object({
  startDate: z.date(),
  endDate: z.date(),
}).superRefine((data, ctx) => {
  if (data.endDate < data.startDate) {
    ctx.addIssue({
      code: z.ZodIssueCode.custom,
      message: 'End date must be after start date',
      path: ['endDate'],
    });
  }
});
```

### React Hook Form with Cross-Field

```typescript
<input
  {...register('confirmPassword', {
    validate: (value) =>
      value === watch('password') || "Passwords don't match",
  })}
/>
```
