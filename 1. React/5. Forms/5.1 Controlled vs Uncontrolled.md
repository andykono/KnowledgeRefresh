# Controlled vs Uncontrolled Components

[[5. Forms.md]]

React offers two approaches for form handling: **controlled** components (React state drives the value) and **uncontrolled** components (the DOM holds the value). Choosing the right approach depends on your use case.

## Controlled Components

In a controlled component, React state is the single source of truth. Each input's value is tied to state, and changes update state via `onChange`:

```typescript
import { useState, ChangeEvent, FormEvent } from 'react';

function ControlledForm() {
  const [email, setEmail] = useState('');
  const [password, setPassword] = useState('');

  const handleEmailChange = (e: ChangeEvent<HTMLInputElement>) => {
    setEmail(e.target.value);
  };

  const handlePasswordChange = (e: ChangeEvent<HTMLInputElement>) => {
    setPassword(e.target.value);
  };

  const handleSubmit = (e: FormEvent<HTMLFormElement>) => {
    e.preventDefault();
    console.log({ email, password });
  };

  return (
    <form onSubmit={handleSubmit}>
      <input
        type="email"
        value={email}
        onChange={handleEmailChange}
      />
      <input
        type="password"
        value={password}
        onChange={handlePasswordChange}
      />
      <button type="submit">Submit</button>
    </form>
  );
}
```

## Uncontrolled Components

Uncontrolled components use refs to access DOM values. The DOM maintains the value; you read it when needed (e.g., on submit):

```typescript
import { useRef, FormEvent } from 'react';

function UncontrolledForm() {
  const emailRef = useRef<HTMLInputElement>(null);
  const passwordRef = useRef<HTMLInputElement>(null);

  const handleSubmit = (e: FormEvent<HTMLFormElement>) => {
    e.preventDefault();
    const email = emailRef.current?.value ?? '';
    const password = passwordRef.current?.value ?? '';
    console.log({ email, password });
  };

  return (
    <form onSubmit={handleSubmit}>
      <input ref={emailRef} type="email" defaultValue="" />
      <input ref={passwordRef} type="password" defaultValue="" />
      <button type="submit">Submit</button>
    </form>
  );
}
```

Note: Use `defaultValue` (not `value`) for uncontrolled inputs to avoid making them controlled.

## TypeScript Form Event Types

| Event | Type | Use Case |
|-------|------|----------|
| Input change | `ChangeEvent<HTMLInputElement>` | Text, checkbox, radio |
| Form submit | `FormEvent<HTMLFormElement>` | Form submission |
| Select change | `ChangeEvent<HTMLSelectElement>` | Dropdown selection |
| Textarea change | `ChangeEvent<HTMLTextAreaElement>` | Multi-line text |

```typescript
const handleInputChange = (e: ChangeEvent<HTMLInputElement>) => {
  setValue(e.target.value);
};

const handleCheckboxChange = (e: ChangeEvent<HTMLInputElement>) => {
  setChecked(e.target.checked);
};

const handleSelectChange = (e: ChangeEvent<HTMLSelectElement>) => {
  setSelected(e.target.value);
};
```

## When to Use Which Approach

| Controlled | Uncontrolled |
|------------|--------------|
| Need validation on every keystroke | Simple forms, minimal validation |
| Conditional disabling of fields | File inputs |
| Dynamic UI based on input (e.g., character count) | Integrating with non-React libraries |
| Formatting input as user types | Performance-critical forms with many fields |
| Multi-step wizards | Quick prototypes |

## Hybrid Approaches

You can mix both: use refs for some fields and state for others:

```typescript
function HybridForm() {
  const [email, setEmail] = useState('');
  const fileRef = useRef<HTMLInputElement>(null);

  const handleSubmit = (e: FormEvent<HTMLFormElement>) => {
    e.preventDefault();
    const file = fileRef.current?.files?.[0];
    console.log({ email, file });
  };

  return (
    <form onSubmit={handleSubmit}>
      <input
        type="email"
        value={email}
        onChange={(e) => setEmail(e.target.value)}
      />
      <input ref={fileRef} type="file" />
      <button type="submit">Submit</button>
    </form>
  );
}
```

## Common Patterns

### Multi-Field Forms (Controlled)

Use a single state object for cleaner code:

```typescript
interface FormState {
  name: string;
  email: string;
  age: string;
}

function MultiFieldForm() {
  const [form, setForm] = useState<FormState>({
    name: '',
    email: '',
    age: '',
  });

  const handleChange = (e: ChangeEvent<HTMLInputElement>) => {
    const { name, value } = e.target;
    setForm((prev) => ({ ...prev, [name]: value }));
  };

  return (
    <form>
      <input name="name" value={form.name} onChange={handleChange} />
      <input name="email" value={form.email} onChange={handleChange} />
      <input name="age" value={form.age} onChange={handleChange} />
    </form>
  );
}
```

### Dynamic Forms (Controlled)

Add/remove fields dynamically with state:

```typescript
function DynamicForm() {
  const [items, setItems] = useState<string[]>(['']);

  const addItem = () => setItems((prev) => [...prev, '']);
  const removeItem = (index: number) =>
    setItems((prev) => prev.filter((_, i) => i !== index));
  const updateItem = (index: number, value: string) =>
    setItems((prev) => prev.map((v, i) => (i === index ? value : v)));

  return (
    <form>
      {items.map((item, i) => (
        <div key={i}>
          <input
            value={item}
            onChange={(e) => updateItem(i, e.target.value)}
          />
          <button type="button" onClick={() => removeItem(i)}>Remove</button>
        </div>
      ))}
      <button type="button" onClick={addItem}>Add</button>
    </form>
  );
}
```

### File Inputs (Uncontrolled)

File inputs are always uncontrolled—you cannot programmatically set their value:

```typescript
function FileUpload() {
  const fileRef = useRef<HTMLInputElement>(null);

  const handleSubmit = (e: FormEvent<HTMLFormElement>) => {
    e.preventDefault();
    const files = fileRef.current?.files;
    if (files?.length) {
      Array.from(files).forEach((file) => console.log(file.name));
    }
  };

  return (
    <form onSubmit={handleSubmit}>
      <input ref={fileRef} type="file" multiple accept="image/*" />
      <button type="submit">Upload</button>
    </form>
  );
}
```
