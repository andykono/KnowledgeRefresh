# Tailwind CSS

[[7. Styling.md]]

Tailwind CSS is a utility-first CSS framework that provides low-level utility classes to build custom designs without writing custom CSS. It pairs exceptionally well with React and TypeScript.

## Setup with Vite + React

```bash
npm create vite@latest my-app -- --template react-ts
cd my-app
npm install -D tailwindcss postcss autoprefixer
npx tailwindcss init -p
```

Configure `tailwind.config.ts`:

```typescript
import type { Config } from 'tailwindcss'

export default {
  content: ['./index.html', './src/**/*.{js,ts,jsx,tsx}'],
  theme: { extend: {} },
  plugins: [],
} satisfies Config
```

Add directives to `src/index.css`:

```css
@tailwind base;
@tailwind components;
@tailwind utilities;
```

## Utility Classes Basics

Apply styles directly in JSX using utility classes:

```tsx
function Card({ children }: { children: React.ReactNode }) {
  return (
    <div className="rounded-lg border border-gray-200 bg-white p-6 shadow-md">
      {children}
    </div>
  )
}
```

Common utilities: `flex`, `grid`, `gap-4`, `text-center`, `font-bold`, `text-blue-600`, `hover:bg-gray-100`, `transition-colors`.

## Responsive Design

Tailwind uses mobile-first breakpoints: `sm` (640px), `md` (768px), `lg` (1024px), `xl` (1280px), `2xl` (1536px).

```tsx
<div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-4">
  {items.map((item) => (
    <Card key={item.id}>{item.title}</Card>
  ))}
</div>
```

## Dark Mode (Class Strategy)

In `tailwind.config.ts`:

```typescript
export default {
  darkMode: 'class',
  // ...
}
```

Toggle dark mode by adding `dark` to `<html>`:

```tsx
function ThemeToggle() {
  const [dark, setDark] = useState(false)
  useEffect(() => {
    document.documentElement.classList.toggle('dark', dark)
  }, [dark])
  return (
    <button
      onClick={() => setDark((d) => !d)}
      className="rounded bg-gray-200 px-4 py-2 dark:bg-gray-800 dark:text-white"
    >
      Toggle
    </button>
  )
}
```

## cn() Utility with clsx + tailwind-merge

Install: `npm install clsx tailwind-merge`. Create `lib/utils.ts`:

```typescript
import { clsx, type ClassValue } from 'clsx'
import { twMerge } from 'tailwind-merge'

export function cn(...inputs: ClassValue[]) {
  return twMerge(clsx(inputs))
}
```

Usage for conditional and merged classes:

```tsx
function Button({
  variant = 'primary',
  className,
  ...props
}: ButtonProps) {
  return (
    <button
      className={cn(
        'rounded px-4 py-2 font-medium transition',
        variant === 'primary' && 'bg-blue-600 text-white hover:bg-blue-700',
        variant === 'secondary' && 'bg-gray-200 hover:bg-gray-300',
        className
      )}
      {...props}
    />
  )
}
```

## Component Patterns

### Button

```tsx
type ButtonVariant = 'primary' | 'secondary' | 'ghost'

interface ButtonProps extends React.ButtonHTMLAttributes<HTMLButtonElement> {
  variant?: ButtonVariant
}

const variantStyles: Record<ButtonVariant, string> = {
  primary: 'bg-blue-600 text-white hover:bg-blue-700',
  secondary: 'bg-gray-200 hover:bg-gray-300',
  ghost: 'hover:bg-gray-100',
}

export function Button({ variant = 'primary', className, ...props }: ButtonProps) {
  return (
    <button
      className={cn('rounded px-4 py-2 font-medium', variantStyles[variant], className)}
      {...props}
    />
  )
}
```

### Card

```tsx
export function Card({ className, children }: { className?: string; children: React.ReactNode }) {
  return (
    <div className={cn('rounded-xl border bg-white p-6 shadow-sm', className)}>
      {children}
    </div>
  )
}
```

### Input

```tsx
export function Input({ className, ...props }: React.InputHTMLAttributes<HTMLInputElement>) {
  return (
    <input
      className={cn(
        'w-full rounded border border-gray-300 px-3 py-2 focus:border-blue-500 focus:outline-none focus:ring-1 focus:ring-blue-500',
        className
      )}
      {...props}
    />
  )
}
```

## Conditional Styling

```tsx
function StatusBadge({ status }: { status: 'success' | 'error' | 'pending' }) {
  return (
    <span
      className={cn(
        'rounded-full px-2 py-0.5 text-sm',
        status === 'success' && 'bg-green-100 text-green-800',
        status === 'error' && 'bg-red-100 text-red-800',
        status === 'pending' && 'bg-yellow-100 text-yellow-800'
      )}
    >
      {status}
    </span>
  )
}
```

## Animation Utilities

Tailwind includes `animate-spin`, `animate-pulse`, `animate-bounce`. Use `transition-*` for smooth changes:

```tsx
<button className="transition-all duration-300 hover:scale-105 active:scale-95">
  Hover me
</button>
```

For custom animations, add to config:

```typescript
theme: {
  extend: {
    keyframes: {
      fadeIn: { '0%': { opacity: '0' }, '100%': { opacity: '1' } },
    },
    animation: { fadeIn: 'fadeIn 0.3s ease-out' },
  },
}
```
