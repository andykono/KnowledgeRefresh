# CSS Modules

[[7. Styling.md]]

---

CSS Modules provide locally scoped CSS by default. Class names are automatically hashed at build time, eliminating naming collisions and enabling predictable, maintainable styles.

## Setup

### Vite (Default Support)

Vite supports CSS Modules out of the box. Name your file with `.module.css`:

```
src/
  components/
    Button/
      Button.tsx
      Button.module.css
```

### Create React App

CRA also supports CSS Modules by default with the `.module.css` convention.

### Manual Webpack Config

```javascript
// webpack.config.js
{
  test: /\.module\.css$/,
  use: [
    'style-loader',
    {
      loader: 'css-loader',
      options: {
        modules: {
          localIdentName: '[name]__[local]--[hash:base64:5]',
        },
      },
    },
  ],
}
```

## Basic Usage with TypeScript

```typescript
// Button.module.css
.primary {
  background: #007bff;
  color: white;
  padding: 0.5rem 1rem;
  border-radius: 4px;
}

.secondary {
  background: #6c757d;
  color: white;
  padding: 0.5rem 1rem;
  border-radius: 4px;
}
```

```typescript
// Button.tsx
import styles from './Button.module.css';

interface ButtonProps {
  variant?: 'primary' | 'secondary';
  children: React.ReactNode;
}

export function Button({ variant = 'primary', children }: ButtonProps) {
  const className = variant === 'primary' ? styles.primary : styles.secondary;
  return <button className={className}>{children}</button>;
}
```

## TypeScript Typings

For strict typing of class names, create a declaration file:

```typescript
// css-modules.d.ts or vite-env.d.ts
declare module '*.module.css' {
  const classes: { readonly [key: string]: string };
  export default classes;
}
```

Usage with typed class access:

```typescript
// TypeScript will error if 'invalidClass' doesn't exist
const className = styles.primary;  // ✅
const bad = styles.invalidClass;   // ❌ Type error
```

## Composition

Use `composes` to reuse styles from another class:

```css
/* Button.module.css */
.base {
  padding: 0.5rem 1rem;
  border-radius: 4px;
  font-weight: 500;
}

.primary {
  composes: base;
  background: #007bff;
  color: white;
}

.primary:hover {
  background: #0056b3;
}
```

Compose from another module:

```css
/* Button.module.css */
.primary {
  composes: button from '../shared/typography.module.css';
  background: #007bff;
}
```

## Global Styles

Use `:global()` for unscoped styles (use sparingly):

```css
/* Layout.module.css */
.container {
  max-width: 1200px;
  margin: 0 auto;
}

/* Global override for third-party components */
:global(.react-datepicker) {
  font-family: inherit;
}

/* Global keyframes */
:global(@keyframes fadeIn) {
  from { opacity: 0; }
  to { opacity: 1; }
}
```

## Naming Conventions

### BEM-like (Recommended)

```css
/* Card.module.css */
.card { }
.card__header { }
.card__body { }
.card__footer { }
.card--elevated { }
```

### camelCase (JavaScript-friendly)

```css
.cardHeader { }
.cardBody { }
.cardElevated { }
```

```typescript
// Access: styles.cardHeader, styles.cardBody
```

## Dynamic Class Names

```typescript
// Button.tsx
import styles from './Button.module.css';

interface ButtonProps {
  variant: 'primary' | 'secondary' | 'danger';
  size?: 'sm' | 'md' | 'lg';
  disabled?: boolean;
  children: React.ReactNode;
}

export function Button({ variant, size = 'md', disabled, children }: ButtonProps) {
  const classNames = [
    styles.button,
    styles[variant],
    styles[`size_${size}`],
    disabled && styles.disabled,
  ].filter(Boolean).join(' ');

  return <button className={classNames} disabled={disabled}>{children}</button>;
}
```

## Combining with External Classes

```typescript
import styles from './Button.module.css';

export function Button({ className, ...props }: ButtonProps & { className?: string }) {
  return (
    <button
      className={`${styles.button} ${className ?? ''}`.trim()}
      {...props}
    />
  );
}
```

## Complete Example

```typescript
// Card/Card.tsx
import styles from './Card.module.css';

interface CardProps {
  title: string;
  children: React.ReactNode;
  variant?: 'default' | 'outlined' | 'elevated';
}

export function Card({ title, children, variant = 'default' }: CardProps) {
  return (
    <article className={`${styles.card} ${styles[`card--${variant}`]}`}>
      <header className={styles.cardHeader}>
        <h2 className={styles.cardTitle}>{title}</h2>
      </header>
      <div className={styles.cardBody}>
        {children}
      </div>
    </article>
  );
}
```

```css
/* Card.module.css */
.card {
  border-radius: 8px;
  overflow: hidden;
}

.card--default {
  background: white;
  border: 1px solid #e0e0e0;
}

.card--outlined {
  background: transparent;
  border: 2px solid #007bff;
}

.card--elevated {
  background: white;
  box-shadow: 0 4px 6px rgba(0, 0, 0, 0.1);
}

.cardHeader {
  padding: 1rem 1.5rem;
  border-bottom: 1px solid #eee;
}

.cardTitle {
  margin: 0;
  font-size: 1.25rem;
}

.cardBody {
  padding: 1.5rem;
}
```
