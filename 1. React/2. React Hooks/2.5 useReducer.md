# useReducer Hook

[[2. React Hooks.md]]

---

useReducer manages complex state with a reducer function.

## Basic Usage

```typescript
interface State {
  count: number;
}

interface Action {
  type: 'INCREMENT' | 'DECREMENT' | 'RESET';
}

function reducer(state: State, action: Action): State {
  switch (action.type) {
    case 'INCREMENT':
      return { count: state.count + 1 };
    case 'DECREMENT':
      return { count: state.count - 1 };
    case 'RESET':
      return { count: 0 };
    default:
      return state;
  }
}

function Counter() {
  const [state, dispatch] = useReducer(reducer, { count: 0 });

  return (
    <div>
      <p>Count: {state.count}</p>
      <button onClick={() => dispatch({ type: 'INCREMENT' })}>+</button>
      <button onClick={() => dispatch({ type: 'DECREMENT' })}>-</button>
      <button onClick={() => dispatch({ type: 'RESET' })}>Reset</button>
    </div>
  );
}
```

## With Payload

```typescript
interface Action {
  type: 'ADD' | 'REMOVE';
  payload: string;  // New items have payload
}

function reducer(state: string[], action: Action): string[] {
  switch (action.type) {
    case 'ADD':
      return [...state, action.payload];
    case 'REMOVE':
      return state.filter(item => item !== action.payload);
    default:
      return state;
  }
}

function List() {
  const [items, dispatch] = useReducer(reducer, []);

  return (
    <>
      <button onClick={() => dispatch({ type: 'ADD', payload: 'Item 1' })}>
        Add
      </button>
      <ul>
        {items.map(item => (
          <li key={item}>
            {item}
            <button onClick={() => dispatch({ type: 'REMOVE', payload: item })}>
              Delete
            </button>
          </li>
        ))}
      </ul>
    </>
  );
}
```

## vs useState

| Scenario | useState | useReducer |
| :-- | :-- | :-- |
| **Simple state** | ✅ Better | ❌ Overkill |
| **Complex logic** | ❌ Hard | ✅ Better |
| **Multiple related** | 🟡 Messy | ✅ Clean |

```typescript
// useState: Many variables
const [name, setName] = useState('');
const [email, setEmail] = useState('');
const [phone, setPhone] = useState('');

// useReducer: One dispatch
const [form, dispatch] = useReducer(formReducer, initialForm);
```

## Lazy Initialization

```typescript
const [state, dispatch] = useReducer(
  reducer,
  undefined,  // Initial state undefined
  init  // Initialization function
);

function init(initialValue) {
  // Complex initialization
  return { count: expensiveCalculation() };
}
```
