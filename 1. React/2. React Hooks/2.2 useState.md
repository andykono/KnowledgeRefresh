# useState Hook

[[2. React Hooks.md]]

---

useState lets you add state to functional components.

## Basic Usage

```typescript
import { useState } from 'react';

function Counter() {
  const [count, setCount] = useState(0);

  return (
    <div>
      <p>Count: {count}</p>
      <button onClick={() => setCount(count + 1)}>Increment</button>
    </div>
  );
}
```

**Anatomy:**
```typescript
const [state, setState] = useState(initialValue);
// state: current value
// setState: function to update state
// initialValue: what state starts as
```

## Multiple State Variables

```typescript
function LoginForm() {
  const [email, setEmail] = useState('');
  const [password, setPassword] = useState('');
  const [isLoading, setIsLoading] = useState(false);

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    setIsLoading(true);
    try {
      await loginUser(email, password);
    } finally {
      setIsLoading(false);
    }
  };

  return (
    <form onSubmit={handleSubmit}>
      <input
        value={email}
        onChange={(e) => setEmail(e.target.value)}
        type="email"
      />
      <input
        value={password}
        onChange={(e) => setPassword(e.target.value)}
        type="password"
      />
      <button disabled={isLoading}>
        {isLoading ? 'Logging in...' : 'Login'}
      </button>
    </form>
  );
}
```

## State with Objects

```typescript
interface User {
  name: string;
  email: string;
  age: number;
}

function UserForm() {
  const [user, setUser] = useState<User>({
    name: '',
    email: '',
    age: 0
  });

  // ❌ DON'T: Mutate directly
  const handleBadChange = (name: string) => {
    user.name = name;  // Won't trigger re-render!
  };

  // ✅ DO: Create new object
  const handleChange = (name: string) => {
    setUser(prev => ({ ...prev, name }));
  };

  // ✅ OR: Spread with override
  const handleChangeAlt = (newData: Partial<User>) => {
    setUser({ ...user, ...newData });
  };

  return (
    <div>
      <input
        value={user.name}
        onChange={(e) => handleChange(e.target.value)}
      />
      <input
        value={user.email}
        onChange={(e) => handleChangeAlt({ email: e.target.value })}
      />
    </div>
  );
}
```

## Lazy Initialization

Use a function for expensive calculations:

```typescript
// ❌ BAD: Runs every render
function Component() {
  const [state, setState] = useState(expensiveCalculation());
}

// ✅ GOOD: Runs only once
function Component() {
  const [state, setState] = useState(() => expensiveCalculation());
}

// Example
const [todos, setTodos] = useState(() => {
  // Only runs on first render
  return JSON.parse(localStorage.getItem('todos') || '[]');
});
```

## Update Patterns

### Direct Update
```typescript
const [count, setCount] = useState(0);
setCount(5);  // Set to 5
```

### Callback Update (Functional Update)
```typescript
const [count, setCount] = useState(0);

// Use previous state in update
setCount(prev => prev + 1);

// Useful when new state depends on previous
const handleMultipleClicks = () => {
  setCount(prev => prev + 1);
  setCount(prev => prev + 1);
  setCount(prev => prev + 1);
  // Will increase by 3 (not batch)
};
```

### Conditional Update
```typescript
const handleToggle = () => {
  setIsActive(prev => !prev);
};
```

## Common Patterns

### Form Input
```typescript
function TextInput() {
  const [value, setValue] = useState('');

  return (
    <input
      value={value}
      onChange={(e) => setValue(e.target.value)}
      placeholder="Type here..."
    />
  );
}
```

### Toggle
```typescript
function Toggle() {
  const [isOn, setIsOn] = useState(false);

  return (
    <button onClick={() => setIsOn(!isOn)}>
      {isOn ? 'On' : 'Off'}
    </button>
  );
}
```

### List Management
```typescript
function TodoList() {
  const [todos, setTodos] = useState<string[]>([]);

  const addTodo = (text: string) => {
    setTodos(prev => [...prev, text]);
  };

  const removeTodo = (index: number) => {
    setTodos(prev => prev.filter((_, i) => i !== index));
  };

  return (
    <div>
      <button onClick={() => addTodo('New todo')}>Add</button>
      <ul>
        {todos.map((todo, i) => (
          <li key={i}>
            {todo}
            <button onClick={() => removeTodo(i)}>Remove</button>
          </li>
        ))}
      </ul>
    </div>
  );
}
```

## State Update Behavior

### Batching
React batches state updates in event handlers:

```typescript
function Component() {
  const [count, setCount] = useState(0);

  const handleClick = () => {
    setCount(1);
    setCount(2);
    setCount(3);
    // Only 1 re-render, final count = 3
  };
}
```

### Timing
State updates are asynchronous:

```typescript
function Component() {
  const [count, setCount] = useState(0);

  const handleClick = () => {
    setCount(1);
    console.log(count);  // Still 0! Update hasn't happened yet
  };
}
```

### Using New State
If you need the new state value:

```typescript
function Component() {
  const [count, setCount] = useState(0);

  const handleClick = () => {
    const newCount = 1;
    setCount(newCount);
    console.log(newCount);  // Use calculated value directly
  };
}
```

## useState Best Practices

✅ **DO:**
- Use separate state variables for unrelated data
- Use objects only when values are always updated together
- Use functional updates when new state depends on old
- Initialize with correct type
- Use TypeScript for type safety

❌ **DON'T:**
- Don't mutate state directly
- Don't put all state in one object
- Don't rely on state being updated synchronously
- Don't initialize state from props (use useEffect)
- Don't forget dependency arrays in useEffect

## Comparison: useState vs useReducer

| Use Case | useState | useReducer |
| :-- | :-- | :-- |
| **Simple state** | ✅ Better | ❌ Overkill |
| **Multiple state** | ✅ Fine | ✅ Better |
| **Complex logic** | ❌ Hard | ✅ Better |
| **Updating from prev** | 🟡 Callback | ✅ Natural |
