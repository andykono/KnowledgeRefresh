# useEffect Hook

[[2. React Hooks.md]]

---

useEffect lets you perform side effects in functional components (API calls, subscriptions, timers, etc).

## Basic Usage

```typescript
import { useEffect, useState } from 'react';

function Component() {
  const [data, setData] = useState(null);

  // Run effect after render
  useEffect(() => {
    fetch('/api/data')
      .then(res => res.json())
      .then(setData);
  }, []);  // Empty dependency = run once on mount

  return <div>{data}</div>;
}
```

**Anatomy:**
```typescript
useEffect(() => {
  // Effect code here
  console.log('After render');

  return () => {
    // Cleanup (optional)
    console.log('Before unmount or re-run');
  };
}, [dependencies]);
// Dependency array: when to re-run
```

---

## Dependency Array Patterns

### Run Once (on mount)
```typescript
useEffect(() => {
  console.log('Component mounted');
  return () => console.log('Component unmounted');
}, []);  // Empty array = run once
```

### Run on Every Render
```typescript
useEffect(() => {
  console.log('After every render');
});  // No array = run after every render (usually bad!)
```

### Run When Dependencies Change
```typescript
function UserProfile({ userId }: { userId: string }) {
  const [user, setUser] = useState(null);

  useEffect(() => {
    if (!userId) return;
    fetchUser(userId).then(setUser);
  }, [userId]);  // Re-run when userId changes

  return <div>{user?.name}</div>;
}
```

### Multiple Dependencies
```typescript
useEffect(() => {
  fetchUserPosts(userId, postType);
}, [userId, postType]);  // Re-run if either changes
```

## Cleanup Function

Return a cleanup function to run before unmount or effect re-run:

```typescript
function ChatComponent() {
  useEffect(() => {
    const socket = connectToChat();

    // Cleanup: disconnect when component unmounts
    return () => {
      socket.disconnect();
    };
  }, []);
}

function UserListener({ userId }) {
  useEffect(() => {
    const unsubscribe = subscribeToUser(userId, (user) => {
      console.log('User updated:', user);
    });

    // Cleanup: unsubscribe when userId changes or unmounts
    return () => {
      unsubscribe();
    };
  }, [userId]);
}
```

## Common Patterns

### Fetch Data
```typescript
function UserList() {
  const [users, setUsers] = useState([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  useEffect(() => {
    const fetchData = async () => {
      try {
        const response = await fetch('/api/users');
        const data = await response.json();
        setUsers(data);
      } catch (err) {
        setError(err.message);
      } finally {
        setLoading(false);
      }
    };

    fetchData();
  }, []);

  if (loading) return <div>Loading...</div>;
  if (error) return <div>Error: {error}</div>;
  return <ul>{users.map(u => <li key={u.id}>{u.name}</li>)}</ul>;
}
```

### Event Listener
```typescript
function WindowSize() {
  const [size, setSize] = useState(window.innerWidth);

  useEffect(() => {
    const handleResize = () => setSize(window.innerWidth);

    window.addEventListener('resize', handleResize);

    // Cleanup: remove listener
    return () => {
      window.removeEventListener('resize', handleResize);
    };
  }, []);

  return <div>Width: {size}px</div>;
}
```

### Timer/Interval
```typescript
function Timer() {
  const [seconds, setSeconds] = useState(0);

  useEffect(() => {
    const interval = setInterval(() => {
      setSeconds(s => s + 1);
    }, 1000);

    // Cleanup: clear interval
    return () => clearInterval(interval);
  }, []);

  return <div>Seconds: {seconds}</div>;
}
```

### Document Title
```typescript
function DocumentTitle({ title }: { title: string }) {
  useEffect(() => {
    document.title = title;
  }, [title]);

  return <div>{title}</div>;
}
```

## Dependency Array Guide

**Don't include objects created in render:**
```typescript
// ❌ BAD: options created every render
const options = { timeout: 1000 };
useEffect(() => {
  fetchData(options);
}, [options]);  // Will re-run every time!

// ✅ GOOD: Move outside or use useCallback
const options = { timeout: 1000 };
useEffect(() => {
  fetchData(options);
}, []);

// ✅ ALSO GOOD: If needs to be in dependency
useEffect(() => {
  fetchData(options);
}, [options.timeout]);  // Or just the needed value
```

**Include all used values:**
```typescript
function Component({ userId, userName }) {
  useEffect(() => {
    console.log(userId, userName);
  }, [userId, userName]);  // Both must be included
}
```

**Use ESLint plugin:**
```json
{
  "rules": {
    "react-hooks/exhaustive-deps": "warn"
  }
}
```

## Common Mistakes

### Missing Dependency
```typescript
// ❌ WRONG: Missing userId dependency
function Component({ userId }: { userId: string }) {
  useEffect(() => {
    fetchUser(userId);
  }, []);  // userId not included!
  // Will fetch with old userId on first render forever
}

// ✅ CORRECT
function Component({ userId }: { userId: string }) {
  useEffect(() => {
    fetchUser(userId);
  }, [userId]);  // Include userId
}
```

### Not Cleaning Up
```typescript
// ❌ WRONG: Memory leak
useEffect(() => {
  const interval = setInterval(() => {
    // ...
  }, 1000);
  // Forgot to clear!
}, []);

// ✅ CORRECT
useEffect(() => {
  const interval = setInterval(() => {
    // ...
  }, 1000);
  return () => clearInterval(interval);  // Cleanup
}, []);
```

### Infinite Loop
```typescript
// ❌ WRONG: Infinite loop
useEffect(() => {
  setData(newData);  // Causes re-render
}, [data]);  // data changes, effect runs, data changes, effect runs...

// ✅ CORRECT: Only set on mount
useEffect(() => {
  setData(newData);
}, []);  // Run only once
```

## Multiple Effects

You can use multiple useEffect hooks:

```typescript
function Component({ userId }) {
  // Fetch user
  useEffect(() => {
    fetchUser(userId);
  }, [userId]);

  // Track page view
  useEffect(() => {
    trackPageView();
  }, []);

  // Listen to window resize
  useEffect(() => {
    const handleResize = () => { /* ... */ };
    window.addEventListener('resize', handleResize);
    return () => window.removeEventListener('resize', handleResize);
  }, []);
}
```

Each effect:
- ✅ Has single responsibility
- ✅ Has its own cleanup
- ✅ Has its own dependencies

## useEffect vs useLayoutEffect

| Hook | When | Use Case |
| :-- | :-- | :-- |
| **useEffect** | After render (async) | Most cases |
| **useLayoutEffect** | Before browser paint | Measure DOM, animations |

```typescript
// useEffect: Typical
useEffect(() => {
  // Runs after DOM is painted
  analyticsTrack('page-viewed');
}, []);

// useLayoutEffect: Rare (avoid unless needed)
useLayoutEffect(() => {
  // Runs before browser paints
  // Use for: measuring DOM, adjusting layout
  const height = ref.current.offsetHeight;
  setHeight(height);
}, []);
```

See: [[2.11 React v19 Hooks/2.11.7 useLayoutEffect & useInsertionEffect.md|useLayoutEffect & useInsertionEffect]]

## useEffect Best Practices

✅ **DO:**
- Include all dependencies in array
- Return cleanup function for subscriptions
- Use multiple useEffect for different concerns
- Fetch data in effects, not render
- Use ESLint plugin for dependency checking

❌ **DON'T:**
- Don't forget cleanup functions
- Don't skip dependencies
- Don't use in render (side effects in render)
- Don't update props inside effect
- Don't create infinite loops
