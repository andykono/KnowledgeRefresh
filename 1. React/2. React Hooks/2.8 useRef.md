# useRef Hook

[[2. React Hooks.md]]

---

useRef gives you direct access to DOM elements without causing re-renders.

## Basic Usage

```typescript
function TextInput() {
  const inputRef = useRef<HTMLInputElement>(null);

  const focusInput = () => {
    inputRef.current?.focus();
  };

  return (
    <>
      <input ref={inputRef} />
      <button onClick={focusInput}>Focus input</button>
    </>
  );
}
```

---

## Store Mutable Value

```typescript
function Component() {
  const countRef = useRef(0);

  const handleClick = () => {
    countRef.current++;
    console.log(`Clicked ${countRef.current} times`);
    // No re-render!
  };

  return <button onClick={handleClick}>Click me</button>;
}
```

---

## vs State

| Use | Purpose |
| :-- | :-- |
| **State** | Triggers re-render |
| **Ref** | Doesn't trigger re-render |

```typescript
// Use state for UI updates
const [count, setCount] = useState(0);

// Use ref for values that don't affect UI
const timerRef = useRef<number>();
```

---

## Common Use Cases

```typescript
// Focus input
const inputRef = useRef<HTMLInputElement>(null);
inputRef.current?.focus();

// Measure element
const divRef = useRef<HTMLDivElement>(null);
const height = divRef.current?.offsetHeight;

// Store timer ID
const timerRef = useRef<number>();
useEffect(() => {
  timerRef.current = setInterval(() => {}, 1000);
  return () => clearInterval(timerRef.current);
}, []);
```
