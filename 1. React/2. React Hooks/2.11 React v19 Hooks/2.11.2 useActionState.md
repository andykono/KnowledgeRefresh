# useActionState Hook

[[2.11 React v19 Hooks.md]]

---

useActionState simplifies form submissions with automatic pending/error state management.

## Basic Usage

```typescript
'use client'

import { useActionState } from 'react';

// Server Action
async function loginUser(formData) {
  'use server'
  const email = formData.get('email');
  const password = formData.get('password');

  const user = await db.users.findOne({ email });
  if (!user || !verifyPassword(password, user.password)) {
    return { error: 'Invalid credentials' };
  }

  return { success: true, user };
}

// Client Component
function LoginForm() {
  const [state, formAction] = useActionState(loginUser, null);

  return (
    <form action={formAction}>
      <input name="email" type="email" required />
      <input name="password" type="password" required />
      <button type="submit">Login</button>

      {state?.error && <p className="error">{state.error}</p>}
      {state?.success && <p className="success">Logged in!</p>}
    </form>
  );
}
```

**Anatomy:**
```typescript
const [state, formAction] = useActionState(action, initialState);
// state: Return value from action
// formAction: Pass to form action prop
// action: Server action or async function
// initialState: Initial state value
```

## vs useState

### Old Way (v18)
```typescript
function LoginForm() {
  const [email, setEmail] = useState('');
  const [password, setPassword] = useState('');
  const [error, setError] = useState(null);
  const [isPending, setIsPending] = useState(false);

  const handleSubmit = async (e) => {
    e.preventDefault();
    setError(null);
    setIsPending(true);

    try {
      const result = await loginUser(email, password);
      if (!result.success) {
        setError(result.error);
        return;
      }
      // Redirect or navigate
    } catch (err) {
      setError(err.message);
    } finally {
      setIsPending(false);
    }
  };

  return (
    <form onSubmit={handleSubmit}>
      <input value={email} onChange={(e) => setEmail(e.target.value)} />
      <input value={password} onChange={(e) => setPassword(e.target.value)} />
      <button disabled={isPending}>{isPending ? 'Logging in...' : 'Login'}</button>
      {error && <p>{error}</p>}
    </form>
  );
}
```

### New Way (v19)
```typescript
function LoginForm() {
  const [state, formAction] = useActionState(loginUser, null);

  return (
    <form action={formAction}>
      <input name="email" type="email" />
      <input name="password" type="password" />
      <button type="submit">Login</button>
      {state?.error && <p>{state.error}</p>}
    </form>
  );
}
```

Much simpler! React handles form data collection automatically.

## FormData & Server Actions

```typescript
'use client'

// Server action receives FormData
async function updateProfile(formData) {
  'use server'

  const name = formData.get('name');
  const email = formData.get('email');
  const avatar = formData.get('avatar');  // File!

  // Process data
  await db.users.update(userId, { name, email });

  if (avatar) {
    await uploadFile(avatar, `avatars/${userId}`);
  }

  return { success: true, message: 'Profile updated' };
}

// Client component
export function ProfileForm({ user }) {
  const [state, formAction] = useActionState(updateProfile, null);

  return (
    <form action={formAction}>
      <input name="name" defaultValue={user.name} />
      <input name="email" defaultValue={user.email} />
      <input name="avatar" type="file" />
      <button type="submit">Update</button>
      {state?.message && <p>{state.message}</p>}
    </form>
  );
}
```

## Showing Pending State

Combining with `useFormStatus`:

```typescript
'use client'

import { useFormStatus } from 'react-dom';

// Submit button with pending feedback
function SubmitButton() {
  const { pending } = useFormStatus();

  return (
    <button type="submit" disabled={pending}>
      {pending ? 'Submitting...' : 'Submit'}
    </button>
  );
}

function SignupForm() {
  const [state, formAction] = useActionState(signup, null);

  return (
    <form action={formAction}>
      <input name="email" type="email" required />
      <input name="password" type="password" required />
      <SubmitButton />  {/* Shows pending state */}
      {state?.error && <p className="error">{state.error}</p>}
    </form>
  );
}
```

## Advanced Patterns

### Multiple Validation Steps
```typescript
async function createUser(formData) {
  'use server'

  const email = formData.get('email');
  const password = formData.get('password');

  // Validation
  if (!isValidEmail(email)) {
    return { error: 'Invalid email' };
  }

  if (password.length < 8) {
    return { error: 'Password too short' };
  }

  const existing = await db.users.findOne({ email });
  if (existing) {
    return { error: 'Email already registered' };
  }

  // Create
  const user = await db.users.create({
    email,
    password: await hashPassword(password),
  });

  return { success: true, user };
}
```

### With Redirects
```typescript
import { redirect } from 'next/navigation';

async function loginUser(formData) {
  'use server'

  const email = formData.get('email');
  const password = formData.get('password');

  const user = await authenticateUser(email, password);
  if (!user) {
    return { error: 'Invalid credentials' };
  }

  // Redirect after successful login
  redirect('/dashboard');
}
```

### Revalidating Data
```typescript
import { revalidatePath } from 'next/cache';

async function updatePost(formData) {
  'use server'

  const id = formData.get('id');
  const title = formData.get('title');
  const content = formData.get('content');

  await db.posts.update(id, { title, content });

  // Revalidate cache
  revalidatePath('/posts/[id]', 'page');

  return { success: true };
}
```

## Error Handling

```typescript
async function submitForm(formData) {
  'use server'

  try {
    const data = validateFormData(formData);
    await saveToDatabase(data);
    return { success: true };
  } catch (error) {
    if (error instanceof ValidationError) {
      return { error: error.message };
    }
    // Log error, show generic message
    console.error('Form submission error:', error);
    return { error: 'Something went wrong' };
  }
}
```

## useActionState Best Practices

✅ **DO:**
- Use for form submissions
- Return state from action
- Use FormData for complex data
- Show error states
- Use with useFormStatus for pending feedback
- Validate on both client and server

❌ **DON'T:**
- Don't use for onClick handlers (use form submission)
- Don't forget type="submit" on buttons
- Don't skip client-side validation
- Don't expose sensitive errors to users
- Don't modify formData directly
