# useOptimistic Hook

[[2.11 React v19 Hooks.md]]

---

useOptimistic shows updated UI while the server request is still pending, then reverts if it fails.

## Basic Usage

```typescript
'use client'

import { useOptimistic } from 'react';

function TodoList({ initialTodos }) {
  const [optimisticTodos, addOptimistic] = useOptimistic(
    initialTodos,
    // Update function: receives state + new value
    (state, newTodo) => [...state, newTodo]
  );

  const handleAddTodo = async (formData) => {
    const text = formData.get('text');

    // Show optimistic update immediately
    addOptimistic({ id: '...', text, done: false });

    // Send to server
    const result = await addTodoToServer(text);

    // Server response replaces optimistic state
    // If fails, automatic rollback
  };

  return (
    <form action={handleAddTodo}>
      <input name="text" />
      <button type="submit">Add</button>

      <ul>
        {optimisticTodos.map(todo => (
          <li key={todo.id}>{todo.text}</li>
        ))}
      </ul>
    </form>
  );
}
```

**Anatomy:**
```typescript
const [optimisticState, addOptimisticUpdate] = useOptimistic(
  currentState,
  updateFn: (state, value) => newState
);
```

---

## How It Works

```
1. User submits form
   ↓
2. addOptimistic() called with new data
   → UI updates IMMEDIATELY (optimistic)
   ↓
3. Server request sent
   ↓
4a. Server succeeds
    → Optimistic state replaced with real data
4b. Server fails
    → Automatic rollback to original state
```

## Common Patterns

### Add Item
```typescript
'use client'

import { useOptimistic } from 'react';

async function createItem(formData) {
  'use server'
  const name = formData.get('name');
  const item = await db.items.create({ name });
  return item;
}

export function ItemForm({ initialItems }) {
  const [optimisticItems, addOptimistic] = useOptimistic(
    initialItems,
    (state, newItem) => [...state, newItem]
  );

  const handleSubmit = async (formData) => {
    // Optimistic: show new item immediately
    const newItem = {
      id: crypto.randomUUID(),
      name: formData.get('name'),
    };
    addOptimistic(newItem);

    // Then actually save
    await createItem(formData);
    // Form action completes
  };

  return (
    <form action={handleSubmit}>
      <input name="name" />
      <button type="submit">Add</button>

      <ul>
        {optimisticItems.map(item => (
          <li key={item.id}>{item.name}</li>
        ))}
      </ul>
    </form>
  );
}
```

### Update Item
```typescript
'use client'

import { useOptimistic } from 'react';

async function updateItem(formData) {
  'use server'
  const id = formData.get('id');
  const done = formData.get('done') === 'on';
  await db.items.update(id, { done });
}

export function TodoItem({ id, text, initialDone }) {
  const [optimisticDone, setOptimisticDone] = useOptimistic(
    initialDone,
    (_, newDone) => newDone  // Simple toggle
  );

  const handleToggle = async () => {
    // Optimistic: toggle immediately
    setOptimisticDone(!optimisticDone);

    // Then save to server
    const formData = new FormData();
    formData.set('id', id);
    formData.set('done', !optimisticDone);
    await updateItem(formData);
  };

  return (
    <li style={{ textDecoration: optimisticDone ? 'line-through' : 'none' }}>
      {text}
      <input
        type="checkbox"
        checked={optimisticDone}
        onChange={handleToggle}
      />
    </li>
  );
}
```

### Delete Item
```typescript
'use client'

import { useOptimistic } from 'react';

async function deleteItem(id) {
  'use server'
  await db.items.delete(id);
}

export function ItemList({ initialItems }) {
  const [optimisticItems, removeOptimistic] = useOptimistic(
    initialItems,
    (state, id) => state.filter(item => item.id !== id)
  );

  const handleDelete = async (id) => {
    // Optimistic: remove from list immediately
    removeOptimistic(id);

    try {
      // Then delete from server
      await deleteItem(id);
    } catch (error) {
      // Error handling - state automatically reverts
      console.error('Failed to delete:', error);
    }
  };

  return (
    <ul>
      {optimisticItems.map(item => (
        <li key={item.id}>
          {item.name}
          <button onClick={() => handleDelete(item.id)}>
            Delete
          </button>
        </li>
      ))}
    </ul>
  );
}
```

### Complex Updates
```typescript
'use client'

import { useOptimistic } from 'react';

async function updateTodo(formData) {
  'use server'
  const id = formData.get('id');
  const updates = {
    text: formData.get('text'),
    priority: formData.get('priority'),
  };
  return db.todos.update(id, updates);
}

export function TodoEditor({ todo }) {
  const [optimisticTodo, updateOptimistic] = useOptimistic(
    todo,
    (state, formData) => ({
      ...state,
      text: formData.get('text'),
      priority: formData.get('priority'),
    })
  );

  const handleChange = async (e) => {
    const formData = new FormData(e.currentTarget);

    // Optimistic: update UI
    updateOptimistic(formData);

    // Then save
    await updateTodo(formData);
  };

  return (
    <form onChange={handleChange}>
      <input
        name="text"
        value={optimisticTodo.text}
        onChange={() => {}}  // Controlled by form
      />
      <select
        name="priority"
        value={optimisticTodo.priority}
        onChange={() => {}}
      >
        <option>Low</option>
        <option>Medium</option>
        <option>High</option>
      </select>
    </form>
  );
}
```

## Error Handling

Automatic rollback on error:

```typescript
'use client'

import { useOptimistic } from 'react';

export function TodoList({ initialTodos }) {
  const [optimisticTodos, addOptimistic] = useOptimistic(
    initialTodos,
    (state, newTodo) => [...state, newTodo]
  );
  const [error, setError] = useState(null);

  const handleAdd = async (formData) => {
    setError(null);

    const newTodo = {
      id: '...',
      text: formData.get('text'),
    };

    // Show optimistically
    addOptimistic(newTodo);

    try {
      // Save to server
      await addTodoToServer(formData.get('text'));
    } catch (err) {
      // Automatic rollback!
      // State reverts to initialTodos
      setError(err.message);
    }
  };

  return (
    <>
      <form action={handleAdd}>
        <input name="text" />
        <button>Add</button>
      </form>

      {error && <p className="error">{error}</p>}

      <ul>
        {optimisticTodos.map(todo => (
          <li key={todo.id}>{todo.text}</li>
        ))}
      </ul>
    </>
  );
}
```

## With useActionState

```typescript
'use client'

import { useActionState } from 'react';
import { useOptimistic } from 'react';

async function addTodo(formData) {
  'use server'
  const text = formData.get('text');
  return db.todos.create({ text });
}

export function TodoForm({ initialTodos }) {
  const [optimisticTodos, addOptimistic] = useOptimistic(
    initialTodos,
    (state, newTodo) => [...state, newTodo]
  );

  const [state, formAction] = useActionState(addTodo, null);

  const handleFormAction = (formData) => {
    const text = formData.get('text');
    addOptimistic({ id: Date.now(), text });
    formAction(formData);
  };

  return (
    <form action={handleFormAction}>
      <input name="text" required />
      <button type="submit">Add</button>

      {state?.error && <p className="error">{state.error}</p>}

      <ul>
        {optimisticTodos.map(todo => (
          <li key={todo.id}>{todo.text}</li>
        ))}
      </ul>
    </form>
  );
}
```

## useOptimistic Best Practices

✅ **DO:**
- Use for instant UI feedback
- Combine with try/catch for error handling
- Show loading states while pending
- Use with form actions
- Provide clear undo/rollback messaging

❌ **DON'T:**
- Don't use for data that doesn't need instant feedback
- Don't ignore errors (handle gracefully)
- Don't forget to handle rollback UI
- Don't use for security-sensitive operations
- Don't overcomplicate with multiple optimistic updates

## Comparison: With & Without Optimistic

### Without Optimistic (Slow UX)
```
User clicks delete
  ↓
Network request starts
  ↓ (Wait for server)
  ↓
Item removed from list
  → User sees delay
```

### With Optimistic (Fast UX)
```
User clicks delete
  ↓
Item removed from list IMMEDIATELY
  ↓ (Network request in background)
  ↓
Server confirms or rolls back
  → Instant feedback!
```
