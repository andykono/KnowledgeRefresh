# useFormStatus Hook

[[2.11 React v19 Hooks.md]]

---

useFormStatus lets you show feedback while a form is being submitted.

## Basic Usage

```typescript
'use client'

import { useFormStatus } from 'react-dom';

function SubmitButton() {
  const { pending } = useFormStatus();

  return (
    <button type="submit" disabled={pending}>
      {pending ? 'Saving...' : 'Save'}
    </button>
  );
}

export function Form() {
  const [state, formAction] = useActionState(submitAction, null);

  return (
    <form action={formAction}>
      <input name="name" required />
      <input name="email" required />
      <SubmitButton />  {/* Component shows pending state */}
    </form>
  );
}
```

**Key Point:** useFormStatus watches the **nearest parent form's** action.

---

## Return Value

```typescript
const { pending, data, method, action } = useFormStatus();

// pending: boolean - Form is submitting
// data: FormData - Current form data (null if not submitting)
// method: string - Form method (usually 'POST')
// action: function - The form's action function
```

## Common Patterns

### Show Loading Spinner
```typescript
'use client'

import { useFormStatus } from 'react-dom';

function SubmitButton() {
  const { pending } = useFormStatus();

  return (
    <button type="submit" disabled={pending} className="submit-btn">
      {pending ? (
        <>
          <Spinner />
          Saving...
        </>
      ) : (
        'Save'
      )}
    </button>
  );
}
```

### Disable Multiple Fields
```typescript
'use client'

import { useFormStatus } from 'react-dom';

function Form() {
  const { pending } = useFormStatus();

  return (
    <form action={submitAction}>
      <input name="email" disabled={pending} required />
      <input name="password" disabled={pending} required />
      <select disabled={pending}>
        <option>Select role</option>
        <option>Admin</option>
        <option>User</option>
      </select>
      <button type="submit" disabled={pending}>
        {pending ? 'Submitting...' : 'Submit'}
      </button>
    </form>
  );
}
```

### Show Progress
```typescript
'use client'

import { useFormStatus } from 'react-dom';

function ProgressForm() {
  const { pending, data } = useFormStatus();

  return (
    <form action={uploadAction}>
      <input type="file" name="file" disabled={pending} />
      {pending && <ProgressBar />}
      <button type="submit" disabled={pending}>
        {pending ? 'Uploading...' : 'Upload'}
      </button>
    </form>
  );
}
```

### Access Form Data
```typescript
'use client'

import { useFormStatus } from 'react-dom';

function FormFeedback() {
  const { pending, data } = useFormStatus();

  if (!pending) return null;

  // Show which field is being saved
  const email = data?.get('email');

  return (
    <p className="feedback">
      Saving email: {email}...
    </p>
  );
}

export function Form() {
  const [state, formAction] = useActionState(submitAction, null);

  return (
    <form action={formAction}>
      <input name="email" required />
      <button type="submit">Save</button>
      <FormFeedback />
    </form>
  );
}
```

## With useActionState

Perfect combination for complete form handling:

```typescript
'use client'

import { useActionState } from 'react';
import { useFormStatus } from 'react-dom';

// Server action
async function saveProfile(formData) {
  'use server'
  const name = formData.get('name');
  const email = formData.get('email');

  try {
    await db.users.update(userId, { name, email });
    return { success: true, message: 'Saved!' };
  } catch (error) {
    return { error: error.message };
  }
}

// Submit button component
function SaveButton() {
  const { pending } = useFormStatus();
  return (
    <button type="submit" disabled={pending}>
      {pending ? 'Saving...' : 'Save Profile'}
    </button>
  );
}

// Form component
export function ProfileForm({ user }) {
  const [state, formAction] = useActionState(saveProfile, null);

  return (
    <form action={formAction}>
      <input name="name" defaultValue={user.name} />
      <input name="email" defaultValue={user.email} />
      <SaveButton />

      {state?.error && <p className="error">{state.error}</p>}
      {state?.success && <p className="success">{state.message}</p>}
    </form>
  );
}
```

## Advanced Patterns

### Optimistic UI with useFormStatus
```typescript
'use client'

import { useOptimistic } from 'react';
import { useFormStatus } from 'react-dom';

async function addTodo(formData) {
  'use server'
  const text = formData.get('text');
  const todo = await db.todos.create({ text });
  return todo;
}

function AddTodoForm({ todos }) {
  const [optimisticTodos, addOptimistic] = useOptimistic(
    todos,
    (state, newTodo) => [...state, newTodo]
  );
  const { pending } = useFormStatus();

  const handleSubmit = async (formData) => {
    const text = formData.get('text');
    addOptimistic({ id: '...', text, done: false });
    // Form action submits automatically
  };

  return (
    <form action={addTodo} onSubmit={handleSubmit}>
      <input name="text" disabled={pending} required />
      <button type="submit" disabled={pending}>
        {pending ? 'Adding...' : 'Add Todo'}
      </button>
      <ul>
        {optimisticTodos.map(todo => (
          <li key={todo.id}>{todo.text}</li>
        ))}
      </ul>
    </form>
  );
}
```

See: [[2.11.4 useOptimistic.md|useOptimistic]]

### Multi-step Forms
```typescript
'use client'

import { useFormStatus } from 'react-dom';

function StepForm({ step }) {
  const { pending } = useFormStatus();

  return (
    <form action={step === 1 ? step1Action : step2Action}>
      {step === 1 && (
        <>
          <input name="email" type="email" disabled={pending} />
          <button name="step" value="next" disabled={pending}>
            Next
          </button>
        </>
      )}
      {step === 2 && (
        <>
          <input name="password" type="password" disabled={pending} />
          <button type="submit" disabled={pending}>
            {pending ? 'Creating...' : 'Create Account'}
          </button>
        </>
      )}
    </form>
  );
}
```

## useFormStatus Location

**Important:** useFormStatus works in **child components** of the form:

```typescript
// ✅ CORRECT: Button inside form
<form action={formAction}>
  <input name="name" />
  <SubmitButton />  {/* Can use useFormStatus here */}
</form>

function SubmitButton() {
  const { pending } = useFormStatus();
  return <button disabled={pending}>Save</button>;
}

// ❌ WRONG: useFormStatus outside form
function OutsideForm() {
  const { pending } = useFormStatus();  // Returns null
  return <button disabled={pending}>Save</button>;
}
```

## useFormStatus Best Practices

✅ **DO:**
- Use for form submission feedback
- Disable inputs while submitting
- Show clear loading messages
- Extract into separate button component
- Combine with useActionState
- Use for file uploads

❌ **DON'T:**
- Don't use outside a form context
- Don't use in parent form component (use in children)
- Don't overuse animations (keep UX clear)
- Don't hide error states
- Don't use for non-form actions
