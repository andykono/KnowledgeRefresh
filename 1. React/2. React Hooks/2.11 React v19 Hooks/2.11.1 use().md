# use() Hook

[[2.11 React v19 Hooks.md]]

---

The `use()` hook lets you unwrap promises and context values, enabling conditional usage of hooks.

## Basic Usage

### Unwrapping Promises
```typescript
'use client'

import { use } from 'react';

function Component() {
  // Unwrap promise
  const data = use(fetchData());

  return <div>{data.name}</div>;
}
```

### Unwrapping Context
```typescript
'use client'

import { use } from 'react';

const ThemeContext = createContext();

function Component() {
  // use() works like useContext
  const theme = use(ThemeContext);

  return <div style={{ background: theme.bg }}>Content</div>;
}
```

## Promise Unwrapping

### Conditional Promise Handling
```typescript
'use client'

import { use } from 'react';

function UserProfile({ userId }) {
  // Can now conditionally unwrap promises!
  const promise = userId ? fetchUser(userId) : null;
  const user = promise ? use(promise) : null;

  return (
    <div>
      {user ? <h1>{user.name}</h1> : <p>No user selected</p>}
    </div>
  );
}

// Before use(), you couldn't do this:
// function UserProfile({ userId }) {
//   // This would error - conditional hooks!
//   if (userId) {
//     const user = await fetchUser(userId);  // ❌ Can't await in render
//   }
// }
```

### vs useEffect
```typescript
// OLD: useEffect approach
function Component() {
  const [data, setData] = useState(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  useEffect(() => {
    fetchData()
      .then(setData)
      .catch(setError)
      .finally(() => setLoading(false));
  }, []);

  if (loading) return <div>Loading...</div>;
  if (error) return <div>Error: {error}</div>;
  return <div>{data}</div>;
}

// NEW: use() approach
function Component() {
  const data = use(fetchData());
  return <div>{data}</div>;
}

// Suspense handles loading automatically!
<Suspense fallback={<div>Loading...</div>}>
  <Component />
</Suspense>
```

## Common Patterns

### Fetch in Conditional
```typescript
'use client'

import { use } from 'react';
import { Suspense } from 'react';

function SearchResults({ query }) {
  // Only fetch if query exists
  const promise = query ? searchAPI(query) : null;
  const results = promise ? use(promise) : [];

  return (
    <ul>
      {results.map(result => (
        <li key={result.id}>{result.name}</li>
      ))}
    </ul>
  );
}

// Usage
<Suspense fallback={<div>Loading results...</div>}>
  <SearchResults query={searchTerm} />
</Suspense>
```

### Unwrap Multiple Promises
```typescript
'use client'

import { use } from 'react';

function Dashboard({ userId }) {
  const userPromise = fetchUser(userId);
  const statsPromise = fetchStats(userId);

  const user = use(userPromise);
  const stats = use(statsPromise);

  return (
    <div>
      <h1>{user.name}</h1>
      <p>Score: {stats.score}</p>
    </div>
  );
}

// Suspense waits for both promises
<Suspense fallback={<div>Loading dashboard...</div>}>
  <Dashboard userId="123" />
</Suspense>
```

### Context with use()
```typescript
'use client'

import { use } from 'react';

const AuthContext = createContext();

function LoginForm() {
  // use() works with context too
  const auth = use(AuthContext);

  return (
    <div>
      {auth.user ? (
        <p>Welcome, {auth.user.name}</p>
      ) : (
        <button onClick={auth.login}>Login</button>
      )}
    </div>
  );
}
```

## Error Handling

Wrap with Error Boundary:

```typescript
'use client'

import { use } from 'react';

function Component({ id }) {
  const data = use(fetchData(id));

  return <div>{data.name}</div>;
}

// Error Boundary catches promise rejection
<ErrorBoundary fallback={<div>Error loading data</div>}>
  <Suspense fallback={<div>Loading...</div>}>
    <Component id="123" />
  </Suspense>
</ErrorBoundary>
```

## use() vs useContext

```typescript
// useContext: Always works
function Component() {
  const theme = useContext(ThemeContext);
  return <div>{theme.name}</div>;
}

// use(): More flexible
function Component({ themeOrPromise }) {
  // Works with both context and promises
  const theme = use(themeOrPromise instanceof Promise ? themeOrPromise : ThemeContext);
  return <div>{theme.name}</div>;
}
```

## With Server Components

// Server Component
async function ServerComponent({ userId }) {

'use client'

function ClientComponent({ userPromise }) {

// Compose them
<ServerComponent userId="123" />  // Returns promise

## Best Practices

✅ **DO:**
- Use with Suspense for loading states
- Use for conditional promise unwrapping
- Use with Error Boundary for error handling
- Keep promises stable (memoize if needed)

❌ **DON'T:**
- Don't use inside loops or conditionals (it's just unwrapping)
- Don't forget Suspense boundary
- Don't forget Error Boundary
- Don't create promises in render (move to effect or server)

## Common Issues

### Promise Not Updating
```typescript
// ❌ WRONG: Promise created in render, changes every time
function Component({ id }) {
  const data = use(fetchData(id));  // New promise each render!
  return <div>{data}</div>;
}

// ✅ CORRECT: Promise is stable
const userPromise = fetchUser(userId);
function Component() {
  const user = use(userPromise);
  return <div>{user.name}</div>;
}

// ✅ OR: Revalidate on change
function Component({ id }) {
  const promise = useMemo(() => fetchData(id), [id]);
  const data = use(promise);
  return <div>{data}</div>;
}
```

### Missing Suspense
```typescript
// ❌ WRONG: No Suspense = crash
function Component() {
  const data = use(fetchData());
  return <div>{data}</div>;
}

// ✅ CORRECT: Wrap with Suspense
<Suspense fallback={<div>Loading...</div>}>
  <Component />
</Suspense>
```
