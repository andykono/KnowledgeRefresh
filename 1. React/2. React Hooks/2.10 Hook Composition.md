# Hook Composition

[[2. React Hooks.md]]

Combine multiple hooks to create powerful custom hooks.

## Basic Composition

```typescript
// Use multiple hooks in one custom hook
function useUser(userId: string) {
  const [user, setUser] = useState(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  useEffect(() => {
    if (!userId) return;

    (async () => {
      try {
        const data = await fetchUser(userId);
        setUser(data);
      } catch (err) {
        setError(err);
      } finally {
        setLoading(false);
      }
    })();
  }, [userId]);

  return { user, loading, error };
}

// Usage
function UserProfile({ userId }) {
  const { user, loading, error } = useUser(userId);

  if (loading) return <div>Loading...</div>;
  if (error) return <div>Error: {error}</div>;
  return <div>{user.name}</div>;
}
```

## Advanced Composition

```typescript
// Compose multiple custom hooks
function useSearchUsers(query: string) {
  const debouncedQuery = useDebounce(query, 300);
  const { data: results, loading, error } = useAsync(
    () => searchUsers(debouncedQuery),
    [debouncedQuery]
  );

  return { results, loading, error };
}

// Usage
function SearchForm() {
  const [query, setQuery] = useState('');
  const { results, loading } = useSearchUsers(query);

  return (
    <>
      <input value={query} onChange={(e) => setQuery(e.target.value)} />
      {loading && <div>Searching...</div>}
      <ul>
        {results?.map(user => (
          <li key={user.id}>{user.name}</li>
        ))}
      </ul>
    </>
  );
}
```

## Composition Pattern: useAsync + useFetch

```typescript
function useFetch<T>(url: string) {
  return useAsync(() => fetch(url).then(r => r.json()) as Promise<T>, [url]);
}

function usePagedFetch<T>(baseUrl: string, page: number) {
  const url = `${baseUrl}?page=${page}`;
  return useFetch<T>(url);
}

// Usage
const { data, loading } = usePagedFetch('/api/users', currentPage);
```
