# Virtualization

[[6. Performance.md]]

---

Virtualization (windowing) renders only visible items in a list. Instead of 10,000 DOM nodes, you might render 20. Essential for large lists and tables.

## Why Virtualize?

| Items | Without Virtualization | With Virtualization |
| :-- | :-- | :-- |
| 100 | Usually fine | Optional |
| 1,000 | Slow, janky scroll | Recommended |
| 10,000+ | Unusable | Essential |

Rendering 10,000 list items creates 10,000 DOM nodes. Virtualization renders only what's in view (plus a small buffer).

## react-window

Lightweight, fixed-size list virtualization.

```bash
npm install react-window
npm install -D @types/react-window
```

### FixedSizeList

```typescript
import { FixedSizeList as List } from 'react-window';

interface Item {
  id: string;
  name: string;
}

function VirtualList({ items }: { items: Item[] }) {
  const Row = ({ index, style }: { index: number; style: React.CSSProperties }) => (
    <div style={style}>
      {items[index].name}
    </div>
  );

  return (
    <List
      height={400}
      itemCount={items.length}
      itemSize={50}
      width="100%"
    >
      {Row}
    </List>
  );
}
```

### VariableSizeList

When items have different heights:

```typescript
import { VariableSizeList as List } from 'react-window';

function VariableList({ items }: { items: Item[] }) {
  const getItemSize = (index: number) => {
    return items[index].expanded ? 120 : 50;
  };

  const Row = ({ index, style }: { index: number; style: React.CSSProperties }) => (
    <div style={style}>
      <ItemRow item={items[index]} />
    </div>
  );

  return (
    <List
      height={400}
      itemCount={items.length}
      itemSize={getItemSize}
      width="100%"
    >
      {Row}
    </List>
  );
}
```

### Grid

```typescript
import { FixedSizeGrid as Grid } from 'react-window';

function VirtualGrid({ items }: { items: Item[] }) {
  const Cell = ({ columnIndex, rowIndex, style }: {
    columnIndex: number;
    rowIndex: number;
    style: React.CSSProperties;
  }) => {
    const index = rowIndex * 4 + columnIndex;
    if (index >= items.length) return null;
    return (
      <div style={style}>
        {items[index].name}
      </div>
    );
  };

  return (
    <Grid
      columnCount={4}
      columnWidth={150}
      height={400}
      rowCount={Math.ceil(items.length / 4)}
      rowHeight={80}
      width={600}
    >
      {Cell}
    </Grid>
  );
}
```

## react-virtuoso

More flexible, handles dynamic heights and infinite scroll out of the box.

```bash
npm install react-virtuoso
```

### Basic List

```typescript
import { Virtuoso } from 'react-virtuoso';

function VirtuosoList({ items }: { items: Item[] }) {
  return (
    <Virtuoso
      style={{ height: 400 }}
      data={items}
      itemContent={(index, item) => (
        <div className="list-item">
          {item.name}
        </div>
      )}
    />
  );
}
```

### Infinite Scroll

```typescript
import { Virtuoso } from 'react-virtuoso';

function InfiniteList() {
  const [items, setItems] = useState<Item[]>([]);
  const [loading, setLoading] = useState(false);

  const loadMore = useCallback(async () => {
    setLoading(true);
    const newItems = await fetchItems(items.length, 20);
    setItems(prev => [...prev, ...newItems]);
    setLoading(false);
  }, [items.length]);

  return (
    <Virtuoso
      style={{ height: 400 }}
      data={items}
      endReached={loadMore}
      itemContent={(index, item) => <div>{item.name}</div>}
      components={{
        Footer: () => loading ? <Spinner /> : null,
      }}
    />
  );
}
```

## Virtual Scrolling Concepts

### Windowing

Only items in the "window" (viewport + overscan) are mounted. As you scroll, items are unmounted and new ones mounted.

```
Viewport
┌─────────────────┐
│  Item 1         │  ← Rendered
│  Item 2         │  ← Rendered
│  Item 3         │  ← Rendered
│  ...            │
│  Item 20        │  ← Rendered
└─────────────────┘
  Item 21–10000   ← Not in DOM
```

### Overscan

Render a few items above/below the viewport to reduce blank space during fast scrolling.

```typescript
<List
  overscanCount={5}  // Render 5 extra items above/below
  // ...
/>
```

## Infinite Scroll Implementation

Without a library, basic infinite scroll:

```typescript
function useInfiniteScroll(
  loadMore: () => Promise<void>,
  hasMore: boolean,
  loading: boolean
) {
  const sentinelRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    if (!hasMore || loading) return;

    const observer = new IntersectionObserver(
      (entries) => {
        if (entries[0].isIntersecting) {
          loadMore();
        }
      },
      { threshold: 0.1 }
    );

    if (sentinelRef.current) {
      observer.observe(sentinelRef.current);
    }
    return () => observer.disconnect();
  }, [loadMore, hasMore, loading]);

  return sentinelRef;
}

function InfiniteList() {
  const [items, setItems] = useState<Item[]>([]);
  const [page, setPage] = useState(0);
  const [loading, setLoading] = useState(false);
  const [hasMore, setHasMore] = useState(true);

  const loadMore = useCallback(async () => {
    setLoading(true);
    const newItems = await fetchItems(page);
    setItems(prev => [...prev, ...newItems]);
    setHasMore(newItems.length === 20);
    setPage(p => p + 1);
    setLoading(false);
  }, [page]);

  const sentinelRef = useInfiniteScroll(loadMore, hasMore, loading);

  return (
    <div>
      {items.map(item => (
        <div key={item.id}>{item.name}</div>
      ))}
      <div ref={sentinelRef} style={{ height: 1 }} />
    </div>
  );
}
```

## Choosing a Library

| Library | Use Case | Pros | Cons |
| :-- | :-- | :-- | :-- |
| **react-window** | Fixed/variable lists, grids | Small, fast, simple | No built-in infinite scroll |
| **react-virtuoso** | Dynamic heights, tables, infinite | Flexible, great DX | Larger bundle |
| **@tanstack/react-virtual** | Headless, full control | Framework-agnostic, flexible | More setup |
